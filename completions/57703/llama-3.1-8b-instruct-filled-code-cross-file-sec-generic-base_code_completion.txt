if (is_generational(gc)) {
  /* clear all the old objects back to young */
  clear_all_old(mrb, gc);
  gc->full = TRUE;
  gc->generational = FALSE; /* temporarily disable generational mode */
  incremental_gc_sweep(mrb, gc); /* perform incremental sweep */
  incremental_gc_finish(mrb, gc); /* finish the sweep */
  gc->generational = TRUE; /* re-enable generational mode */
  gc->full = FALSE;
}
else {
  /* finish half baked GC cycle */
  incremental_gc_finish(mrb, gc);
}
