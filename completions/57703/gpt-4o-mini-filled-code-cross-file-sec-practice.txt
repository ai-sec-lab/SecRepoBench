// Disable generational mode temporarily
  gc->generational = FALSE;

  // Perform the incremental sweep to transition all live objects to white
  size_t limit = gc->live; // Set limit to the number of live objects
  size_t swept = incremental_sweep_phase(mrb, gc, limit);

  // Log the number of objects swept
  mrb_log_gc_sweep(mrb, swept);

  // Re-enable generational mode
