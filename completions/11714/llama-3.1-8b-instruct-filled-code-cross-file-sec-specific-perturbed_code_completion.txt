size_t blend_count = interp_env.get_arg_count();
if (blend_count > 0) {
    size_t default_blend_count = interp_env.get_default_blend_count();
    if (blend_count > default_blend_count) {
        // Handle error: blend count exceeds default blend count
        interp_env.push_error("Blend count exceeds default blend count");
        return;
    }
    size_t arg_offset = interp_env.get_arg_offset();
    size_t blend_offset = interp_env.get_blend_offset();
    size_t default_blend_offset = interp_env.get_default_blend_offset();
    for (size_t i = 0; i < blend_count; i++) {
        interp_env.transfer_arg_to_blend(arg_offset + i * 2, blend_offset + i * 2);
    }
    interp_env.remove_args(blend_count);
}
