    int region_count = param.arg_stack.pop();
    int numeric_value = param.arg_stack.pop();

    // Initialize the blend array with default values.
    std::vector<int> blend_array(region_count, numeric_value);

    // Transfer blend values from the argument stack to the blend array.
    for (int i = 0; i < region_count; ++i) {
        blend_array[i] = param.arg_stack.pop();
    }

    // Apply blend modifications to the interp_env.
    for (int i = 0; i < region_count; ++i) {
        interp_env.blend_values[i] = blend_array[i];
    }

    // Remove the blend values from the argument stack.
    for (int i = 0; i < region_count + 1; ++i) {
        param.arg_stack.pop();
