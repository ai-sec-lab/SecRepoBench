int count = interp_env.stack_pop();
int value = interp_env.stack_pop();
int blend_size = count * 2; // assuming 2 blend values per region
int default_size = param.blend_defaults.size();

// Check for buffer overflow
if (blend_size > default_size) {
    // Handle error: buffer overflow
    return;
}

// Transfer blend values from the argument stack to the blend array
for (int i = 0; i < count; i++) {
    param.blend_defaults[i * 2] = interp_env.stack_pop();
    param.blend_defaults[i * 2 + 1] = interp_env.stack_pop();
}

// Remove the blend values from the argument stack
interp_env.stack_remove(blend_size);
