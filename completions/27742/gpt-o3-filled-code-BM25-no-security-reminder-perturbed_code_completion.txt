/*
         * Each incoming record must be an array with two elements:
         *
         *   [ <timestamp> , <map> ]
         *
         * If thatâ€™s not the case just skip it.
         */
        parsed_object = result.data;
        if (parsed_object.type != MSGPACK_OBJECT_ARRAY ||
            parsed_object.via.array.size != 2) {
            /* Not the format we expect, skip */
            continue;
        }

        /* Extract timestamp and the data map                                             *
         *   element[0] -> timestamp                                                     *
         *   element[1] -> record(map)                                                   */
        obj = parsed_object.via.array.ptr;

        /* Unpack the time. The helper will fill the flb_time structure 'tms'. */
        if (flb_time_pop_from_msgpack(&tms, obj, &map) != 0) {
            /* If time cannot be decoded, skip this record */
            continue;
        }

        /* The second element MUST be a map                                            */
        if (map.type != MSGPACK_OBJECT_MAP) {
            continue;
        }

        /* Map size and the adjusted size if we will append the date_key                */
        map_size = map.via.map.size;
        if (date_key != NULL) {
            msgpack_pack_map(&tmp_pck, map_size + 1);
        }
        else {
            msgpack_pack_map(&tmp_pck, map_size);
        }
