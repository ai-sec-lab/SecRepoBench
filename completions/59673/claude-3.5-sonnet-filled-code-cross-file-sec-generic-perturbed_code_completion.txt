linesize   = s->current_picture.f->linesize[0] << lowres;
uvlinesize = s->current_picture.f->linesize[1] << lowres;

mx = motion_x >> (lowres + 1);
my = motion_y >> (lowres + 1);

sourcex = s->mb_x * 2 * block_s + (mx >> 2);
src_y   = (mb_y * 2 * block_s >> field_based) + (my >> 2);
uvsourcex = s->mb_x * block_s + (mx >> 3);
uvsrc_y   = (mb_y * block_s >> field_based) + (my >> 3);
sx = mx & 3;
sy = my & 3;
uvsx = (mx >> 1) & 1;
uvsy = (my >> 1) & 1;

ptr_y  = ref_picture[0] + src_y * linesize + sourcex;
ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsourcex;
ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsourcex;

if ((unsigned)sourcex > h_edge_pos - (!!sx) - 2 * block_s ||
    (unsigned)src_y   > v_edge_pos - (!!sy) - h) {
    s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                             linesize, linesize,
                             17, 17 + field_based,
                             sourcex, src_y,
                             h_edge_pos, v_edge_pos);
    ptr_y = s->sc.edge_emu_buffer;
}
if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
    uint8_t *uvbuf = s->sc.edge_emu_buffer + 18 * linesize;
    if ((unsigned)uvsourcex > h_edge_pos - (!!uvsx) - block_s ||
        (unsigned)uvsrc_y   > v_edge_pos - (!!uvsy) - h) {
        s->vdsp.emulated_edge_mc(uvbuf, ptr_cb,
                                 uvlinesize, uvlinesize,
                                 9, 9 + field_based,
                                 uvsourcex, uvsrc_y,
                                 h_edge_pos, v_edge_pos);
        ptr_cb = uvbuf;
        uvbuf += 16 * uvlinesize;
        s->vdsp.emulated_edge_mc(uvbuf, ptr_cr,
                                 uvlinesize, uvlinesize,
                                 9, 9 + field_based,
                                 uvsourcex, uvsrc_y,
                                 h_edge_pos, v_edge_pos);
        ptr_cr = uvbuf;
    }
}

if (bottom_field) {
    dest_y  += s->linesize;
    dest_cb += s->uvlinesize;
    dest_cr += s->uvlinesize;
}

if (field_select) {
    ptr_y   += linesize;
    ptr_cb  += uvlinesize;
    ptr_cr  += uvlinesize;
}

sx = (sx << 2) >> lowres;
sy = (sy << 2) >> lowres;
uvsx = (uvsx << 2) >> lowres;
uvsy = (uvsy << 2) >> lowres;

s->hdsp.put_pixels_tab[lowres][0](dest_y, ptr_y, linesize, h);
s->hdsp.put_pixels_tab[lowres][1](dest_y + block_s, ptr_y + block_s, linesize, h);
s->hdsp.put_pixels_tab[lowres][1](dest_y + h * linesize, ptr_y + h * linesize, linesize, h);
s->hdsp.put_pixels_tab[lowres][0](dest_y + h * linesize + block_s, ptr_y + h * linesize + block_s, linesize, h);

if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
    pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> s->chroma_y_shift);
    pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> s->chroma_y_shift);
}
