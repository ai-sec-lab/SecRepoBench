linesize   = s->current_picture.f->linesize[0] << lowres;
uvlinesize = s->current_picture.f->linesize[1] << lowres;

mx = motion_x >> (lowres + 1);
my = motion_y >> (lowres + 1);

sx = motion_x & s_mask;
sy = motion_y & s_mask;

src_x = s->mb_x * 2 * block_s + (mx);
src_y = (mb_y * 2 * block_s >> field_based) + (my >> field_based);

if (   (unsigned)src_x > FFMAX(h_edge_pos - (!!sx) - 2 * block_s, 0)
    || (unsigned)src_y > FFMAX(v_edge_pos - (!!sy) - h, 0)) {
    s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                             linesize, linesize,
                             17, 17 + field_based,
                             src_x, src_y << field_based,
                             h_edge_pos, v_edge_pos);
    ptr_y = s->sc.edge_emu_buffer;
}

sx = (sx << 2) >> lowres;
sy = (sy << 2) >> lowres;
pix_op[lowres - 1](dest_y, ptr_y, linesize, h, sx, sy);

if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
    uvsrc_x = src_x >> 1;
    uvsrc_y = src_y >> 1;
    uvsx = (motion_x << 1) & s_mask;
    uvsy = (motion_y << 1) & s_mask;
    if (   (unsigned)uvsrc_x > FFMAX(h_edge_pos - (!!uvsx) - block_s, 0)
        || (unsigned)uvsrc_y > FFMAX(v_edge_pos - (!!uvsy) - (h >> 1), 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb,
                                 uvlinesize, uvlinesize,
                                 9, 9 + field_based,
                                 uvsrc_x, uvsrc_y << field_based,
                                 h_edge_pos, v_edge_pos);
        ptr_cb = s->sc.edge_emu_buffer;
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer + 16, ptr_cr,
                                 uvlinesize, uvlinesize,
                                 9, 9 + field_based,
                                 uvsrc_x, uvsrc_y << field_based,
                                 h_edge_pos, v_edge_pos);
        ptr_cr = s->sc.edge_emu_buffer + 16;
    }
    uvsx = (uvsx << 2) >> lowres;
    uvsy = (uvsy << 2) >> lowres;
    pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> 1, uvsx, uvsy);
    pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> 1, uvsx, uvsy);
}
