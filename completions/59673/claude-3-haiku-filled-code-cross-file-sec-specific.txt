mx = motion_x & s_mask;
my = motion_y & s_mask;
src_x = s->mb_x * block_s + (motion_x >> lowres + 1);
src_y = mb_y * block_s + (motion_y >> lowres + 1);

ptrdiff_t offset = src_y * linesize + src_x;
ptr_y = ref_picture[0] + offset;
if ((unsigned)src_x > FFMAX(h_edge_pos - (!!mx) - block_s, 0) ||
    (unsigned)src_y > FFMAX(v_edge_pos - (!!my) - block_s, 0)) {
    s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                             linesize, linesize,
                             block_s, block_s,
                             src_x, src_y, h_edge_pos, v_edge_pos);
    ptr_y = s->sc.edge_emu_buffer;
}
mx = (mx << 2) >> lowres;
my = (my << 2) >> lowres;
pix_op[op_index](dest_y, ptr_y, linesize, block_s, mx, my);

uvsrc_x = s->mb_x * block_s + (motion_x >> lowres + s->chroma_x_shift + 1);
uvsrc_y = mb_y * block_s + (motion_y >> lowres + s->chroma_y_shift + 1);
offset = uvsrc_y * uvlinesize + uvsrc_x;
ptr_cb = ref_picture[1] + offset;
ptr_cr = ref_picture[2] + offset;
if ((unsigned)uvsrc_x > FFMAX(h_edge_pos - (!!mx) - block_s, 0) ||
    (unsigned)uvsrc_y > FFMAX(v_edge_pos - (!!my) - block_s, 0)) {
    s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb,
                             uvlinesize, uvlinesize,
                             block_s, block_s,
                             uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
    ptr_cb = s->sc.edge_emu_buffer;
    s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr,
                             uvlinesize, uvlinesize,
                             block_s, block_s,
                             uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
    ptr_cr = s->sc.edge_emu_buffer;
}
mx = (mx << 2) >> lowres;
my = (my << 2) >> lowres;
pix_op[op_index](dest_cb, ptr_cb, uvlinesize, block_s, mx, my);
pix_op[op_index](dest_cr, ptr_cr, uvlinesize, block_s, mx, my);
