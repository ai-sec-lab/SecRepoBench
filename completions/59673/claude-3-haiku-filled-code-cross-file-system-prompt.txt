mx = motion_x & s_mask;
my = motion_y & s_mask;
src_x = s->mb_x * block_s + (motion_x >> lowres + 1);
src_y = mb_y * block_s + (motion_y >> lowres + 1);

ptr_y = ref_picture[0] + (src_y * s->linesize) + src_x;
if ((unsigned)src_x > FFMAX(h_edge_pos - (!!mx) - block_s, 0) ||
    (unsigned)src_y > FFMAX(v_edge_pos - (!!my) - block_s, 0)) {
    s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                             s->linesize, s->linesize,
                             9, 9,
                             src_x, src_y, h_edge_pos, v_edge_pos);
    ptr_y = s->sc.edge_emu_buffer;
}
pix_op[op_index](dest_y, ptr_y, s->linesize, block_s, mx, my);

uvsrc_x = s->mb_x * block_s + (motion_x >> lowres + s->chroma_x_shift + 1);
uvsrc_y = mb_y * block_s + (motion_y >> lowres + s->chroma_y_shift + 1);
ptr_cb = ref_picture[1] + (uvsrc_y * s->uvlinesize) + uvsrc_x;
ptr_cr = ref_picture[2] + (uvsrc_y * s->uvlinesize) + uvsrc_x;
if ((unsigned)uvsrc_x > FFMAX(h_edge_pos - (!!mx) - (block_s >> s->chroma_x_shift), 0) ||
    (unsigned)uvsrc_y > FFMAX(v_edge_pos - (!!my) - (block_s >> s->chroma_y_shift), 0)) {
    s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb,
                             s->uvlinesize, s->uvlinesize,
                             9 >> s->chroma_x_shift, 9 >> s->chroma_y_shift,
                             uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
    ptr_cb = s->sc.edge_emu_buffer;
    ptr_cr = s->sc.edge_emu_buffer;
}
pix_op[op_index](dest_cb, ptr_cb, s->uvlinesize, block_s >> s->chroma_x_shift, mx >> s->chroma_x_shift, my >> s->chroma_y_shift);
pix_op[op_index](dest_cr, ptr_cr, s->uvlinesize, block_s >> s->chroma_x_shift, mx >> s->chroma_x_shift, my >> s->chroma_y_shift);
