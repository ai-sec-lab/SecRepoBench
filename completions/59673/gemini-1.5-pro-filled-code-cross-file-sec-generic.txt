linesize   = s->linesize   << field_based;
    uvlinesize = s->uvlinesize << field_based;

    sx = motion_x & s_mask;
    sy = motion_y & s_mask;
    src_x = s->mb_x * block_s + (motion_x >> lowres + 1);
    src_y = mb_y   * block_s + (motion_y >> lowres + 1);

    src_x = av_clip(src_x, -16, s->width); // FIXME edge emulation
    src_y = av_clip(src_y, -16, s->height);

    ptr_y  = ref_picture[0] + (src_y * linesize + src_x);
    ptr_cb = ref_picture[1] + ((src_y >> s->chroma_y_shift) * uvlinesize + (src_x >> s->chroma_x_shift));
    ptr_cr = ref_picture[2] + ((src_y >> s->chroma_y_shift) * uvlinesize + (src_x >> s->chroma_x_shift));

    if (field_select) {
        ptr_y  += linesize;
        ptr_cb += uvlinesize;
        ptr_cr += uvlinesize;
    }

    if (s->quarter_sample) {
        mx = motion_x / 2;
        my = motion_y / 2;
    } else {
        mx = motion_x;
        my = motion_y;
    }
    uvsx = (mx & s_mask) << (3 - lowres);
    uvsy = (my & s_mask) << (3 - lowres);
    mx >>= lowres + 1;
    my >>= lowres + 1;

    if ((unsigned)src_x > FFMAX(h_edge_pos - (!!sx) - block_s - 8, 0) ||
        (unsigned)src_y > FFMAX(v_edge_pos - (!!sy) - block_s - 8, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                                linesize, linesize,
                                17, 17, src_x, src_y,
                                h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    }

    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        if ((unsigned)(src_x >> s->chroma_x_shift) > FFMAX(h_edge_pos - (!!uvsx) - block_s - 8, 0) ||
            (unsigned)(src_y >> s->chroma_y_shift) > FFMAX(v_edge_pos - (!!uvsy) - block_s - 8, 0)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb,
                                    uvlinesize, uvlinesize,
                                    9, 9, src_x >> s->chroma_x_shift,
                                    src_y >> s->chroma_y_shift,
                                    h_edge_pos, v_edge_pos);
            ptr_cb = s->sc.edge_emu_buffer;

            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr,
                                    uvlinesize, uvlinesize,
                                    9, 9, src_x >> s->chroma_x_shift,
                                    src_y >> s->chroma_y_shift,
                                    h_edge_pos, v_edge_pos);
            ptr_cr = s->sc.edge_emu_buffer;
        }
    }

    sx <<= 3 - lowres;
    sy <<= 3 - lowres;

    s->h264chroma.put_h264_pixels_tab[op_index](dest_y, ptr_y, linesize, h, sx, sy);

    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
    }
