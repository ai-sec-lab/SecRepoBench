linesize   = s->current_picture.f->linesize[0] << lowres;
uvlinesize = s->current_picture.f->linesize[1] << lowres;

mx = motion_x >> (lowres + 1);
my = motion_y >> (lowres + 1);

sx = motion_x & s_mask;
sy = motion_y & s_mask;

src_x = s->mb_x * 2 * block_s + (mx);
src_y = (mb_y * 2 * block_s >> field_based) + (my >> field_based);

if (   (unsigned)src_x > FFMAX(h_edge_pos - (!!sx) - 2 * block_s, 0)
    || (unsigned)src_y > FFMAX(v_edge_pos - (!!sy) - h, 0)) {
    s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y - 2 * block_s - 2 * linesize,
                             linesize, linesize,
                             17, 17 + field_based,
                             src_x - 2 * block_s - 2,
                             src_y - 2 * field_based,
                             h_edge_pos, v_edge_pos);
    ptr_y = s->sc.edge_emu_buffer + 2 * block_s + 2 * linesize;
}

sx = (sx << 2) >> lowres;
sy = (sy << 2) >> lowres;
if (field_select)
    ptr_y += linesize;
s->hdsp.put_pixels_tab[lowres][((sy & 3) << 2) | (sx & 3)](
    dest_y, ptr_y, linesize, h);

if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
    uvsrc_x = src_x >> 1;
    uvsrc_y = src_y >> 1;
    uvsx = (2 * sx) >> (lowres + 1);
    uvsy = (2 * sy) >> (lowres + 1);
    if (field_select)
        ptr_cb += s->current_picture.f->linesize[1];
    if (field_select)
        ptr_cr += s->current_picture.f->linesize[2];
    pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> s->chroma_y_shift);
    pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> s->chroma_y_shift);
}
