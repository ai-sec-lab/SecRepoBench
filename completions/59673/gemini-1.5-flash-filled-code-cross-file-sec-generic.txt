mx = motion_x & s_mask;
    my = motion_y & s_mask;
    src_x = s->mb_x * block_s + (motion_x >> lowres);
    src_y = mb_y * block_s + (motion_y >> lowres);
    uvsrc_x = src_x >> s->chroma_x_shift;
    uvsrc_y = src_y >> s->chroma_y_shift;
    sx = (mx << 2) >> lowres;
    sy = (my << 2) >> lowres;
    uvsx = sx >> s->chroma_x_shift;
    uvsy = sy >> s->chroma_y_shift;
    linesize = s->linesize;
    uvlinesize = s->uvlinesize;
    ptr_y = ref_picture[0] + src_y * linesize + src_x;
    ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x;
    ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x;
    if ((unsigned)src_x > h_edge_pos - (!!mx) - block_s ||
        (unsigned)src_y > v_edge_pos - (!!my) - h) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y, linesize, linesize, h, h, src_x, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
        if (s->chroma_x_shift <= 1) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer + 16, ptr_cb, uvlinesize, uvlinesize, h >> s->chroma_y_shift, h >> s->chroma_y_shift, uvsrc_x, uvsrc_y, h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
            ptr_cb = s->sc.edge_emu_buffer + 16;
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer + 16 + 16, ptr_cr, uvlinesize, uvlinesize, h >> s->chroma_y_shift, h >> s->chroma_y_shift, uvsrc_x, uvsrc_y, h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
            ptr_cr = s->sc.edge_emu_buffer + 16 + 16;
        }
    }
    pix_op[op_index](dest_y, ptr_y, linesize, h, sx, sy);
    if (s->chroma_x_shift <= 1)
        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
    if (s->chroma_x_shift <= 1)
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
