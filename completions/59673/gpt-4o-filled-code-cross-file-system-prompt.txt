if (s->quarter_sample) {
        motion_x /= 2;
        motion_y /= 2;
    }

    mx = ff_h263_round_chroma(motion_x);
    my = ff_h263_round_chroma(motion_y);

    sx = mx & s_mask;
    sy = my & s_mask;
    src_x = s->mb_x * block_s + (mx >> (lowres + 1));
    src_y = mb_y * block_s + (my >> (lowres + 1));

    linesize = s->linesize >> lowres;
    uvlinesize = s->uvlinesize >> lowres;

    ptr_y = ref_picture[0] + src_y * linesize + src_x;
    ptr_cb = ref_picture[1] + (src_y >> s->chroma_y_shift) * uvlinesize + (src_x >> s->chroma_x_shift);
    ptr_cr = ref_picture[2] + (src_y >> s->chroma_y_shift) * uvlinesize + (src_x >> s->chroma_x_shift);

    if ((unsigned)src_x > FFMAX(h_edge_pos - (!!sx) - block_s, 0) ||
        (unsigned)src_y > FFMAX(v_edge_pos - (!!sy) - h, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y, linesize, linesize, block_s, h, src_x, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    }

    sx = (sx << 2) >> lowres;
    sy = (sy << 2) >> lowres;
    s->dsp.put_pixels_tab[0][lowres](dest_y, ptr_y, linesize, h, sx, sy);

    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        if ((unsigned)src_x > FFMAX(h_edge_pos - (!!sx) - block_s, 0) ||
            (unsigned)src_y > FFMAX(v_edge_pos - (!!sy) - h, 0)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb, uvlinesize, uvlinesize, block_s, h >> s->chroma_y_shift, src_x >> s->chroma_x_shift, src_y >> s->chroma_y_shift, h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
            ptr_cb = s->sc.edge_emu_buffer;
        }
        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, block_s, sx, sy);

        if ((unsigned)src_x > FFMAX(h_edge_pos - (!!sx) - block_s, 0) ||
            (unsigned)src_y > FFMAX(v_edge_pos - (!!sy) - h, 0)) {
            s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cr, uvlinesize, uvlinesize, block_s, h >> s->chroma_y_shift, src_x >> s->chroma_x_shift, src_y >> s->chroma_y_shift, h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
            ptr_cr = s->sc.edge_emu_buffer;
        }
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, block_s, sx, sy);
    }
