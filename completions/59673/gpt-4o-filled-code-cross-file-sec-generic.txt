mx = motion_x >> lowres;
    my = motion_y >> lowres;

    if (s->quarter_sample) {
        mx /= 2;
        my /= 2;
    }

    sx = mx & s_mask;
    sy = my & s_mask;
    src_x = s->mb_x * block_s + (mx >> lowres + 1);
    src_y = mb_y * block_s + (my >> lowres + 1);

    uvsrc_x = s->mb_x * (block_s >> s->chroma_x_shift) + (mx >> (lowres + 1 + s->chroma_x_shift));
    uvsrc_y = mb_y * (block_s >> s->chroma_y_shift) + (my >> (lowres + 1 + s->chroma_y_shift));

    uvsx = (sx << 2) >> (lowres + s->chroma_x_shift);
    uvsy = (sy << 2) >> (lowres + s->chroma_y_shift);

    linesize = s->linesize >> lowres;
    uvlinesize = s->uvlinesize >> lowres;

    ptr_y = ref_picture[0] + src_y * linesize + src_x;
    ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x;
    ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x;

    if ((unsigned)src_x > FFMAX(h_edge_pos - (!!sx) - block_s, 0) ||
        (unsigned)src_y > FFMAX(v_edge_pos - (!!sy) - block_s, 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y, linesize, linesize, block_s + 1, block_s + 1, src_x, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    }

    if ((unsigned)uvsrc_x > FFMAX((h_edge_pos >> s->chroma_x_shift) - (!!uvsx) - (block_s >> s->chroma_x_shift), 0) ||
        (unsigned)uvsrc_y > FFMAX((v_edge_pos >> s->chroma_y_shift) - (!!uvsy) - (block_s >> s->chroma_y_shift), 0)) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_cb, uvlinesize, uvlinesize, (block_s >> s->chroma_x_shift) + 1, (block_s >> s->chroma_y_shift) + 1, uvsrc_x, uvsrc_y, h_edge_pos >> s->chroma_x_shift, v_edge_pos >> s->chroma_y_shift);
        ptr_cb = s->sc.edge_emu_buffer;
        ptr_cr = s->sc.edge_emu_buffer + 16;
    }

    s->dsp.put_pixels_tab[0][0](dest_y, ptr_y, linesize, block_s);
    pix_op[op_index](dest_cb, ptr_cb, uvlinesize, block_s >> s->chroma_y_shift, uvsx, uvsy);
    pix_op[op_index](dest_cr, ptr_cr, uvlinesize, block_s >> s->chroma_y_shift, uvsx, uvsy);
