linesize   = s->linesize   << field_based;
    uvlinesize = s->uvlinesize << field_based;

    sx = motion_x & s_mask;
    sy = motion_y & s_mask;
    src_x = s->mb_x * block_s + (motion_x >> lowres + 1);
    src_y = mb_y   * block_s + (motion_y >> lowres + 1);

    src_x = av_clip(src_x, -16, h_edge_pos + block_s); // avoid overreads
    src_y = av_clip(src_y, -16, v_edge_pos + block_s);

    ptr_y  = ref_picture[0] + (src_y * linesize + src_x);
    ptr_cb = ref_picture[1] + ((src_y >> s->chroma_y_shift) * uvlinesize + (src_x >> s->chroma_x_shift));
    ptr_cr = ref_picture[2] + ((src_y >> s->chroma_y_shift) * uvlinesize + (src_x >> s->chroma_x_shift));

    if (field_select) {
        ptr_y  += linesize;
        ptr_cb += uvlinesize;
        ptr_cr += uvlinesize;
    }
    if (bottom_field) { //FIXME merge with field select
        ptr_y  += s->linesize;
        ptr_cb += s->uvlinesize;
        ptr_cr += s->uvlinesize;
    }


    sx = (sx << 2) >> lowres;
    sy = (sy << 2) >> lowres;

    s->vdsp.put_pixels_tab[0][op_index](dest_y, ptr_y, linesize, h);

    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> s->chroma_y_shift, sx, sy);
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> s->chroma_y_shift, sx, sy);
    }
