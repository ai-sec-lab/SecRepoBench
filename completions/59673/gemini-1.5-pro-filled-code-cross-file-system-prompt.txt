linesize   = s->linesize   << field_based;
    uvlinesize = s->uvlinesize << field_based;

    if (s->quarter_sample) {
        motion_x /= 2;
        motion_y /= 2;
    }

    sx = motion_x & s_mask;
    sy = motion_y & s_mask;
    src_x = s->mb_x * block_s + (motion_x >> lowres + 1);
    src_y = mb_y   * block_s + (motion_y >> lowres + 1);

    src_x = av_clip(src_x, -16, (s->width  >> lowres) + 15); // FIXME
    if (field_select)
        src_y += s->height >> lowres + 1;
    src_y = av_clip(src_y, -16, (s->height >> lowres) + 15);

    ptr_y  = ref_picture[0] + src_y * linesize + src_x;
    ptr_cb = ref_picture[1] + (src_y >> s->chroma_y_shift) * uvlinesize + (src_x >> s->chroma_x_shift);
    ptr_cr = ref_picture[2] + (src_y >> s->chroma_y_shift) * uvlinesize + (src_x >> s->chroma_x_shift);

    if (field_based) {
        ptr_y  += bottom_field * linesize;
        ptr_cb += bottom_field * uvlinesize;
        ptr_cr += bottom_field * uvlinesize;
    }

    if ((unsigned)src_x > FFMAX(h_edge_pos - (!!sx) - block_s, 0) ||
        (unsigned)src_y > FFMAX(v_edge_pos - (!!sy) - h, 0)) {
        uint8_t *uvbuf = s->sc.edge_emu_buffer;

        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ptr_y,
                                 linesize, linesize,
                                 17, 17 + field_based,
                                 src_x, src_y << field_based,
                                 h_edge_pos, v_edge_pos << field_based);
        ptr_y = s->sc.edge_emu_buffer;

        if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
            src_x >>= s->chroma_x_shift;
            if (field_select)
                src_y += s->height >> lowres + 1;
            src_y >>= s->chroma_y_shift;

            s->vdsp.emulated_edge_mc(uvbuf, ptr_cb,
                                     s->uvlinesize, s->uvlinesize,
                                     9, 9 + field_based,
                                     src_x, src_y << field_based,
                                     h_edge_pos >> s->chroma_x_shift,
                                     v_edge_pos >> s->chroma_y_shift << field_based);
            ptr_cb = uvbuf;

            s->vdsp.emulated_edge_mc(uvbuf, ptr_cr,
                                     s->uvlinesize, s->uvlinesize,
                                     9, 9 + field_based,
                                     src_x, src_y << field_based,
                                     h_edge_pos >> s->chroma_x_shift,
                                     v_edge_pos >> s->chroma_y_shift << field_based);
            ptr_cr = uvbuf;
        }
    }

    uvsx = (sx << 2) >> lowres;
    uvsy = (sy << 2) >> lowres;
    sx   = (sx << 2) >> lowres;
    sy   = (sy << 2) >> lowres;

    if (!field_based)
        s->hdsp.put_pixels_tab[0][op_index](dest_y, ptr_y, linesize, h);
    else {
        s->hdsp.put_pixels_tab[1][op_index](dest_y, ptr_y, linesize, h >> 1);
        dest_y += linesize;
        s->hdsp.put_pixels_tab[1][op_index](dest_y, ptr_y + linesize, linesize, (h + 1) >> 1);
    }

    if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY)) {
        pix_op[op_index](dest_cb, ptr_cb, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
        pix_op[op_index](dest_cr, ptr_cr, uvlinesize, h >> s->chroma_y_shift, uvsx, uvsy);
    }
