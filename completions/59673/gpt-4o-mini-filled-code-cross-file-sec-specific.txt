// Calculate source positions for luma and chroma
    src_x = s->mb_x * block_s + (motion_x >> lowres);
    src_y = s->mb_y * block_s + (motion_y >> lowres);
    uvsrc_x = src_x >> s->chroma_x_shift;
    uvsrc_y = src_y >> s->chroma_y_shift;

    // Ensure we do not exceed the boundaries
    if ((unsigned)src_x >= h_edge_pos || (unsigned)src_y >= v_edge_pos) {
        // Handle edge cases with emulated edges
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ref_picture[0] + src_y * linesize + src_x,
                                  linesize, linesize, 9, 9, src_x, src_y, h_edge_pos, v_edge_pos);
        ptr_y = s->sc.edge_emu_buffer;
    } else {
        ptr_y = ref_picture[0] + src_y * linesize + src_x;
    }

    // Perform motion compensation for luma
    pix_op[op_index](dest_y, ptr_y, linesize, block_s, motion_x & s_mask, motion_y & s_mask);

    // Handle chroma components
    if (field_based) {
        // Adjust for field-based motion compensation
        dest_cb += (mb_y * s->uvlinesize);
        dest_cr += (mb_y * s->uvlinesize);
    }

    // Ensure we do not exceed the boundaries for chroma
    if ((unsigned)uvsrc_x >= h_edge_pos || (unsigned)uvsrc_y >= v_edge_pos) {
        s->vdsp.emulated_edge_mc(s->sc.edge_emu_buffer, ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x,
                                  uvlinesize, uvlinesize, 9, 9, uvsrc_x, uvsrc_y, h_edge_pos, v_edge_pos);
        ptr_cb = s->sc.edge_emu_buffer;
        ptr_cr = s->sc.edge_emu_buffer;
    } else {
        ptr_cb = ref_picture[1] + uvsrc_y * uvlinesize + uvsrc_x;
        ptr_cr = ref_picture[2] + uvsrc_y * uvlinesize + uvsrc_x;
    }

    // Perform motion compensation for chroma
    pix_op[op_index](dest_cb, ptr_cb, uvlinesize, block_s, motion_x & s_mask, motion_y & s_mask);
    pix_op[op_index](dest_cr, ptr_cr, uvlinesize, block_s, motion_x & s_mask, motion_y & s_mask);
