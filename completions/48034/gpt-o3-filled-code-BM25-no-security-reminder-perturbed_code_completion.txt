/* Extract response type */
	  responsetype = get16(&x, packet->payload);

	  /* Skip the CLASS field (2 bytes) */
	  x += 2;

	  /* Extract the TTL (Time-To-Live) */
	  rsp_ttl = get32(&x, packet->payload);

#ifdef DNS_DEBUG
	  NDPI_LOG_DBG2(ndpi_struct,
			"[DNS] response type: %u, ttl: %u\n",
			responsetype, rsp_ttl);
	  printf("[DNS] [reply] response_type=%u ttl=%u\n",
		 responsetype, rsp_ttl);
#endif

	  /* TTL equal to zero is often associated with suspicious / malformed answers */
	  if(rsp_ttl == 0) {
	    ndpi_set_risk(ndpi_struct, flow,
			  NDPI_MALFORMED_PACKET,
			  "DNS answer with TTL equal to 0");
	  }
