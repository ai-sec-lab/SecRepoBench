for (++iterator; iterator; ++iterator)
{
  hb_codepoint_t gid = (*iterator).first;
  unsigned klass = (*iterator).second;

  if (gid == prev_gid + 1 && klass == prev_klass)
  {
    range_rec.last = gid;
  }
  else
  {
    range_rec.last = prev_gid;
    record->last = range_rec.last;

    range_rec.first = gid;
    range_rec.last = gid;
    range_rec.value = klass;

    record = c->copy (range_rec);
    if (unlikely (!record)) return_trace (false);

    num_ranges++;
  }

  prev_gid = gid;
  prev_klass = klass;
}

record->last = prev_gid;
rangeRecord.len = num_ranges;

// Sort the range records to handle any unsorted glyph order.
// This is a simplification and actual implementation may vary based on the specific requirements.
// For example, it might be more efficient to use a data structure that maintains sorted order.
std::sort (c->range_records, c->range_records + rangeRecord.len);

return_trace (true);
