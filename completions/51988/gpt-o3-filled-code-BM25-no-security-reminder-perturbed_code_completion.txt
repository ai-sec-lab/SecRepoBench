// basic sanity checks
  if (!word || index < 0)
    return 0;

  int wlen = static_cast<int>(strlen(word));
  if (index + 1 >= wlen)  // need two consecutive characters
    return 0;

  unsigned char c1 = static_cast<unsigned char>(word[index]);
  unsigned char c2 = static_cast<unsigned char>(word[index + 1]);

  // hyphens are not considered
  if (c1 == '-' || c2 == '-')
    return 0;

  // check if either character is already uppercase
  if (std::toupper(c1) == c1 || std::toupper(c2) == c2)
    return 1;
