#if !defined(NDEBUG)
		// in debug builds, stop decoding when we hit numChannels
		if (channelIndex >= numChannels)
			goto NoMoreChannels ;
#else
		// in production builds, stop decoding when there are no more bits to read
		// - this is to avoid reading bits that do not belong to audio data
		// - if we don't have this check, we might read bits that are not part of the audio data
		// - this could lead to unexpected behavior or crashes
		if (bits->cur >= bits->end)
			goto NoMoreChannels ;
#endif
