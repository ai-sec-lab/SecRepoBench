CASE(OP_EXT1, Z) {
  FETCH_B();
  switch (b) {
    case 0x00: // OP_SEND_SYM
      FETCH_BB();
      c = 0;
      mid = syms[b];
      goto L_SEND_SYM;
    case 0x01: // OP_SEND_SYM_V
      FETCH_BB();
      c = CALL_MAXARGS;
      mid = syms[b];
      goto L_SEND_SYM;
    case 0x02: // OP_SEND_SYM_B
      FETCH_BBB();
      c = b;
      mid = syms[a];
      goto L_SEND_SYM;
    case 0x03: // OP_SEND_SYM_V_B
      FETCH_BBB();
      c = b;
      mid = syms[a];
      goto L_SEND_SYM;
    default:
      goto L_STOP;
  }
  pc--;
  JUMP;
}
