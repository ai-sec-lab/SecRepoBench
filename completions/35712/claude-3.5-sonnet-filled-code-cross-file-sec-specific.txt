CASE(OP_EXT1, Z) {
  insn = READ_B();
  switch (insn) {
    CASE(OP_ENTER, W) {
      FETCH_W();
      goto L_ENTER;
    }
    CASE(OP_ARYPUSH, W) {
      FETCH_W();
      mrb_ary_push(mrb, regs[a], regs[a+1]);
      NEXT;
    }
    default:
      break;
  }
  pc--;
  NEXT;
}

CASE(OP_EXT2, Z) {
  insn = READ_B();
  switch (insn) {
    CASE(OP_LAMBDA, BB) {
      FETCH_BB();
      c = OP_L_LAMBDA | OP_L_CAPTURE;
      goto L_MAKE_LAMBDA;
    }
    default:
      break;
  }
  pc--;
  NEXT;
}

CASE(OP_EXT3, Z) {
  insn = READ_B();
  switch (insn) {
    CASE(OP_SEND, BBB) {
      FETCH_BBB();
      goto L_SEND;
    }
    CASE(OP_BLOCK, BB) {
      FETCH_BB();
      c = OP_L_BLOCK | OP_L_CAPTURE;
      goto L_MAKE_LAMBDA;
    }
    default:
      break;
  }
  pc--;
  NEXT;
}
