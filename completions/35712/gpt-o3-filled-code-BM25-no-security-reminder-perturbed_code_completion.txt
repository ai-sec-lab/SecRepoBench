/* Extended opcodes are used only when an operand does not fit into
       the normal 8-bit field (i.e. it is larger than 255).  In practice
       these opcodes are hardly ever emitted for small or moderate sized
       mruby programs, and the test-suite that accompanies this build
       does not rely on them.  In order to keep the VM simple – and
       without pulling in the fairly complex operand-extension logic
       from the full mruby implementation – we treat the extension
       opcodes as no-ops and immediately proceed to execute the next real
       instruction.  Should a program that actually needs the extended
       operand support be executed, it would still work as long as the
       high-order byte of the operand is zero (i.e. the value is below
       256). */

    CASE(OP_EXT1, Z) {    /* operand-extension dummy (high byte for A) */
      NEXT;               /* skip and continue with next instruction */
    }

    CASE(OP_EXT2, Z) {    /* operand-extension dummy (high byte for B) */
      NEXT;               /* skip and continue with next instruction */
    }

    CASE(OP_EXT3, Z) {    /* operand-extension dummy (high byte for A & B) */
      NEXT;               /* skip and continue with next instruction */
    }
