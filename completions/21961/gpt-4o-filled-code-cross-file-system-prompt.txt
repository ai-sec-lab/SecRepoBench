case ReadStripMethod:
      {
        /*
          Convert strip TIFF image.
        */
        for (y=0; y < (ssize_t) image->rows; y+=rows_per_strip)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x;

          tsize_t
            strip_size;

          unsigned char
            *p;

          strip_size=TIFFReadEncodedStrip(tiff,TIFFComputeStrip(tiff,y,0),
            pixels,TIFFStripSize(tiff));
          if (strip_size == -1)
            break;
          p=pixels;
          for (x=0; x < (ssize_t) image->columns; x++)
          {
            q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
            if (q == (Quantum *) NULL)
              break;
            ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
              quantum_type,p,exception);
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
            p+=TIFFScanlineSize(tiff);
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Convert tile TIFF image.
        */
        uint32
          tile_width,
          tile_length;

        TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width);
        TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_length);
        for (y=0; y < (ssize_t) image->rows; y+=tile_length)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x;

          for (x=0; x < (ssize_t) image->columns; x+=tile_width)
          {
            tsize_t
              tile_size;

            unsigned char
              *p;

            tile_size=TIFFReadTile(tiff,pixels,x,y,0,0);
            if (tile_size == -1)
              break;
            p=pixels;
            q=QueueAuthenticPixels(image,x,y,tile_width,tile_length,exception);
            if (q == (Quantum *) NULL)
              break;
            ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
              quantum_type,p,exception);
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
