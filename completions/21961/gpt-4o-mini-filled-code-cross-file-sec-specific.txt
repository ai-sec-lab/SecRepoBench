case ReadStripMethod:
      {
        uint32 strip_size = rows_per_strip * TIFFScanlineSize(tiff);
        MemoryInfo *strip_info = AcquireVirtualMemory(strip_size, sizeof(uint32));
        if (strip_info == (MemoryInfo *) NULL)
          ThrowTIFFException(ResourceLimitError, "MemoryAllocationFailed");

        for (y = 0; y < (ssize_t) image->rows; y += rows_per_strip)
        {
          uint32 current_strip_height = MagickMin(rows_per_strip, image->rows - y);
          tiff_status = TIFFReadEncodedStrip(tiff, y / rows_per_strip, GetVirtualMemoryBlob(strip_info), strip_size);
          if (tiff_status == -1)
            break;

          register Quantum *magick_restrict q = QueueAuthenticPixels(image, 0, y, image->columns, current_strip_height, exception);
          if (q == (Quantum *) NULL)
            break;

          for (ssize_t row = 0; row < current_strip_height; row++)
          {
            uint32 *p = (uint32 *) GetVirtualMemoryBlob(strip_info) + row * (strip_size / sizeof(uint32));
            for (ssize_t x = 0; x < (ssize_t) image->columns; x++)
            {
              SetPixelRed(image, ScaleCharToQuantum((unsigned char) TIFFGetR(p[x])), q);
              SetPixelGreen(image, ScaleCharToQuantum((unsigned char) TIFFGetG(p[x])), q);
              SetPixelBlue(image, ScaleCharToQuantum((unsigned char) TIFFGetB(p[x])), q);
              if (image->alpha_trait != UndefinedPixelTrait)
                SetPixelAlpha(image, ScaleCharToQuantum((unsigned char) TIFFGetA(p[x])), q);
              q += GetPixelChannels(image);
            }
          }

          if (SyncAuthenticPixels(image, exception) == MagickFalse)
            break;

          if (image->previous == (Image *) NULL)
          {
            status = SetImageProgress(image, LoadImageTag, (MagickOffsetType) y, image->rows);
            if (status == MagickFalse)
              break;
          }
        }
        strip_info = RelinquishVirtualMemory(strip_info);
        break;
      }
      case ReadTileMethod:
      {
        uint32 tile_width, tile_height;
        TIFFGetField(tiff, TIFFTAG_TILEWIDTH, &tile_width);
        TIFFGetField(tiff, TIFFTAG_TILELENGTH, &tile_height);
        MemoryInfo *tile_info = AcquireVirtualMemory(tile_width * tile_height, sizeof(uint32));
        if (tile_info == (MemoryInfo *) NULL)
          ThrowTIFFException(ResourceLimitError, "MemoryAllocationFailed");

        for (y = 0; y < (ssize_t) image->rows; y += tile_height)
        {
          for (ssize_t x = 0; x < (ssize_t) image->columns; x += tile_width)
          {
            tiff_status = TIFFReadTile(tiff, GetVirtualMemoryBlob(tile_info), x, y, 0, 0);
            if (tiff_status == -1)
              break;

            register Quantum *magick_restrict q = QueueAuthenticPixels(image, x, y, tile_width, tile_height, exception);
            if (q == (Quantum *) NULL)
              break;

            uint32 *p = (uint32 *) GetVirtualMemoryBlob(tile_info);
            for (ssize_t tile_y = 0; tile_y < tile_height; tile_y++)
            {
              for (ssize_t tile_x = 0; tile_x < tile_width; tile_x++)
              {
                SetPixelRed(image, ScaleCharToQuantum((unsigned char) TIFFGetR(p[tile_y * tile_width + tile_x])), q);
                SetPixelGreen(image, ScaleCharToQuantum((unsigned char) TIFFGetG(p[tile_y * tile_width + tile_x])), q);
                SetPixelBlue(image, ScaleCharToQuantum((unsigned char) TIFFGetB(p[tile_y * tile_width + tile_x])), q);
                if (image->alpha_trait != UndefinedPixelTrait)
                  SetPixelAlpha(image, ScaleCharToQuantum((unsigned char) TIFFGetA(p[tile_y * tile_width + tile_x])), q);
                q += GetPixelChannels(image);
              }
            }

            if (SyncAuthenticPixels(image, exception) == MagickFalse)
              break;

            if (image->previous == (Image *) NULL)
            {
              status = SetImageProgress(image, LoadImageTag, (MagickOffsetType) y, image->rows);
              if (status == MagickFalse)
                break;
            }
          }
        }
        tile_info = RelinquishVirtualMemory(tile_info);
        break;
      }
