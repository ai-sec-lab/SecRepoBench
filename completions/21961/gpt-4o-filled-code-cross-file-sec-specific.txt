case ReadStripMethod:
      {
        /*
          Convert strip TIFF image.
        */
        for (y=0; y < (ssize_t) image->rows; y+=rows_per_strip)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x;

          tsize_t
            strip_size;

          unsigned char
            *strip_pixels;

          strip_size=TIFFStripSize(tiff);
          strip_pixels=(unsigned char *) AcquireQuantumMemory(strip_size,
            sizeof(*strip_pixels));
          if (strip_pixels == (unsigned char *) NULL)
            {
              pixel_info=RelinquishVirtualMemory(pixel_info);
              ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");
            }
          tiff_status=TIFFReadEncodedStrip(tiff,TIFFComputeStrip(tiff,y,0),
            strip_pixels,strip_size);
          if (tiff_status == -1)
            {
              strip_pixels=(unsigned char *) RelinquishMagickMemory(strip_pixels);
              break;
            }
          for (x=0; x < (ssize_t) image->columns; x++)
          {
            q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
            if (q == (Quantum *) NULL)
              break;
            ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
              quantum_type,strip_pixels,exception);
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
          }
          strip_pixels=(unsigned char *) RelinquishMagickMemory(strip_pixels);
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Convert tile TIFF image.
        */
        uint32
          tile_width,
          tile_length;

        tsize_t
          tile_size;

        unsigned char
          *tile_pixels;

        TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width);
        TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_length);
        tile_size=TIFFTileSize(tiff);
        tile_pixels=(unsigned char *) AcquireQuantumMemory(tile_size,
          sizeof(*tile_pixels));
        if (tile_pixels == (unsigned char *) NULL)
          {
            pixel_info=RelinquishVirtualMemory(pixel_info);
            ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");
          }
        for (y=0; y < (ssize_t) image->rows; y+=tile_length)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x;

          for (x=0; x < (ssize_t) image->columns; x+=tile_width)
          {
            tiff_status=TIFFReadTile(tiff,tile_pixels,x,y,0,0);
            if (tiff_status == -1)
              break;
            q=QueueAuthenticPixels(image,x,y,tile_width,tile_length,exception);
            if (q == (Quantum *) NULL)
              break;
            ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
              quantum_type,tile_pixels,exception);
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        tile_pixels=(unsigned char *) RelinquishMagickMemory(tile_pixels);
        break;
      }
