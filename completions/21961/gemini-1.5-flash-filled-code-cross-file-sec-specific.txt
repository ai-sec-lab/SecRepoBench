case ReadStripMethod:
      {
        /*
          Convert strip-based TIFF image.
        */
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x;

          size_t
            length;

          unsigned char
            *p;

          tiff_status=TIFFReadScanline(tiff,pixels,y);
          if (tiff_status == -1)
            break;
          q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
          if (q == (Quantum *) NULL)
            break;
          p=pixels;
          for (x=0; x < (ssize_t) image->columns; x++)
          {
            length=ImportQuantumPixels(image,quantum_info,quantum_type,p,q,
              exception);
            if (length == 0)
              break;
            p+=length;
            q+=GetPixelChannels(image);
          }
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        uint32
          tile_height,
          tile_width;

        /*
          Convert tile-based TIFF image.
        */
        if (TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width) != 1)
          ThrowTIFFException(CoderError,"UnableToReadTileWidth");
        if (TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_height) != 1)
          ThrowTIFFException(CoderError,"UnableToReadTileLength");
        for (y=0; y < (ssize_t) image->rows; y+=tile_height)
        {
          register ssize_t
            x;

          register Quantum
            *magick_restrict q;

          size_t
            length;

          ssize_t
            tile_y;

          unsigned char
            *tile_pixels;

          for (x=0; x < (ssize_t) image->columns; x+=tile_width)
          {
            size_t
              tile_length;

            tile_length=(size_t) TIFFTileSize(tiff);
            tile_pixels=(unsigned char *) AcquireQuantumMemory(tile_length,
              sizeof(*tile_pixels));
            if (tile_pixels == (unsigned char *) NULL)
              ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");
            if (TIFFReadTile(tiff,tile_pixels,x,y,0,0) == -1)
              {
                tile_pixels=(unsigned char *) RelinquishMagickMemory(
                  tile_pixels);
                ThrowTIFFException(CoderError,"UnableToReadTile");
              }
            for (tile_y=0; tile_y < (ssize_t) tile_height; tile_y++)
            {
              register ssize_t
                tile_x;

              q=QueueAuthenticPixels(image,x,y+tile_y,tile_width,1,exception);
              if (q == (Quantum *) NULL)
                break;
              for (tile_x=0; tile_x < (ssize_t) tile_width; tile_x++)
              {
                length=ImportQuantumPixels(image,quantum_info,quantum_type,
                  tile_pixels,q,exception);
                if (length == 0)
                  break;
                tile_pixels+=length;
                q+=GetPixelChannels(image);
              }
              if (SyncAuthenticPixels(image,exception) == MagickFalse)
                break;
            }
            tile_pixels=(unsigned char *) RelinquishMagickMemory(tile_pixels);
            if (image->previous == (Image *) NULL)
              {
                status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                  image->rows);
                if (status == MagickFalse)
                  break;
              }
          }
        }
        break;
      }
