case ReadStripMethod:
      {
        /*
          Convert TIFF image to pixel packets.
        */
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          register Quantum
            *magick_restrict q;

          register ssize_t
            x;

          size_t
            length;

          unsigned char
            *p;

          tiff_status=TIFFReadScanline(tiff,pixels,y);
          if (tiff_status == -1)
            break;
          q=QueueAuthenticPixels(image,0,y,image->columns,1,exception);
          if (q == (Quantum *) NULL)
            break;
          p=pixels;
          for (x=0; x < (ssize_t) image->columns; x++)
          {
            length=ImportQuantumPixels(image,quantum_info,quantum_type,p,q,
              exception);
            if (length == 0)
              break;
            p+=length;
            q+=GetPixelChannels(image);
          }
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        uint32
          tile_height,
          tile_width;

        /*
          Convert tiled TIFF image.
        */
        if (TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width) != 1)
          ThrowTIFFException(CoderError,"UnableToReadTileWidth");
        if (TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_height) != 1)
          ThrowTIFFException(CoderError,"UnableToReadTileLength");
        for (y=0; y < (ssize_t) image->rows; y+=tile_height)
        {
          register ssize_t
            x;

          register Quantum
            *magick_restrict q;

          size_t
            length;

          ssize_t
            tile_y;

          unsigned char
            *tile_pixels;

          uint32
            tile_row;

          /*
            Read tile scanlines.
          */
          tile_y=MagickMin((ssize_t) y+tile_height,(ssize_t) image->rows)-y;
          if (HeapOverflowSanityCheck(tile_width*tile_y,sizeof(*pixels)) !=
              MagickFalse)
            ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");
          number_pixels=(MagickSizeType) tile_width*tile_y;
          pixel_info=AcquireVirtualMemory(number_pixels,sizeof(uint32));
          if (pixel_info == (MemoryInfo *) NULL)
            ThrowTIFFException(ResourceLimitError,"MemoryAllocationFailed");
          tile_pixels=(unsigned char *) GetVirtualMemoryBlob(pixel_info);
          (void) memset(tile_pixels,0,number_pixels*sizeof(uint32));
          for (tile_row=0; tile_row < (uint32) tile_y; tile_row++)
          {
            tiff_status=TIFFReadTile(tiff,tile_pixels,0,y+tile_row,0,0);
            if (tiff_status == -1)
              break;
          }
          q=QueueAuthenticPixels(image,0,y,image->columns,tile_y,exception);
          if (q == (Quantum *) NULL)
            break;
          for (x=0; x < (ssize_t) image->columns; x++)
          {
            for (tile_row=0; tile_row < (uint32) tile_y; tile_row++)
            {
              p=tile_pixels+tile_row*tile_width*
                ((bits_per_sample+7) >> 3)*samples_per_pixel+
                x*((bits_per_sample+7) >> 3)*samples_per_pixel;
              length=ImportQuantumPixels(image,quantum_info,quantum_type,p,q,
                exception);
              if (length == 0)
                break;
              q+=GetPixelChannels(image);
            }
            if (tile_row < (uint32) tile_y)
              break;
          }
          pixel_info=RelinquishVirtualMemory(pixel_info);
          if (SyncAuthenticPixels(image,exception) == MagickFalse)
            break;
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
