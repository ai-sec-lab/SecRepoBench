case ReadStripMethod:
      {
        /*
          Convert strip TIFF image.
        */
        for (y=0; y < (ssize_t) image->rows; y+=rows_per_strip)
        {
          tsize_t
            strip_size;

          strip_size=TIFFReadEncodedStrip(tiff,TIFFComputeStrip(tiff,y,0),
            pixels,TIFFStripSize(tiff));
          if (strip_size == -1)
            break;
          for (ssize_t row=0; row < (ssize_t) rows_per_strip; row++)
          {
            if ((y+row) >= (ssize_t) image->rows)
              break;
            register Quantum
              *magick_restrict q;

            q=QueueAuthenticPixels(image,0,y+row,image->columns,1,exception);
            if (q == (Quantum *) NULL)
              break;
            ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
              quantum_type,pixels+row*TIFFScanlineSize(tiff),exception);
            if (SyncAuthenticPixels(image,exception) == MagickFalse)
              break;
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
      case ReadTileMethod:
      {
        /*
          Convert tile TIFF image.
        */
        uint32
          tile_width,
          tile_length;

        TIFFGetField(tiff,TIFFTAG_TILEWIDTH,&tile_width);
        TIFFGetField(tiff,TIFFTAG_TILELENGTH,&tile_length);
        for (y=0; y < (ssize_t) image->rows; y+=tile_length)
        {
          for (ssize_t x=0; x < (ssize_t) image->columns; x+=tile_width)
          {
            tsize_t
              tile_size;

            tile_size=TIFFReadTile(tiff,pixels,x,y,0,0);
            if (tile_size == -1)
              break;
            for (ssize_t row=0; row < (ssize_t) tile_length; row++)
            {
              if ((y+row) >= (ssize_t) image->rows)
                break;
              register Quantum
                *magick_restrict q;

              q=QueueAuthenticPixels(image,x,y+row,tile_width,1,exception);
              if (q == (Quantum *) NULL)
                break;
              ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
                quantum_type,pixels+row*TIFFTileRowSize(tiff),exception);
              if (SyncAuthenticPixels(image,exception) == MagickFalse)
                break;
            }
          }
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
