/* The token encodes a run of the same byte ``token`` repeated for the
         whole block.  Reconstruct the block by filling the destination buffer
         with this value.  We already consumed one extra byte (the token),
         so account for it in the compressed-bytes counter.  The compressed
         payload for this stream is fully represented by the token, hence we
         reset ``cbytes`` to 0 so that the common footer logic does not move
         the ``src`` pointer any further. */
      ctbytes += 1;                     /* account for the token byte          */
      memset(comp_buffer, token, (size_t)neblock);
      nbytes = neblock;                 /* decompressed bytes in this stream   */
      cbytes = 0;                       /* no extra compressed bytes to skip   */
