} else if (!strncmp("null", str_val, str_len)) {
        /* JSON null */
        primitive_value = plist_new_null();
    } else {
        /* JSON number (integer or real) */
        char *num_str = (char*)malloc(str_len + 1);
        if (!num_str) {
            PLIST_JSON_ERR("%s: Out of memory\n", __func__);
            return NULL;
        }
        memcpy(num_str, str_val, str_len);
        num_str[str_len] = '\0';

        /* Decide if the value is real (contains '.' or 'e'/'E') */
        int is_real = 0;
        for (size_t i = 0; i < str_len; i++) {
            if (num_str[i] == '.' || num_str[i] == 'e' || num_str[i] == 'E') {
                is_real = 1;
                break;
            }
        }

        if (is_real) {
            char *endptr = NULL;
            double dval = strtod(num_str, &endptr);
            if (!endptr || *endptr != '\0') {
                PLIST_JSON_ERR("%s: invalid real value '%s'\n", __func__, num_str);
                free(num_str);
                return NULL;
            }
            primitive_value = plist_new_real(dval);
        } else {
            char *endptr = NULL;
            long long ival = strtoll(num_str, &endptr, 10);
            if (!endptr || *endptr != '\0') {
                PLIST_JSON_ERR("%s: invalid integer value '%s'\n", __func__, num_str);
                free(num_str);
                return NULL;
            }
            if (ival < 0) {
                primitive_value = plist_new_int((int64_t)ival);
            } else {
                primitive_value = plist_new_uint((uint64_t)ival);
            }
        }
        free(num_str);
    }
