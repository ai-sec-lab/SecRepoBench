// Remove leading whitespace characters
    while (data_size > 0 && (data[0] == ' ' || data[0] == '\n' || data[0] == '\r' || data[0] == '\t')) {
        data++;
        data_size--;
    }

    // Ensure that the data is not empty after stripping
    if (data_size == 0) {
        if (dst) gf_free(dst);
        return NULL;
    }
