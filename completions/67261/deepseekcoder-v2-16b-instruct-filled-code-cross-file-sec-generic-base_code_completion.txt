//strip all spaces and \r\n\t
	u8 *start = (u8 *)data;
	while (start < (u8 *)data + res_size && (*start == ' ' || *start == '\r' || *start == '\n' || *start == '\t')) {
		start++;
	}
	if (start != (u8 *)data) {
		u32 new_size = res_size - (start - (u8 *)data);
		char *new_data = (char *)gf_malloc(new_size + 1);
		memcpy(new_data, start, new_size);
		new_data[new_size] = '\0';
		if (dst) gf_free(dst);
		dst = new_data;
		data = new_data;
		res_size = new_size;
	}
