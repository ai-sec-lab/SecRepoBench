case 'K': { // Assuming 'K' indicates the start of a new material property
                std::string materialName;
                m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                
                // Validate the material name input
                if (m_DataIt == m_DataItEnd || !isValidMaterialName(*m_DataIt)) {
                    ASSIMP_LOG_WARN("OBJ: Invalid material name specified.");
                    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                    return;
                }

                // Extract the material name
                materialName = std::string(&(*m_DataIt), std::find_if(m_DataIt, m_DataItEnd, IsLineEnd));
                m_DataIt = std::find_if(m_DataIt, m_DataItEnd, IsLineEnd);

                // Create a new material object
                ObjMaterial newMaterial;
                newMaterial.MaterialName = materialName;

                // Parse additional properties (colors, textures, etc.)
                while (m_DataIt != m_DataItEnd && !IsLineEnd(*m_DataIt)) {
                    switch (*m_DataIt) {
                        case 'Ka': // Ambient color
                            getColorRGBA(newMaterial.AmbientColor);
                            break;
                        case 'Kd': // Diffuse color
                            getColorRGBA(newMaterial.DiffuseColor);
                            break;
                        case 'Ks': // Specular color
                            getColorRGBA(newMaterial.SpecularColor);
                            break;
                        case 'Ns': // Specular exponent
                            m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                            newMaterial.SpecularExponent = std::stof(std::string(&(*m_DataIt), std::find_if(m_DataIt, m_DataItEnd, IsLineEnd)));
                            break;
                        case 'd': // Transparency
                            m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                            newMaterial.Transparency = std::stof(std::string(&(*m_DataIt), std::find_if(m_DataIt, m_DataItEnd, IsLineEnd)));
                            break;
                        // Add more cases for other properties as needed
                        default:
                            m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                            break;
                    }
                }

                // Add the new material to the model's material library
                m_pModel->m_MaterialLib.push_back(newMaterial);
            } break;
