case 'k':
            case 'K': {
                // advance to the second character of the token (e.g. 'a' in "Ka")
                ++m_DataIt;
                if (m_DataIt == m_DataItEnd) {
                    break;
                }

                // Identify which kind of colour is specified (ambient, diffuse, specular, emissive, …)
                char colourId = static_cast<char>(::tolower(static_cast<unsigned char>(*m_DataIt)));
                ++m_DataIt; // skip the colour id character

                // If we do not have a current material, just skip the rest of the line.
                if (nullptr == m_pModel->m_pCurrentMaterial) {
                    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                    break;
                }

                switch (colourId) {
                    case 'a': // Ka – Ambient colour
                        getColorRGBA(m_pModel->m_pCurrentMaterial->ambient);
                        break;

                    case 'd': // Kd – Diffuse colour
                        getColorRGBA(m_pModel->m_pCurrentMaterial->diffuse);
                        break;

                    case 's': // Ks – Specular colour
                        getColorRGBA(m_pModel->m_pCurrentMaterial->specular);
                        break;

                    case 'e': // Ke – Emissive colour
                        getColorRGBA(m_pModel->m_pCurrentMaterial->emissive);
                        break;

                    case 't': // Kt – Transmission filter (not standard but seen in some files)
                        getColorRGBA(m_pModel->m_pCurrentMaterial->transparent);
                        break;

                    default:  // Unknown token – skip the rest of the line
                        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                        break;
                }
            } break;
