case 'm': {
                m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                if (m_DataIt != m_DataItEnd) {
                    std::string materialName(&(*m_DataIt), std::find_if(m_DataIt, m_DataItEnd, IsLineEnd));
                    m_DataIt = std::find_if(m_DataIt, m_DataItEnd, IsLineEnd);
                    if (!materialName.empty()) {
                        // Create a new material if it doesn't exist
                        if (!m_pModel->m_pCurrentMaterial || m_pModel->m_pCurrentMaterial->MaterialName.data != materialName) {
                            m_pModel->m_pCurrentMaterial = new Material();
                            m_pModel->m_pCurrentMaterial->MaterialName.data = materialName;
                            m_pModel->m_Materials.push_back(m_pModel->m_pCurrentMaterial);
                        }
                    }
                }
            } break;

            case 'K': {
                // Handle different material properties based on the character
                if (*(m_DataIt + 1) == 'a') {
                    getColorRGBA(m_pModel->m_pCurrentMaterial->AmbientColor);
                } else if (*(m_DataIt + 1) == 'd') {
                    getColorRGBA(m_pModel->m_pCurrentMaterial->DiffuseColor);
                } else if (*(m_DataIt + 1) == 's') {
                    getColorRGBA(m_pModel->m_pCurrentMaterial->SpecularColor);
                } else if (*(m_DataIt + 1) == 'e') {
                    m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                    m_pModel->m_pCurrentMaterial->Shininess = std::stof(std::string(&(*m_DataIt), std::find_if(m_DataIt, m_DataItEnd, IsLineEnd)));
                }
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            } break;

            case 'T': {
                m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                m_pModel->m_pCurrentMaterial->Transparency = std::stof(std::string(&(*m_DataIt), std::find_if(m_DataIt, m_DataItEnd, IsLineEnd)));
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            } break;

            case 'i': {
                m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                m_pModel->m_pCurrentMaterial->IndexOfRefraction = std::stof(std::string(&(*m_DataIt), std::find_if(m_DataIt, m_DataItEnd, IsLineEnd)));
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
            } break;
