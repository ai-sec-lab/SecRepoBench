case 'K': { // Assuming 'K' indicates the start of a new material property
                std::string materialName;
                m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                if (m_DataIt == m_DataItEnd) {
                    ASSIMP_LOG_WARN("OBJ: No material name specified.");
                    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                    return;
                }

                // Extract material name
                char *pStart = &(*m_DataIt);
                while (m_DataIt != m_DataItEnd && !IsLineEnd(*m_DataIt)) {
                    ++m_DataIt;
                }
                materialName.assign(pStart, &(*m_DataIt));

                // Validate material name
                if (materialName.empty()) {
                    ASSIMP_LOG_WARN("OBJ: Empty material name.");
                    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                    return;
                }

                // Create a new material
                ObjFile::Material *newMaterial = new ObjFile::Material();
                newMaterial->MaterialName = materialName;

                // Set the current material
                m_pModel->m_pCurrentMaterial = newMaterial;
                m_pModel->m_Materials.push_back(newMaterial);

                // Parse additional properties (colors, textures, etc.)
                while (m_DataIt != m_DataItEnd && !IsLineEnd(*m_DataIt)) {
                    switch (*m_DataIt) {
                        case 'Ka': // Ambient color
                            getColorRGBA(newMaterial->AmbientColor);
                            break;
                        case 'Kd': // Diffuse color
                            getColorRGBA(newMaterial->DiffuseColor);
                            break;
                        case 'Ks': // Specular color
                            getColorRGBA(newMaterial->SpecularColor);
                            break;
                        case 'Ke': // Emissive color
                            getColorRGBA(newMaterial->EmissiveColor);
                            break;
                        case 'Ns': // Specular exponent
                            m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                            newMaterial->Shininess = std::stof(std::string(&(*m_DataIt), &(*m_DataItEnd)));
                            break;
                        case 'd': // Transparency
                            m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                            newMaterial->Transparency = std::stof(std::string(&(*m_DataIt), &(*m_DataItEnd)));
                            break;
                        // Add more cases for other properties as needed
                        default:
                            m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                            break;
                    }
                }
            } break;
