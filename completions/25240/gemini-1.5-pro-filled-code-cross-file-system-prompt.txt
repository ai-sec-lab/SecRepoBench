{
    if (m_DataIt + 1 == m_DataItEnd) {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
        break;
    }

    switch (*(m_DataIt + 1)) {
        case 'a':
            // Parse ambient color
            if (strncmp(m_DataIt, "ka", 2) == 0) {
                getColorRGBA(m_pCurrentMaterial->ambient);
            }
            break;

        case 'd':
            // Parse diffuse color or dissolve
            if (strncmp(m_DataIt, "kd", 2) == 0) {
                getColorRGBA(m_pCurrentMaterial->diffuse);
            } else if (strncmp(m_DataIt, "d", 1) == 0) {
                getFloatValue(m_pCurrentMaterial->alpha);
                if (m_pCurrentMaterial->alpha < 1.0f) {
                    m_pCurrentMaterial->isTransparent = true;
                }
            }
            break;

        case 'e':
            // Parse emissive color
            if (strncmp(m_DataIt, "ke", 2) == 0) {
                getColorRGBA(m_pCurrentMaterial->emissive);
            }
            break;

        case 's':
            // Parse specular color or specular exponent or transmission filter
            if (strncmp(m_DataIt, "ks", 2) == 0) {
                getColorRGBA(m_pCurrentMaterial->specular);
            } else if (strncmp(m_DataIt, "ns", 2) == 0) {
                getFloatValue(m_pCurrentMaterial->shininess);
            } else if (strncmp(m_DataIt, "Tf", 2) == 0) {
                getColorRGBA(m_pCurrentMaterial->transmission);
            }
            break;

        case 'i':
            // Parse illumination model
            if (strncmp(m_DataIt, "illum", 5) == 0) {
                getIlluminationModel(m_pCurrentMaterial->illumination_model);
            } else if (strncmp(m_DataIt, "ior", 3) == 0) {
                getFloatValue(m_pCurrentMaterial->ior);
            }
            break;

        case 'm':
        case 'b':
        case 'r':
        case 'c':
            // Parse bump map or map or reflection map or cube map
            getTexture();
            break;

        case 'n':
            // Parse new material or anisotropy
            if (strncmp(m_DataIt, "newmtl", 6) == 0) {
                getNewMaterial();
            } else if (strncmp(m_DataIt, "Ni", 2) == 0) {
                getFloatValue(m_pCurrentMaterial->anisotropy);
            }
            break;


        default: {
            m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
        } break;
    }
} break;
