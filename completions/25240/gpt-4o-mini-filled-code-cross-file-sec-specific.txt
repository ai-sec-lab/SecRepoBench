case 'K': { // Assuming 'K' indicates the start of a new material property
                std::string materialName;
                m_DataIt = getNextToken<DataArrayIt>(m_DataIt, m_DataItEnd);
                if (m_DataIt == m_DataItEnd) {
                    return; // Prevent buffer overflow by checking bounds
                }

                // Capture the material name safely
                const char *pStart = &(*m_DataIt);
                while (m_DataIt != m_DataItEnd && !IsLineEnd(*m_DataIt)) {
                    ++m_DataIt;
                }
                materialName.assign(pStart, m_DataIt); // Use assign to avoid overflow

                // Create a new material and set its properties
                if (!materialName.empty()) {
                    m_pModel->m_pCurrentMaterial = new Material(); // Ensure proper memory management
                    m_pModel->m_pCurrentMaterial->MaterialName = materialName;

                    // Additional parsing for properties can be added here
                } else {
                    ASSIMP_LOG_WARN("OBJ: Material name is empty.");
                }
            } break;
