{
                ++m_DataIt;
                if (m_DataIt == m_DataItEnd) {
                    return;
                }

                switch (*m_DataIt) {
                    case 'a': // Ambient color
                    case 'd': // Diffuse color
                    case 's': // Specular color
                    case 'e': // Emissive color
                    case 't': // Transmission filter
                    {
                        ++m_DataIt;
                        Maybe<aiColor3D> color;
                        getColorRGBA(color);
                        if (color) {
                            switch (*(m_DataIt - 1)) {
                                case 'a': m_pCurrentMaterial->m_Ambient = *color; break;
                                case 'd': m_pCurrentMaterial->m_Diffuse = *color; break;
                                case 's': m_pCurrentMaterial->m_Specular = *color; break;
                                case 'e': m_pCurrentMaterial->m_Emissive = *color; break;
                                case 't': m_pCurrentMaterial->m_TransmissionFilter = *color; break;
                            }
                        }
                    } break;

                    case 'e': // Specular exponent
                    {
                        ++m_DataIt;
                        float exponent;
                        if (getFloatValue(exponent)) {
                            m_pCurrentMaterial->m_SpecularExponent = exponent;
                        }
                    } break;

                    case 'i': // Index of refraction
                    {
                        ++m_DataIt;
                        float ior;
                        if (getFloatValue(ior)) {
                            m_pCurrentMaterial->m_IndexOfRefraction = ior;
                        }
                    } break;

                    case 'd': // Dissolve (transparency)
                    {
                        ++m_DataIt;
                        float alpha;
                        if (getFloatValue(alpha)) {
                            m_pCurrentMaterial->m_Alpha = alpha;
                        }
                    } break;

                    default:
                        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                        break;
                }
            }
