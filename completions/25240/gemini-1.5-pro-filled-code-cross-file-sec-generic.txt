case 'K':
            if (m_DataIt + 1 == m_DataItEnd) {
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                break;
            }
            switch (*++m_DataIt) {
                case 'a':
                    getColorRGBA(m_pCurrentMaterial->ambient);
                    break;
                case 'd':
                    getColorRGBA(m_pCurrentMaterial->diffuse);
                    break;
                case 's':
                    getColorRGBA(m_pCurrentMaterial->specular);
                    break;
                case 'e':
                    getColorRGBA(m_pCurrentMaterial->emissive);
                    break;
                case 't':
                    getColorRGBA(m_pCurrentMaterial->transparent);
                    break;
                default:
                    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                    break;
            }
            break;

        case 'N':
            if (m_DataIt + 1 == m_DataItEnd) {
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                break;
            }
            switch (*++m_DataIt) {
                case 's':
                    getFloatValue(m_pCurrentMaterial->shininess);
                    break;
                case 'i':
                    getFloatValue(m_pCurrentMaterial->ior);
                    break;
                default:
                    m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                    break;
            }
            break;

        case 'd':
            getFloatValue(m_pCurrentMaterial->alpha);
            break;

        case 'T':
            if (m_DataIt + 1 == m_DataItEnd) {
                m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                break;
            }
            if (*++m_DataIt == 'r') {
                // Inverted alpha value.
                getFloatValue(m_pCurrentMaterial->alpha);
                m_pCurrentMaterial->alpha = ai_real(1.0) - m_pCurrentMaterial->alpha;
            } else {
                --m_DataIt;
                getTexture();
            }

            break;
        case 'm':
        case 'b':
        case 'g':
        case 'r':
        case 'M':
            getTexture();
            break;

        case 'n':
            getNewMaterial();
            break;

        case 'i':
            getIlluminationModel();
            break;

        case 'a':
            getAnisotropy();
            break;
