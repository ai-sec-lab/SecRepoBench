{
                ++m_DataIt;
                if (m_DataIt == m_DataItEnd) {
                    return;
                }

                switch (*m_DataIt) {
                    case 'a': // Ambient color
                        ++m_DataIt;
                        aiColor3D ambient;
                        getColorRGBA(ambient);
                        m_pCurrentMaterial->m_Ambient = ambient;
                        break;
                    case 'd': // Diffuse color
                        ++m_DataIt;
                        aiColor3D diffuse;
                        getColorRGBA(diffuse);
                        m_pCurrentMaterial->m_Diffuse = diffuse;
                        break;
                    case 's': // Specular color
                        ++m_DataIt;
                        aiColor3D specular;
                        getColorRGBA(specular);
                        m_pCurrentMaterial->m_Specular = specular;
                        break;
                    case 'e': // Emissive color
                        ++m_DataIt;
                        aiColor3D emissive;
                        getColorRGBA(emissive);
                        m_pCurrentMaterial->m_Emissive = emissive;
                        break;
                    case 't': // Transmission filter
                        ++m_DataIt;
                        aiColor3D transmission;
                        getColorRGBA(transmission);
                        m_pCurrentMaterial->m_TransmissionFilter = transmission;
                        break;
                    case 'T': // Transparency
                        ++m_DataIt;
                        float transparency;
                        getFloat(transparency);
                        m_pCurrentMaterial->m_Transparency = transparency;
                        break;
                    case 'N': // Specular exponent or IOR
                        ++m_DataIt;
                        if (*m_DataIt == 's') { // Specular exponent
                            ++m_DataIt;
                            float specularExponent;
                            getFloat(specularExponent);
                            m_pCurrentMaterial->m_SpecularExponent = specularExponent;
                        } else if (*m_DataIt == 'i') { // Index of Refraction
                            ++m_DataIt;
                            float ior;
                            getFloat(ior);
                            m_pCurrentMaterial->m_IndexOfRefraction = ior;
                        }
                        break;
                    case 'n': // New material
                        ++m_DataIt;
                        std::string materialName;
                        getName(materialName);
                        m_pCurrentMaterial = new ObjFile::Material;
                        m_pCurrentMaterial->MaterialName = materialName;
                        m_pModel->m_MaterialLib.push_back(m_pCurrentMaterial);
                        break;
                    case 'm': // Map (texture)
                        ++m_DataIt;
                        std::string textureType;
                        getName(textureType);
                        if (textureType == "Kd") { // Diffuse texture
                            std::string textureFile;
                            getName(textureFile);
                            m_pCurrentMaterial->m_DiffuseTexture = textureFile;
                        }
                        // Handle other texture types similarly
                        break;
                    case 'i': // Illumination model
                        ++m_DataIt;
                        int illumModel;
                        getInt(illumModel);
                        m_pCurrentMaterial->m_IlluminationModel = illumModel;
                        break;
                    default:
                        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
                        break;
                }
            }
