case 'k':
{
    // Ambient color
    if (strncmp(m_DataIt, "Ka", 2) == 0) {
        m_DataIt += 2;
        getColorRGBA(m_pModel->m_pCurrentMaterial->ambient);
    }
    // Diffuse color
    else if (strncmp(m_DataIt, "Kd", 2) == 0) {
        m_DataIt += 2;
        getColorRGBA(m_pModel->m_pCurrentMaterial->diffuse);
    }
    // Specular color
    else if (strncmp(m_DataIt, "Ks", 2) == 0) {
        m_DataIt += 2;
        getColorRGBA(m_pModel->m_pCurrentMaterial->specular);
    }
    // Emissive color
    else if (strncmp(m_DataIt, "Ke", 2) == 0) {
        m_DataIt += 2;
        getColorRGBA(m_pModel->m_pCurrentMaterial->emissive);
    }
    // Transmission filter
    else if (strncmp(m_DataIt, "Tf", 2) == 0) {
        m_DataIt += 2;
        getColorRGBA(m_pModel->m_pCurrentMaterial->transparent);
    }
    // Transparency
    else if (strncmp(m_DataIt, "d", 1) == 0) {
        m_DataIt += 1;
        getFloat(m_pModel->m_pCurrentMaterial->alpha);
    }
    // Specular exponent (shininess)
    else if (strncmp(m_DataIt, "Ns", 2) == 0) {
        m_DataIt += 2;
        getFloat(m_pModel->m_pCurrentMaterial->shininess);
    }
    // Index of refraction
    else if (strncmp(m_DataIt, "Ni", 2) == 0) {
        m_DataIt += 2;
        getFloat(m_pModel->m_pCurrentMaterial->ior);
    }
    // New material
    else if (strncmp(m_DataIt, "newmtl", 6) == 0) {
        m_DataIt += 6;
        createMaterial();
    }
    // Texture
    else if (strncmp(m_DataIt, "map_", 4) == 0) {
        m_DataIt += 4;
        handleTexture();
    }
    // Illumination model
    else if (strncmp(m_DataIt, "illum", 5) == 0) {
        m_DataIt += 5;
        getInteger(m_pModel->m_pCurrentMaterial->illumination_model);
    }
    // Anisotropy
    else if (strncmp(m_DataIt, "aniso", 5) == 0) {
        m_DataIt += 5;
        getFloat(m_pModel->m_pCurrentMaterial->anisotropy);
    }
    else {
        m_DataIt = skipLine<DataArrayIt>(m_DataIt, m_DataItEnd, m_uiLine);
    }
} break;
