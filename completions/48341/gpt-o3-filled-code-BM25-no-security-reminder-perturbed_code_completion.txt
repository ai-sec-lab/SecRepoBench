case ReadStripMethod:
      case ReadTileMethod:
      case ReadGenericMethod:
      default:
      {
        /*
          Read TIFF image data a scan-line at a time and import the pixels.
          For Strip / Tile methods we simply iterate over the image rows;
          libtiff transparently handles the underlying organisation.
        */
        for (y=0; y < (ssize_t) image->rows; y++)
        {
          /*
            Read a row (or the required portion of a tile / strip) from the
            TIFF image.
          */
          tiff_status=TIFFReadPixels(tiff,0,y,(char *) pixels);
          if (tiff_status == -1)
            break;

          /*
            Import the quantum pixels we just read into the Image pixel cache.
          */
          (void) ImportQuantumPixels(image,(CacheView *) NULL,quantum_info,
            quantum_type,pixels,exception);

          /*
            Provide progress feedback.
          */
          if (image->previous == (Image *) NULL)
            {
              status=SetImageProgress(image,LoadImageTag,(MagickOffsetType) y,
                image->rows);
              if (status == MagickFalse)
                break;
            }
        }
        break;
      }
