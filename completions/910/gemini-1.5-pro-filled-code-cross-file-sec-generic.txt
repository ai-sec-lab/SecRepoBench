if (!io ->Seek(io, Offset)) goto Error;

    // Get the tag descriptor
    TagDescriptor = _cmsGetTagDescriptor(Icc->ContextID, sig);
    if (TagDescriptor == NULL) goto Error;

    // Read base type
    if (!io ->Read(io, &BaseType, sizeof(cmsTagTypeSignature), 1, io)) goto Error;

    // Check if the type is supported by the tag descriptor
    if (!IsTypeSupported(TagDescriptor, BaseType)) goto Error;

    TagSize -= sizeof(cmsTagTypeSignature);  // The base type has been already read
