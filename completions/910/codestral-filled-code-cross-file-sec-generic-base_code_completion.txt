// Seek to the tag's location in the IO handler
if (!Icc->IOhandler->Seek(Icc->IOhandler, Offset)) goto Error;

// Read the base type of the tag
if (!_cmsReadTypeBase(Icc->IOhandler, &BaseType)) goto Error;

// Verify if the base type is supported by the tag descriptor
TagDescriptor = _cmsGetTagDescriptor(Icc->ContextID, sig);
if (TagDescriptor == NULL) goto Error;
if (!IsTypeSupported(TagDescriptor, BaseType)) goto Error;

// Deduct 8 bytes from the tag size to account for the tag's base type header
TagSize -= 8;
