// Seek to the tag's location in the IO handler.
    if (!io ->Seek(io, Offset))
        goto Error;

    // Read base type (4 bytes signature + 4 bytes reserved)
    if (!_cmsReadTypeBase(io, &BaseType) || BaseType == (cmsTagTypeSignature) 0)
        goto Error;

    // Obtain tag descriptor and be sure this type is supported for this tag
    TagDescriptor = _cmsGetTagDescriptor(Icc->ContextID, tagSignature);
    if (TagDescriptor == NULL)
        goto Error;

    if (!IsTypeSupported(TagDescriptor, BaseType))
        goto Error;

    // Adjust tag size so it reflects only the payload (exclude the 8-byte base)
    if (TagSize < (cmsUInt32Number)(sizeof(cmsTagTypeSignature) + sizeof(cmsUInt32Number)))
        goto Error;

    TagSize -= (cmsUInt32Number)(sizeof(cmsTagTypeSignature) + sizeof(cmsUInt32Number));
