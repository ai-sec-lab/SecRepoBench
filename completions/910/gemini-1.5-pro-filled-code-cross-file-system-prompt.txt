if (!io ->Seek(io, Offset)) goto Error;

    // Check if tag is supported.
    TagDescriptor = _cmsGetTagDescriptor(Icc->ContextID, sig);
    if (TagDescriptor == NULL) {
        cmsSignalError(Icc ->ContextID, cmsERROR_UNKNOWN_EXTENSION, "Unsupported tag '%X'", sig);
        goto Error;
    }

    if (!io ->Read(io, &BaseType, sizeof(cmsTagTypeSignature), 1, io)) goto Error;
    if (!IsTypeSupported(TagDescriptor, BaseType)) {
        cmsSignalError(Icc ->ContextID, cmsERROR_UNKNOWN_EXTENSION, "Unsupported tag type '%X'", BaseType);
        goto Error;
    }

    TagSize -= sizeof(cmsTagTypeSignature);  // Already read
