hb_set_t *gids_to_retain = hb_set_create ();
while (initial_gids_to_retain->next (&gid))
{
  hb_codepoint_t component_gid;
  glyf.get_components (gid, &component_gid);
  while (component_gid!= HB_SET_VALUE_INVALID)
  {
    gids_to_retain->add (component_gid);
    glyf.get_components (component_gid, &component_gid);
  }
  gids_to_retain->add (gid);
}
hb_set_destroy (initial_gids_to_retain);
glyphs->resize (gids_to_retain->get_population ());
unsigned int i = 0;
while (gids_to_retain->next (&gid))
{
  glyphs->elem (i++) = gid;
}
hb_set_destroy (gids_to_retain);
cmap.fini ();
glyf.fini ();
return full_glyphs_to_retain;
