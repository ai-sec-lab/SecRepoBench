/* the key must be a string */
        if (k->type != MSGPACK_OBJECT_STR) {
            k = NULL;
            v = NULL;
            continue;
        }

        /* the key size must match the expected time key size */
        if (k->via.str.size != slen) {
            k = NULL;
            v = NULL;
            continue;
        }

        /* safety check: pointer must not be NULL */
        if (k->via.str.ptr == NULL) {
            if (tmp_out_buf) {
                flb_free(tmp_out_buf);
            }
            *out_buf = NULL;
            msgpack_unpacked_destroy(&unpacked_data);
            return -1;
        }
