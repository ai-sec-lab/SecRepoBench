// Handle specific keywords related to output image properties.
switch (*keyword)
{
  case 'a':
  case 'A':
  {
    if (LocaleCompare(keyword,"alpha-trait") == 0)
    {
      ssize_t
        alpha_trait;

      alpha_trait=ParseCommandOption(MagickPixelTraitOptions,MagickFalse,options);
      if (alpha_trait < 0)
        break;
      output_image->alpha_trait=(PixelTrait) alpha_trait;
      break;
    }
    (void) SetImageProperty(output_image,keyword,options,exception);
    break;
  }
  // ... (other keyword cases)
  case 'i':
  case 'I':
  {
    if (LocaleCompare(keyword,"id") == 0)
    {
      (void) QueryColorCompliance(options,AllCompliance,&output_image->background_color,exception);
      break;
    }
    // ... (other id-related code)
    break;
  }
  case 'm':
  case 'M':
  {
    if (LocaleCompare(keyword,"matte") == 0)
    {
      ssize_t
        matte;

      matte=ParseCommandOption(MagickBooleanOptions,MagickFalse,options);
      if (matte < 0)
        break;
      output_image->alpha_trait=matte == 0 ? UndefinedPixelTrait : BlendPixelTrait;
      break;
    }
    // ... (other matte-related code)
    break;
  }
  case 'o':
  case 'O':
  {
    if (LocaleCompare(keyword,"orientation") == 0)
    {
      ssize_t
        orientation;

      orientation=ParseCommandOption(MagickOrientationOptions,MagickFalse,options);
      if (orientation < 0)
        break;
      output_image->orientation=(OrientationType) orientation;
      break;
    }
    // ... (other orientation-related code)
    break;
  }
  case 'p':
  case 'P':
  {
    if (LocaleCompare(keyword,"page") == 0)
    {
      char
        *geometry;

      geometry=GetPageGeometry(options);
      (void) ParseAbsoluteGeometry(geometry,&output_image->page);
      geometry=DestroyString(geometry);
      break;
    }
    // ... (other page-related code)
    break;
  }
  case 't':
  case 'T':
  {
    if (LocaleCompare(keyword,"tile-offset") == 0)
    {
      char
        *geometry;

      geometry=GetPageGeometry(options);
      (void) ParseAbsoluteGeometry(geometry,&output_image->tile_offset);
      geometry=DestroyString(geometry);
      break;
    }
    // ... (other tile-offset-related code)
    break;
  }
  case 'u':
  case 'U':
  {
    if (LocaleCompare(keyword,"units") == 0)
    {
      ssize_t
        units;

      units=ParseCommandOption(MagickResolutionOptions,MagickFalse,options);
      if (units < 0)
        break;
      output_image->units=(ResolutionType) units;
      break;
    }
    // ... (other units-related code)
    break;
  }
  case 'v':
  case 'V':
  {
    if (LocaleCompare(keyword,"version") == 0)
    {
      version=StringToDouble(options,(char **) NULL);
      break;
    }
    // ... (other version-related code)
    break;
  }
  case 'w':
  case 'W':
  {
    if (LocaleCompare(keyword,"white-point") == 0)
    {
      flags=ParseGeometry(options,&geometry_info);
      output_image->chromaticity.white_point.x=geometry_info.rho;
      output_image->chromaticity.white_point.y=geometry_info.sigma;
      if ((flags & SigmaValue) == 0)
        output_image->chromaticity.white_point.y=output_image->chromaticity.white_point.x;
      break;
    }
    // ... (other white-point-related code)
    break;
  }
  default:
  {
    (void) SetImageProperty(output_image,keyword,options,exception);
    break;
  }
}
