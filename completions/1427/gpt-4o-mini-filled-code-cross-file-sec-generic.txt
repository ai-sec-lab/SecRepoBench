// Check if the video size is sufficient
    if (c->video_size < aligned_width * h) {
        return AVERROR_INVALIDDATA;
    }

    // Determine the appropriate pixel format
    if (c->bpp <= 8 && c->palette_size > 0) {
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
    } else if ((c->bpp == 6 || c->bpp == 8) && c->palette_size == 256) {
        avctx->pix_fmt = AV_PIX_FMT_BGR24;
    } else if (c->bpp == 24 && c->format == CHUNKY && c->palette_size == 0) {
        avctx->pix_fmt = AV_PIX_FMT_RGB24;
    } else {
        avpriv_request_sample(avctx, "Unsupported pixel format: encoding=%d, palette_size=%d, bpp=%d", encoding, c->palette_size, c->bpp);
        return AVERROR_PATCHWELCOME;
    }
