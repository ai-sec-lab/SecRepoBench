int size = avctx->height * aligned_width * c->bpp;
    size = (size + 7) >> 3;
    if (c->video_size < size)
        return AVERROR_INVALIDDATA;
    if (encoding == 0 && c->palette_size > 0 && c->bpp <= 8)
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
    else if (encoding && (c->bpp == 6 || c->bpp == 8) && c->palette_size == 768)
        avctx->pix_fmt = AV_PIX_FMT_BGR24;
    else if (encoding == 0 && c->bpp == 24 && c->format == CHUNKY && c->palette_size == 0)
        avctx->pix_fmt = AV_PIX_FMT_RGB24;
    else {
        avpriv_request_sample(avctx, "Pixel format 0x%0x, bpp %d, encoding %d, palette %d", c->format, c->bpp, encoding, c->palette_size);
        return AVERROR_PATCHWELCOME;
    }
