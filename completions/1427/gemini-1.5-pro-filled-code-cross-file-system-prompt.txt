if (avctx->height * aligned_width * c->bpp / 8 > c->video_size)
        return AVERROR_INVALIDDATA;

    if (!encoding && c->palette_size && c->bpp <= 8) {
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
    } else if (encoding && (c->bpp == 6 || c->bpp == 8) && c->palette_size == 768) {
        avctx->pix_fmt = AV_PIX_FMT_BGR24;
    } else if (!encoding && c->bpp == 24 && c->format == CHUNKY && !c->palette_size) {
        avctx->pix_fmt = AV_PIX_FMT_RGB24;
    } else {
        avpriv_request_sample(avctx, "Encoding %d, palette_size %d, bpp %d, format %d",
                              encoding, c->palette_size, c->bpp, c->format);
        return AVERROR_PATCHWELCOME;
    }
