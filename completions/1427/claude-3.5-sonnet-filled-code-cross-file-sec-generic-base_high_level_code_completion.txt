if (encoding == 0 && c->palette_size && c->bpp <= 8) {
    avctx->pix_fmt = AV_PIX_FMT_PAL8;
} else if (encoding == 0 && c->bpp == 24) {
    avctx->pix_fmt = AV_PIX_FMT_RGB24;
} else if (encoding != 0 && c->bpp == 6) {
    avctx->pix_fmt = AV_PIX_FMT_RGB24;
} else if (encoding != 0 && c->bpp == 8) {
    avctx->pix_fmt = AV_PIX_FMT_RGB24;
} else {
    avpriv_request_sample(avctx, "Encoding %d, bpp %d, format %d",
                          encoding, c->bpp, c->format);
    return AVERROR_PATCHWELCOME;
}

if (c->video_size < (c->bpp * aligned_width * avctx->height) >> 3)
    return AVERROR_INVALIDDATA;
