if (c->video_size < (aligned_width * c->bpp * avctx->height + 7) / 8)
    return AVERROR_INVALIDDATA;

if (c->palette_size && c->bpp <= 8) {
    avctx->pix_fmt = AV_PIX_FMT_PAL8;
} else if (!compressiontype && ((c->bpp == 6 && c->palette_size == 32) ||
                                (c->bpp == 8 && c->palette_size == 256))) {
    avctx->pix_fmt = AV_PIX_FMT_BGR24;
} else if (!compressiontype && c->bpp == 24 && c->format == CHUNKY && !c->palette_size) {
    avctx->pix_fmt = AV_PIX_FMT_RGB24;
} else {
    avpriv_request_sample(avctx, "Unsupported format (bpp=%d, palette_size=%d, compression=%d, format=%d)",
                          c->bpp, c->palette_size, compressiontype, c->format);
    return AVERROR_PATCHWELCOME;
}
