// Check if the video size is sufficient based on the aligned width, height,
    // and bits per pixel (bpp). Return an error if it is not.
    if (c->video_size < (aligned_width * h * (c->bpp / 8))) {
        return AVERROR_INVALIDDATA;
    }

    // Determine the appropriate pixel format for the AVCodecContext based on
    // encoding, palette size, bits per pixel, and video format.
    if (c->bpp <= 8 && c->palette_size > 0) {
        avctx->pix_fmt = AV_PIX_FMT_PAL8; // Use palette format
    } else if ((c->bpp == 6 || c->bpp == 8) && c->palette_size == 256) {
        avctx->pix_fmt = AV_PIX_FMT_BGR24; // Use BGR format for 6 or 8 bpp
    } else if (c->bpp == 24 && c->format == CHUNKY && c->palette_size == 0) {
        avctx->pix_fmt = AV_PIX_FMT_RGB24; // Use RGB format for 24 bpp
    } else {
        avpriv_request_sample(avctx, "Unsupported pixel format: encoding=%d, bpp=%d, palette_size=%d", encoding, c->bpp, c->palette_size);
        return AVERROR_PATCHWELCOME; // Request further support for unsupported formats
    }
