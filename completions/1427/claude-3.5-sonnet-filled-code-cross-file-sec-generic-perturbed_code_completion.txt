if (c->video_size < (aligned_width * height * c->bpp + 7) / 8)
    return AVERROR_INVALIDDATA;

if (encoding == 0 && c->palette_size && c->bpp <= 8) {
    avctx->pix_fmt = AV_PIX_FMT_PAL8;
} else if (encoding != 0 && ((c->bpp == 6 && c->palette_size == 32) ||
                             (c->bpp == 8 && c->palette_size == 64))) {
    avctx->pix_fmt = AV_PIX_FMT_BGR24;
} else if (encoding == 0 && c->bpp == 24 && c->format == CHUNKY && c->palette_size == 0) {
    avctx->pix_fmt = AV_PIX_FMT_RGB24;
} else {
    avpriv_request_sample(avctx, "Unsupported format (encoding: %d, bpp: %d, palette size: %d, format: %d)",
                          encoding, c->bpp, c->palette_size, c->format);
    return AVERROR_PATCHWELCOME;
}
