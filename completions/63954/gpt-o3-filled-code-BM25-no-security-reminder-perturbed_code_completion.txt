// Make sure we have a valid table
    if (t == NULL)
        return NULL;

    // Validate row (setIndex) and column (nField) boundaries
    if (setIndex < 0 || setIndex >= t->nPatches)
        return NULL;

    if (nField < 0 || nField >= t->nSamples)
        return NULL;

    // Shortcut for clarity
    int nSamples = t->nSamples;

    // Return the requested cell
    return t->Data[setIndex * nSamples + nField];
