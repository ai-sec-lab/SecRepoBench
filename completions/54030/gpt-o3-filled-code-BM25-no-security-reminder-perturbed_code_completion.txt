              /* Try to analyse the rest of the string (from position i) as a
                 compound word.  We do this recursively, increasing the word
                 counters so that the depth/length restrictions of the caller
                 are still honoured. */
              if ((maxwordnum == 0) || (wordnum + 1 < maxwordnum)) {
                const std::string rest(word.substr(i));

                hentry* rv_comp = compound_check(rest,
                                                 static_cast<short>(wordnum + 1),
                                                 numsyllable,
                                                 maxwordnum,
                                                 static_cast<short>(wnum + 1),
                                                 words,
                                                 rwords,
                                                 hu_mov_rule,
                                                 is_sug,
                                                 info);

                if (rv_comp) {
                  /* Store the first recognised element of the compound if the
                     caller asked for the list of components. */
                  if (rwords && (wnum < maxwordnum))
                    rwords[wnum] = rv_first;

                  return rv_first;
