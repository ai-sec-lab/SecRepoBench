if (!vpar || flv_codecid == 0) {
        return 1; // True if codec ID is not set
    }

    switch (flv_codecid) {
        case AV_CODEC_ID_HEVC:
        case AV_CODEC_ID_AV1:
        case AV_CODEC_ID_VP9:
        case AV_CODEC_ID_FLV1:
        case AV_CODEC_ID_FLASHSV:
        case AV_CODEC_ID_FLASHSV2:
        case AV_CODEC_ID_VP6F:
        case AV_CODEC_ID_VP6A:
        case AV_CODEC_ID_H264:
            return vpar->codec_id == flv_codecid; // Match found
        default:
            return vpar->codec_tag == (flv_codecid >> 8); // Compare codec tag
    }
