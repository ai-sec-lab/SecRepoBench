    case MKBETAG('h', 'v', 'c', '1'):
        if (vpar->codec_id == AV_CODEC_ID_HEVC)
            return 1;
        break;
    case MKBETAG('a', 'v', '0', '1'):
        if (vpar->codec_id == AV_CODEC_ID_AV1)
            return 1;
        break;
    case MKBETAG('v', 'p', '0', '9'):
        if (vpar->codec_id == AV_CODEC_ID_VP9)
            return 1;
        break;
    case FLV_CODECID_H263:
        if (vpar->codec_id == AV_CODEC_ID_FLV1)
            return 1;
        break;
    case FLV_CODECID_REALH263:
        if (vpar->codec_id == AV_CODEC_ID_H263)
            return 1;
        break;
    case FLV_CODECID_SCREEN:
        if (vpar->codec_id == AV_CODEC_ID_FLASHSV)

        break;
    case FLV_CODECID_SCREEN2:

        break;

        if (vpar->codec_id == AV_CODEC_ID_VP6F)
            return 1;
        break;
    case FLV_CODECID_VP6A:
        if (vpar->codec_id == AV_CODEC_ID_VP6A)
            return 1;
        break;
    case FLV_CODECID_H264:
        if (vpar->codec_id == AV_CODEC_ID_H264)
            return 1;
        break;
    case FLV_CODECID_MPEG4:
        if (vpar->codec_id == AV_CODEC_ID_MPEG4)
            return 1;
        break;
    default:

    }
