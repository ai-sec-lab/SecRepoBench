for (unsigned int i = 0; i < glyph_count; ++i) {
    unsigned int sid = get_starting_sid(); // Assume this function retrieves the starting SID for the current range
    unsigned int num_consecutive_sids = get_num_consecutive_sids(); // Assume this function retrieves the number of consecutive SIDs
    for (unsigned int j = 0; j < num_consecutive_sids; ++j) {
        if (i >= glyph_count) break; // Ensure we don't go beyond the total number of glyphs
        mapping->set(i, sid + j); // Map the current glyph identifier to the SID
        ++i; // Increment the glyph identifier
    }
    --i; // Adjust the increment to avoid skipping the next range
}
