    *next_image;

    status;

  register ssize_t
    i;

  size_t
    columns,
    rows,
    scene;

  ssize_t
    count;

  /*
    Open image file.
  */
  assert(image_info != (ImageInfo *) NULL);
  assert(image_info->signature == MagickCoreSignature);
  if (image_info->debug != MagickFalse)
    (void) LogMagickEvent(TraceEvent,GetMagickModule(),"%s",
      image_info->filename);
  assert(exception != (ExceptionInfo *) NULL);
  image=AcquireImage(image_info,exception);
  status=OpenBlob(image_info,image,ReadBinaryBlobMode,exception);
  if (status == MagickFalse)
    {
      image=DestroyImageList(image);
      return((Image *) NULL);
    }
  /*
    Translate Postscript to encapsulated Postscript.
  */
  image=DestroyImage(image);
  (void) FormatLocaleString(postscript_filename,MagickPathExtent,"%s.eps",
    image_info->unique);
  file=AcquireUniqueFileResource(postscript_filename);
  if (file == -1)
    ThrowReaderException(FileOpenError,"UnableToCreateTemporaryFile");
  (void) CopyMagickString(input_filename,image_info->filename,MagickPathExtent);
  if (IsPathAccessible(input_filename) != MagickFalse)
    {
      (void) FormatLocaleString(command,MagickPathExtent,
        GetMagickHomeURL() "bin/ps2eps \"%s\" >@%d 2>/dev/null",
        input_filename,file);
      (void) SystemCommand(MagickFalse,command,exception);
    }
  (void) CloseBlob(image);
  /*
    Read image from encapsulated Postscript.
  */
  read_info=CloneImageInfo(image_info);
  (void) CopyMagickString(read_info->filename,postscript_filename,
    MagickPathExtent);
  image=ReadEPSImage(read_info,exception);
  read_info=DestroyImageInfo(read_info);
  (void) RelinquishUniqueFileResource(postscript_filename);
  if (image == (Image *) NULL)
    return((Image *) NULL);
  if (image->eps_info == (void *) NULL)
    return(image);
  /*
    Relinquish resources.
  */
  options=DestroyString(options);
  density=DestroyString(density);
  columns=image->columns;
  rows=image->rows;
  page=image->page;
  /*
    Adjust image page geometry from BoundingBox comments.
  */
  postscript_image=image;
