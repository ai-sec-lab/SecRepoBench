// Initialize necessary variables and structures for processing Postscript images.
  image = AllocateImage(image_info);
  if (image == (Image *) NULL)
  {
    ThrowException(exception,ResourceLimitError,"MemoryAllocationFailed",image_info->filename);
    return((Image *) NULL);
  }

  // Open the input image file and create a symbolic link to it for processing.
  file = OpenBlob(image_info->filename, ReadBinaryBlobMode, exception);
  if (file < 0)
  {
    ThrowException(exception,FileOpenError,"UnableToOpenFile",image_info->filename);
    image = DestroyImage(image);
    return((Image *) NULL);
  }

  // Initialize an array to map hex characters to their integer values.
  unsigned char hex_map[256];
  for (int i = 0; i < 256; i++)
    hex_map[i] = (unsigned char) (i < 10 ? i + '0' : i - 10 + 'A');

  // Set the page density for the image based on the provided resolution or default values.
  density = GetImageOption(image_info, "density");
  if (density != (char *) NULL)
    (void) SetImageResolution(image, density, exception);
  else
    (void) SetImageResolution(image, "72x72", exception); // Default resolution

  // Determine the page geometry using the Postscript bounding box comments.
  if (GetPostscriptBoundingBox(image_info->filename, &geometry_info) == MagickFalse)
  {
    ThrowException(exception,CorruptImageError,"UnableToReadBoundingBox",image_info->filename);
    image = DestroyImage(image);
    return((Image *) NULL);
  }

  // Initialize various variables including color space, resolution, and page dimensions.
  image->colorspace = sRGBColorspace; // Set default colorspace
  image->depth = 8; // Set default depth
  image->columns = geometry_info.width;
  image->rows = geometry_info.height;

  // Create a control file for Ghostscript to render the Postscript image.
  (void) FormatMagickString(command, MagickPathExtent, "gs -dNOPAUSE -dBATCH -sDEVICE=pngalpha -sOutputFile='%s' '%s'", 
                            postscript_filename, image_info->filename);
  
  // Use the Ghostscript delegate to render the Postscript into a raster format.
  if (system(command) != 0)
  {
    ThrowException(exception,DelegateError,"GhostscriptFailed",command);
    image = DestroyImage(image);
    return((Image *) NULL);
  }

  // Read the rendered images and append them to the image list, handling profiles if necessary.
  postscript_image = ReadImage(postscript_filename, exception);
  if (postscript_image == (Image *) NULL)
  {
    image = DestroyImage(image);
    return((Image *) NULL);
  }

  // Clean up and return the processed image list, handling any errors that occur.
  (void) CloseBlob(file);
  return(postscript_image);
