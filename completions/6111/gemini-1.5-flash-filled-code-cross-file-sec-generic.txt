postscript_image=ReadImage(read_info,exception);
  if (postscript_image == (Image *) NULL)
    return((Image *) NULL);
  if (image_info->ping)
    {
      postscript_image=DestroyImage(postscript_image);
      return(GetFirstImageInList(postscript_image));
    }
  columns=postscript_image->columns;
  rows=postscript_image->rows;
  page=postscript_image->page;
  image=NewImageList();
  do
  {
    /*
      Handle multiple pages in the Postscript file.
    */
    (void) CopyMagickString(filename,postscript_image->filename,
      MagickPathExtent);
    (void) FormatMagickString(command,MagickPathExtent,
      "gs -q -dNOPAUSE -dBATCH -dSAFER -dMaxBitmap=512000000"
       " -sDEVICE=ppm -sOutputFile=%s%s%s %s",
       image_info->temporary,filename,image_info->filename,filename);
    file=popen(command,"r");
    if (file == (int) NULL)
      ThrowReaderException(FileOpenError,"UnableToExecuteGhostscript");
    (void) FormatMagickString(filename,MagickPathExtent,"%s%s",
      image_info->temporary,"image");
    next=ReadImage(image_info,exception);
    if (next == (Image *) NULL)
      {
        (void) pclose(file);
        ThrowReaderException(FileOpenError,"UnableToReadImage");
      }
    (void) pclose(file);
    AppendImageToList(&image,next);
    if (GetNextImageInList(next) != (Image *) NULL)
      {
        /*
          More pages.
        */
        postscript_image=DestroyImage(postscript_image);
        continue;
      }
    postscript_image=DestroyImage(postscript_image);
    break;
  } while (1);
  if (image == (Image *) NULL)
    return((Image *) NULL);
  /*
    Remove temporary files.
  */
  (void) RemoveFiles(image_info->temporary,filename,exception);
  /*
    Correct image orientation.
  */
  for (next=GetFirstImageInList(image); next != (Image *) NULL; )
  {
    if (next->columns > next->rows)
      {
        (void) SetImageOrientation(next,UndefinedOrientation);
      }
    next=GetNextImageInList(next);
  }
