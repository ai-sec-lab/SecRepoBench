// Initialize the image list and read the Postscript image
  image=NewImageList();
  read_info=CloneImageInfo(image_info);
  (void) CopyMagickString(read_info->filename,filename,MagickPathExtent);
  postscript_image=ReadImage(read_info,exception);
  read_info=DestroyImageInfo(read_info);
  if (postscript_image == (Image *) NULL)
    return((Image *) NULL);

  // Append the Postscript image to the image list
  AppendImageToList(&image,postscript_image);

  // Set the scene number for each image in the list
  scene=0;
  for (next=GetFirstImageInList(image); next != (Image *) NULL; )
  {
    next->scene=scene++;
    next=GetNextImageInList(next);
  }

  // Return the first image in the list
  return(GetFirstImageInList(image));
