// Read the integer value from the template string
      int value = 0;
      while (tmpl->idx < tlen && ISDIGIT(tptr[tmpl->idx])) {
        value = value * 10 + (tptr[tmpl->idx] - '0');
        tmpl->idx++;
        // Check for overflow
        if (value < 0) {
          mrb_raisef(mrb, E_RUNTIME_ERROR, "Template length is too large");
        }
      }
      count = value;
      // Decrement the index to account for the last increment in the while loop
      tmpl->idx--;
