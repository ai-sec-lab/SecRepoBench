// Check for overflow
    if (unicodeLength > (SizeOfTag / sizeof(wchar_t)))
        goto Done;

    // Allocate memory for Unicode string
    UnicodeString = (wchar_t*) _cmsMalloc(self->ContextID, (unicodeLength + 1) * sizeof(wchar_t));
    if (UnicodeString == NULL) goto Error;

    // Read Unicode string
    if (io->Read(io, UnicodeString, sizeof(wchar_t), unicodeLength) != unicodeLength) goto Error;
    UnicodeString[unicodeLength] = 0;  // Null-terminate the string

    // Set the MLU entry
    if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) goto Error;

    // Free allocated memory
    _cmsFree(self->ContextID, UnicodeString);
    UnicodeString = NULL;

    // Read ScriptCode Code
    if (SizeOfTag < sizeof(cmsUInt16Number)) goto Done;
    if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Done;
    SizeOfTag -= sizeof(cmsUInt16Number);

    // Read ScriptCode Count
    if (SizeOfTag < sizeof(cmsUInt8Number)) goto Done;
    if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Done;
    SizeOfTag -= sizeof(cmsUInt8Number);

    // Ignore ScriptCode strings
    for (i = 0; i < ScriptCodeCount; i++) {
        if (SizeOfTag < sizeof(cmsUInt16Number)) goto Done;
        if (!_cmsReadUInt16Number(io, &Dummy)) goto Done;
        SizeOfTag -= sizeof(cmsUInt16Number);
    }

Done:
    *nItems = 1;
    return mlu;
