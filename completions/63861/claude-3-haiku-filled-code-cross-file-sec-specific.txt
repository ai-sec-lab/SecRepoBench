// <MASK>
if (SizeOfTag < 2 * sizeof(cmsUInt32Number)) {
    goto Done;
}

if (!_cmsReadUInt32Number(io, &UnicodeCount)) {
    goto Done;
}
SizeOfTag -= sizeof(cmsUInt32Number);

if (SizeOfTag < UnicodeCount * sizeof(cmsUInt16Number)) {
    goto Done;
}

UnicodeString = (wchar_t*)_cmsMalloc(self->ContextID, (UnicodeCount + 1) * sizeof(wchar_t));
if (UnicodeString == NULL) {
    goto Error;
}

if (io->Read(io, UnicodeString, sizeof(cmsUInt16Number), UnicodeCount) != UnicodeCount) {
    _cmsFree(self->ContextID, UnicodeString);
    goto Error;
}
UnicodeString[UnicodeCount] = 0;
SizeOfTag -= UnicodeCount * sizeof(cmsUInt16Number);

if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) {
    _cmsFree(self->ContextID, UnicodeString);
    goto Error;
}
_cmsFree(self->ContextID, UnicodeString);
UnicodeString = NULL;

if (SizeOfTag >= 2 * sizeof(cmsUInt16Number)) {
    if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) {
        goto Done;
    }
    if (!_cmsReadUInt16Number(io, &Dummy)) {
        goto Done;
    }
    if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) {
        goto Done;
    }
    SizeOfTag -= 2 * sizeof(cmsUInt16Number) + sizeof(cmsUInt8Number);

    // Skip the ScriptCode data
    if (SizeOfTag < ScriptCodeCount * sizeof(cmsUInt16Number)) {
        goto Done;
    }
    SizeOfTag -= ScriptCodeCount * sizeof(cmsUInt16Number);
}

Done:
*nItems = 1;
return mlu;

Error:
if (Text) _cmsFree(self->ContextID, (void*)Text);
if (UnicodeString) _cmsFree(self->ContextID, UnicodeString);
if (mlu) cmsMLUfree(mlu);
return NULL;
