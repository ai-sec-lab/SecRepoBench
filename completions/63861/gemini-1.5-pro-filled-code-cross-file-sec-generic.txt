if (UnicodeCount > 0 && SizeOfTag >= UnicodeCount * sizeof(cmsUInt16Number)) {

        UnicodeString = (wchar_t*) _cmsMalloc(self ->ContextID, UnicodeCount * sizeof(wchar_t));
        if (UnicodeString == NULL) goto Error;

        if (io ->Read(io, UnicodeString, sizeof(cmsUInt16Number), UnicodeCount) != UnicodeCount) goto Error;
        SizeOfTag -= UnicodeCount * sizeof(cmsUInt16Number);

        if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) goto Error;

        _cmsFree(self ->ContextID, (void*) UnicodeString);
        UnicodeString = NULL;

    }

Done:
    // Read and discard optional ScriptCode elements
    while (SizeOfTag >= 3) {
        if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Error;
        if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Error;
        SizeOfTag -= 3;

        if (SizeOfTag < ScriptCodeCount * 2) goto Error;

        for (i = 0; i < ScriptCodeCount; i++) {
            if (!_cmsReadUInt16Number(io, &Dummy)) goto Error;
        }
        SizeOfTag -= ScriptCodeCount * 2;
    }


    *nItems = 1;
    return mlu;
