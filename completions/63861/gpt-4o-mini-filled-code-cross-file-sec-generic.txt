// Read the Unicode string length
    if (SizeOfTag < UnicodeCount * sizeof(cmsUInt16Number)) goto Done;

    // Allocate memory for the Unicode string
    UnicodeString = (wchar_t*) _cmsMalloc(self->ContextID, (UnicodeCount + 1) * sizeof(wchar_t));
    if (UnicodeString == NULL) goto Error;

    // Read the Unicode string
    if (io->Read(io, UnicodeString, sizeof(cmsUInt16Number), UnicodeCount) != UnicodeCount) goto Error;
    SizeOfTag -= UnicodeCount * sizeof(cmsUInt16Number);

    // Ensure the Unicode string is null-terminated
    UnicodeString[UnicodeCount] = 0;

    // Set the MLU entry for the Unicode string
    if (!cmsMLUsetUnicode(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) goto Error;
    _cmsFree(self->ContextID, (void*) UnicodeString);
    UnicodeString = NULL;

    // Check for optional ScriptCode elements
    if (SizeOfTag >= sizeof(cmsUInt8Number)) {
        if (io->Read(io, &ScriptCodeCount, sizeof(cmsUInt8Number), 1) != 1) goto Done;
        SizeOfTag -= sizeof(cmsUInt8Number);

        for (i = 0; i < ScriptCodeCount; i++) {
            if (SizeOfTag < sizeof(cmsUInt16Number) * 2) goto Done;
            if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Done;
            if (!_cmsReadUInt16Number(io, &Dummy)) goto Done;
            SizeOfTag -= sizeof(cmsUInt16Number) * 2;
        }
    }

Done:
    *nItems = 1; // Indicate that one item has been read successfully
    return mlu;

Error:
    if (UnicodeString) _cmsFree(self->ContextID, (void*) UnicodeString);
    if (mlu) cmsMLUfree(mlu);
    return NULL;
