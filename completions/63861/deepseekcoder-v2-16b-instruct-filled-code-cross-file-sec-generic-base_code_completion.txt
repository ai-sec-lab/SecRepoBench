// Read and handle the Unicode string component of a text description. 
    UnicodeString = (wchar_t*) _cmsMalloc(self ->ContextID, UnicodeCount * sizeof(wchar_t));
    if (UnicodeString == NULL) goto Error;

    // Read the Unicode string
    if (io ->Read(io, UnicodeString, sizeof(wchar_t), UnicodeCount) != UnicodeCount) goto Error;
    SizeOfTag -= UnicodeCount * sizeof(wchar_t);

    // Set the MLU entry for Unicode string
    if (!cmsMLUsetUnicode(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) goto Error;
    _cmsFree(self ->ContextID, (void*) UnicodeString);
    UnicodeString = NULL;

    // Skip ScriptCode elements if present
    if (SizeOfTag < sizeof(ScriptCodeCode) + sizeof(ScriptCodeCount)) goto Done;
    if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Done;
    if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Done;
    SizeOfTag -= sizeof(ScriptCodeCode) + sizeof(ScriptCodeCount);

    // Skip the ScriptCode data
    if (io ->Seek(io, ScriptCodeCount * sizeof(ScriptCodeCode)) == FALSE) goto Done;
    SizeOfTag -= ScriptCodeCount * sizeof(ScriptCodeCode);

    // Return the populated MLU object
    return mlu;

Error:
    if (Text) _cmsFree(self ->ContextID, (void*) Text);
    if (mlu) cmsMLUfree(mlu);
    if (UnicodeString) _cmsFree(self ->ContextID, (void*) UnicodeString);
    return NULL;

Done:
    if (Text) _cmsFree(self ->ContextID, (void*) Text);
    if (mlu) cmsMLUfree(mlu);
    if (UnicodeString) _cmsFree(self ->ContextID, (void*) UnicodeString);
    return NULL;
