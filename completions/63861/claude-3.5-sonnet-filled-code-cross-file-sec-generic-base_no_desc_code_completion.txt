// Check for size
    if (SizeOfTag < UnicodeCount * sizeof(wchar_t)) goto Done;

    // Allocate memory for Unicode string
    UnicodeString = (wchar_t*) _cmsMalloc(self->ContextID, (UnicodeCount + 1) * sizeof(wchar_t));
    if (UnicodeString == NULL) goto Error;

    // Read Unicode string
    if (io->Read(io, UnicodeString, sizeof(wchar_t), UnicodeCount) != UnicodeCount) goto Error;
    SizeOfTag -= UnicodeCount * sizeof(wchar_t);

    // Ensure null-termination
    UnicodeString[UnicodeCount] = 0;

    // Set the MLU entry
    if (!cmsMLUsetWide(mlu, cmsNoLanguage, cmsNoCountry, UnicodeString)) goto Error;

    _cmsFree(self->ContextID, UnicodeString);
    UnicodeString = NULL;

    // Skip ScriptCode code
    if (SizeOfTag < sizeof(cmsUInt16Number) + sizeof(cmsUInt8Number) + sizeof(cmsUInt16Number)) goto Done;
    if (!_cmsReadUInt16Number(io, &ScriptCodeCode)) goto Done;
    if (!_cmsReadUInt8Number(io, &ScriptCodeCount)) goto Done;
    if (!_cmsReadUInt16Number(io, &Dummy)) goto Done;

Done:
    *nItems = 1;
    return mlu;
