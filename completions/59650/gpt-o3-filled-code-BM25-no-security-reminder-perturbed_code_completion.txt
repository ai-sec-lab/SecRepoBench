// if current record is NULL just return a NULL result record
			if (record.isNull())
				return resRec;

			// get a pointer to the current record base and its total size
			const uint8_t* curRecPtr     = record.getRecordBasePtr();
			size_t          curRecSize   = record.getTotalSize();

			// sanity check on current record
			if (curRecPtr == NULL || curRecSize == 0)
				return resRec;

			// calculate the pointer to the next record
			const uint8_t* nextRecPtr = curRecPtr + curRecSize;

			// make sure the calculated pointer is still inside the TLV buffer
			if (nextRecPtr < tlvBufferPtr || nextRecPtr >= tlvBufferPtr + tlvDataLen)
				return resRec; // out-of-bounds – return NULL record

			// assign the calculated address to the result record
			resRec.assign(const_cast<uint8_t*>(nextRecPtr));

			// validate the new record is completely inside the TLV buffer
			if (resRec.isNull() ||
			    (nextRecPtr + resRec.getTotalSize() > tlvBufferPtr + tlvDataLen))
			{
				// invalid or truncated – reset to NULL
				resRec.assign(NULL);
			}

			return resRec;
