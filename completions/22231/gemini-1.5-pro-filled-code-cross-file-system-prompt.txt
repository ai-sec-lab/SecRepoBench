            s->ref_start = s->hdr->ref_seq_start;
            s->ref_end = s->hdr->ref_seq_start + s->hdr->ref_seq_span - 1;

                hts_log_error("Embedded reference is smaller than expected (%"PRId64" vs %"PRId64")",
                              s->hdr->ref_seq_span, (int64_t)b->uncomp_size);
                return -1;
