s->ref = b->data;
s->ref_start = s->hdr->ref_seq_start;
s->ref_end = s->hdr->ref_seq_start + s->hdr->ref_seq_span - 1;

if (s->hdr->ref_seq_span > b->uncomp_size) {
    hts_log_error("Reference sequence span (%"PRId64") is larger than "
                  "available uncompressed data (%d)",
                  s->hdr->ref_seq_span, b->uncomp_size);
    return -1;
}
