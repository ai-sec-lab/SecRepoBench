if (sURL[7] == '/' && sURL[9] == ':') {
    // Windows-style path with drive letter, remove "file:///" and shift the string
    memmove(sURL, sURL + 8, strlen(sURL) - 7);
} else {
    // Unix-style path or Windows-style path without drive letter, remove "file://"
    memmove(sURL, sURL + 7, strlen(sURL) - 6);
}
