char *drive_letter = NULL;
if (sURL[7] == '/') {
    drive_letter = sURL + 8; // Point to the character after "file:///"
} else if (sURL[8] == ':' && sURL[9] == '/') {
    drive_letter = sURL + 8; // Windows drive letter case
}

if (drive_letter) {
    // Move the rest of the string back to remove the "file://" prefix
    size_t prefix_length = (drive_letter - sURL) + 1; // +1 to include the drive letter
    size_t new_length = strlen(sURL) - prefix_length + 1; // +1 for null terminator
    char *new_path = (char *)malloc(new_length);
    if (new_path) {
        strcpy(new_path, drive_letter); // Copy the adjusted path
        gf_url_free(sURL); // Free the original URL
        sURL = new_path; // Update sURL to point to the new path
    }
}
