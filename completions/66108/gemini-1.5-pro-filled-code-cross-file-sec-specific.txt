if (bytes_skipped + size <= remain) {
	if (sync[size] != 0xFF) {
		GF_LOG(GF_LOG_DEBUG, GF_LOG_MEDIA, ("[MP3Dmx] invalid frame (no sync byte after frame) - resyncing\n"));
	} else if ((remain > size + 3) && (sync[size + 1] == 'T') && (sync[size + 2] == 'A') && (sync[size + 3] == 'G')) {
		skip_id3v1 = GF_TRUE;
	}
}
