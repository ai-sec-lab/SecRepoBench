if (bytes_skipped + size > remain) {
			if (!is_eos) {
				GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] truncated frame of size %u (remains %d)\n", size, remain-bytes_skipped));
			}
			break;
		}

		// Check for ID3v1 tag at the end of the file
		if (remain >= 128 && start[remain - 128] == 'T' && start[remain - 127] == 'A' && start[remain - 126] == 'G') {
			skip_id3v1 = GF_TRUE;
		}

		// Check for synchronization
		if (remain > bytes_skipped + size && sync[size] == 0xFF) {
			// Valid frame, ready to process
		} else {
			GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] Invalid frame detected, attempting to resync\n"));
			start += bytes_skipped + 1;
			remain -= bytes_skipped + 1;
			continue;
		}
