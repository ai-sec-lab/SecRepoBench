if (remain >= size + 1 && sync[size] == 0xFF) {
    // Frame is complete and next frame starts with sync word
} else if (remain >= size + 3 && sync[size] == 'T' && sync[size+1] == 'A' && sync[size+2] == 'G') {
    // ID3v1 tag follows, set flag to skip it
    skip_id3v1 = GF_TRUE;
} else if (is_eos) {
    // End of stream, process remaining data
} else {
    // Invalid frame or not enough data, attempt to resync
    GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] Invalid frame or not enough data, attempting to resync\n"));
    bytes_to_drop = 1;
    continue;
}
