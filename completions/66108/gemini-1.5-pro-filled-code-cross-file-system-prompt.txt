if (bytes_skipped + size + 2 <= remain) {
	if (sync[size] != 0xFF) {
		GF_LOG(GF_LOG_DEBUG, GF_LOG_MEDIA, ("[MP3Dmx] Invalid frame, resyncing\n"));
		bytes_to_drop = 1;
		goto drop_byte;
	}
	if ((sync[size + 1] == 'T') && (sync[size + 2] == 'A') && (sync[size + 3] == 'G')) {
		skip_id3v1 = GF_TRUE;
	}
