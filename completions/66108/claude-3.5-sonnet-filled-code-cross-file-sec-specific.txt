if (remain >= size + 1) {
    if (sync[size] == 0xFF) {
        // Next frame starts with sync word, packet is ready
    } else if (remain >= size + 3 && sync[size] == 'T' && sync[size+1] == 'A' && sync[size+2] == 'G') {
        // ID3v1 tag found, set flag to skip it
        skip_id3v1 = GF_TRUE;
    } else {
        // Invalid frame, attempt to resync
        GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] Invalid frame at offset %d, attempting to resync\n", ctx->byte_offset + bytes_skipped));
        bytes_to_drop = 1;
        continue;
    }
}
