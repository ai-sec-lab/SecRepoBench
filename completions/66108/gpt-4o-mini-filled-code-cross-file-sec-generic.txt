        if (*(sync + size) != 0xFF) {
            GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] Invalid frame detected, attempting to resync...\n"));
            break; // Exit the loop to wait for the next frame
        }

        // Check for ID3v1 tag at the end of the frame
        if (remain >= 128 && memcmp(sync + size, "TAG", 3) == 0) {
            skip_id3v1 = GF_TRUE; // Set flag to skip ID3v1 tag
