if (remain >= size + 1 && sync[size] == 0xFF) {
    // Check for ID3v1 tag
    if (remain >= size + 128 && sync[size] == 'T' && sync[size + 1] == 'A' && sync[size + 2] == 'G') {
        skip_id3v1 = GF_TRUE;
    }
} else {
    GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] Invalid frame detected, attempting to resync\n"));
    start += bytes_skipped + 1;
    remain -= bytes_skipped + 1;
    continue;
}
