/* get size of current frame */
		size = gf_mp3_frame_size(ctx->hdr);
		if (!size) {
			/* invalid header, skip one byte and try again */
			GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] invalid MP3 header - resyncing\n"));
			bytes_to_drop = 1;
			goto drop_byte;
		}

		/* we have at least the size of the frame ? */
		if (remain > offsetskipped + size) {
			u32 next_off = offsetskipped + size;

			/* look at the byte just after the frame for resync check */
			if (start[next_off] != 0xFF) {
				/* could be an ID3v1 tag ("TAG" – 128 bytes) */
				if ((remain - next_off >= 3) &&
				    (start[next_off] == 'T') &&
				    (start[next_off + 1] == 'A') &&
				    (start[next_off + 2] == 'G')) {
					skip_id3v1 = GF_TRUE;
					GF_LOG(GF_LOG_INFO, GF_LOG_MEDIA, ("[MP3Dmx] found ID3v1 tag – skipping\n"));
				} else {
					/* not a valid sync nor an ID3v1 tag, warn and try to resync */
					GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA,
					       ("[MP3Dmx] frame not followed by sync byte – resyncing (remain %u, size %u)\n",
					        remain, size));
					bytes_to_drop = 1;
					goto drop_byte;
				}
			}
		}
