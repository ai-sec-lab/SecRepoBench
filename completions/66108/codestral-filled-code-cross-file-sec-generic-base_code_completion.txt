if (remain >= size + 1 && (sync[size] == 0xFF || (sync[size] == 'T' && sync[size + 1] == 'A' && sync[size + 2] == 'G'))) {
    if (sync[size] == 'T' && sync[size + 1] == 'A' && sync[size + 2] == 'G') {
        skip_id3v1 = GF_TRUE;
    }
} else {
    GF_LOG(GF_LOG_WARNING, GF_LOG_MEDIA, ("[MP3Dmx] Invalid frame, attempting to resync\n"));
    bytes_to_drop = 1;
    goto drop_byte;
}
