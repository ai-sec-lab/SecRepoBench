if (remain >= size + 1 && sync[size] == 0xFF) {
    // Frame is complete and next frame starts with sync byte
} else if (remain >= size + 3 && sync[size] == 'T' && sync[size+1] == 'A' && sync[size+2] == 'G') {
    skip_id3v1 = GF_TRUE;
} else if (is_eos) {
    // Process last frame even if incomplete
} else {
    GF_LOG(GF_LOG_DEBUG, GF_LOG_MEDIA, ("[MP3Dmx] Incomplete frame or invalid sync, attempting to resync\n"));
    bytes_to_drop = 1;
    continue;
}
