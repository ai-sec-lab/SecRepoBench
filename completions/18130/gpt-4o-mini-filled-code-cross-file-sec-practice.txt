static inline double ConstrainCoordinate(double x)
{
  const double min_value = 0.0; // Define the minimum allowable value
  const double max_value = 1.0; // Define the maximum allowable value

  // Input validation: Ensure x is a finite number
  if (!isfinite(x)) {
    // Handle the case where x is not a finite number
    return (x < min_value) ? min_value : max_value; // Return constrained value
  }

  // Constrain x within the defined range
  if (x < min_value) {
    return min_value; // Set to minimum if below range
  } else if (x > max_value) {
    return max_value; // Set to maximum if above range
  }

  return x; // Return the original value if within range
}
