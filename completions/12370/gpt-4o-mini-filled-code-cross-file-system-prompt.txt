        nb_slices = (height + slice_height - 1) / slice_height;
        slices_info_offset = buf_size - 8; // Assuming the last 8 bytes contain slice info
        slice_height = (height + nb_slices - 1) / nb_slices;

        if (slices_info_offset < 0 || slice_height <= 0 || nb_slices <= 0) {
            return AVERROR_INVALIDDATA;
        }
    } else {
        nb_slices = 1;
        slice_height = height;
        slices_info_offset = 0;
