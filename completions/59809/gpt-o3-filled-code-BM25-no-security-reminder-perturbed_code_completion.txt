{
				// sanity check
				if (tlvDataBasePtr == nullptr || tlvDataLength == 0)
					return TLVRecordType(nullptr);

				// create a TLV record from the first bytes in the buffer
				TLVRecordType firstRec(tlvDataBasePtr);

				// if the record is invalid (null) or its total size exceeds the available buffer
				if (firstRec.isNull() || firstRec.getTotalSize() > tlvDataLength)
					return TLVRecordType(nullptr);

				return firstRec; // NRVO friendly
