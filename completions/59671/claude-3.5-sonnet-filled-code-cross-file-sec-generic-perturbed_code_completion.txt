case MV_TYPE_FIELD:
    if (s->picture_structure == PICT_FRAME) {
        /* top field */
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 1,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           block_s, macroblocky);
        /* bottom field */
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 1, 1,
                           ref_picture, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1],
                           block_s, macroblocky);
    } else {
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           2 * block_s, macroblocky >> 1);
    }
    break;
case MV_TYPE_16X8:
    for (int i = 0; i < 2; i++) {
        uint8_t *dest_y_ptr = dest_y + i * block_s * s->linesize;
        uint8_t *dest_cb_ptr = dest_cb + i * (block_s >> 1) * s->uvlinesize;
        uint8_t *dest_cr_ptr = dest_cr + i * (block_s >> 1) * s->uvlinesize;

        mpeg_motion_lowres(s, dest_y_ptr, dest_cb_ptr, dest_cr_ptr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][i][0], s->mv[dir][i][1],
                           block_s, block_s);
    }
    break;
case MV_TYPE_DMV:
    if (s->picture_structure == PICT_FRAME) {
        for (int i = 0; i < 2; i++) {
            int j = i ^ s->top_field_first;
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, j, 1,
                               ref_picture, pix_op,
                               s->mv[dir][j][0], s->mv[dir][j][1],
                               block_s, macroblocky);
        }
    } else {
        for (int i = 0; i < 2; i++) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1],
                               2 * block_s, macroblocky >> 1);
        }
    }
    break;
