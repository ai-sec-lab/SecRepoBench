case MV_TYPE_FIELD:
        for (int i = 0; i < 2; i++) {
            int field_select = s->field_select[dir][i];
            int field_based_mv = s->mv[dir][i][1] >> 1;
            mpeg_motion_lowres(s, dest_y + field_select * s->linesize * block_s,
                               dest_cb + field_select * s->uvlinesize * block_s / 2,
                               dest_cr + field_select * s->uvlinesize * block_s / 2,
                               0, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][i][0], field_based_mv,
                               2 * block_s, mb_y);
        }
        break;
    case MV_TYPE_16X8:
        for (int i = 0; i < 2; i++) {
            mpeg_motion_lowres(s, dest_y + i * s->linesize * block_s,
                               dest_cb + i * s->uvlinesize * block_s / 2,
                               dest_cr + i * s->uvlinesize * block_s / 2,
                               0, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1],
                               2 * block_s, mb_y);
        }
        break;
    case MV_TYPE_DMV:
        for (int i = 0; i < 2; i++) {
            int dmv_x = s->mv[dir][i][0] + s->dmv[dir][i][0];
            int dmv_y = s->mv[dir][i][1] + s->dmv[dir][i][1];
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               dmv_x, dmv_y,
                               2 * block_s, mb_y);
        }
        break;
