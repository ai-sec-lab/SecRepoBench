case MV_TYPE_FIELD:
        // Handle field motion vectors
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           block_s, mb_y);
        break;
    case MV_TYPE_16X8:
        // Handle 16x8 motion vectors
        for (int i = 0; i < 2; i++) {
            hpel_motion_lowres(s, dest_y + (i * block_s) * s->linesize,
                               ref_picture[0], 0, 0,
                               (2 * mb_x) * block_s,
                               (mb_y * 2 + i) * block_s,
                               s->width, s->height, s->linesize,
                               s->h_edge_pos >> lowres, s->v_edge_pos >> lowres,
                               block_s, block_s, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1]);
        }
        break;
    case MV_TYPE_DMV:
        // Handle dual-prime motion vectors
        for (int i = 0; i < 2; i++) {
            hpel_motion_lowres(s, dest_y + (i * block_s) * s->linesize,
                               ref_picture[0], 0, 0,
                               (2 * mb_x) * block_s,
                               (mb_y * 2 + i) * block_s,
                               s->width, s->height, s->linesize,
                               s->h_edge_pos >> lowres, s->v_edge_pos >> lowres,
                               block_s, block_s, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1]);
        }
        break;
