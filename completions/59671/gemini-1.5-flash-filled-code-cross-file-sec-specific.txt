case MV_TYPE_FIELD:
        if (s->picture_structure == PICT_FRAME) {
            field_motion_lowres(s, dest_y, dest_cb, dest_cr, dir, ref_picture, pix_op);
        } else {
            field_motion_lowres(s, dest_y, dest_cb, dest_cr, dir, ref_picture, pix_op);
        }
        break;

            mpeg_motion_lowres(s, dest_y + i * 8 * block_s, dest_cb, dest_cr,
                               dir, ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1],
                               block_s, mb_y + i * 8);
        }
        break;
