case MV_TYPE_FIELD:
        if (s->picture_structure == PICT_FRAME) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               2 * block_s, mb_y);

        } else {
            for (int i = 0; i < 2; i++) {
                uint8_t **ref2picture;

                if (s->field_select[dir][i] == 0)
                    ref2picture = ref_picture;
                else
                    ref2picture = s->current_picture_ptr->f->data;

                mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                                   i, i, 0,
                                   ref2picture, pix_op,
                                   s->mv[dir][i][0], s->mv[dir][i][1],
                                   block_s, mb_y >> 1);
                dest_y += s->linesize * block_s;
            }
        }
        break;
    case MV_TYPE_16X8:
        for (int i = 0; i < 2; i++) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, i,
                               ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1],
                               2 * block_s, mb_y >> 1);
            dest_y += s->linesize * block_s;
        }
        break;
    case MV_TYPE_DMV:
        if (s->picture_structure == PICT_FRAME) {
            for (int i = 0; i < 2; i++) {
                mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                                   0, 0, 0,
                                   ref_picture, pix_op,
                                   s->mv[dir][i][0], s->mv[dir][i][1],
                                   2 * block_s, mb_y);

                s->mv[dir][0][0] = (s->mv[dir][0][0] + s->mv[dir][1][0] + 1) >> 1;
                s->mv[dir][0][1] = (s->mv[dir][0][1] + s->mv[dir][1][1] + 1) >> 1;
            }
        } else {
            for (int i = 0; i < 2; i++) {
                mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                                   i, i, 0,
                                   ref_picture, pix_op,
                                   s->mv[dir][i][0], s->mv[dir][i][1],
                                   block_s, mb_y >> 1);
                dest_y += s->linesize * block_s;
            }
        }
        break;
