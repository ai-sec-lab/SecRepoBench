case MV_TYPE_FIELD:
        if (s->picture_structure == PICT_FRAME) {
            /* top field */
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               2 * block_s, mb_y);
            /* bottom field */
            mpeg_motion_lowres(s, dest_y + s->linesize * block_s, dest_cb, dest_cr,
                               1, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][1][0], s->mv[dir][1][1],
                               2 * block_s, mb_y);
        } else {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               2 * block_s, mb_y >> 1);

        }
        break;
    case MV_TYPE_16X8:
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           2 * block_s, mb_y);
        mpeg_motion_lowres(s, dest_y + block_s * s->linesize, dest_cb, dest_cr,
                           1, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1],
                           2 * block_s, mb_y);

        break;
    case MV_TYPE_DMV:
        if (s->picture_structure == PICT_FRAME) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               2 * block_s, mb_y);

            /* Add prediction for opposite field */
            s->mv[dir][0][1] += (s->mv[dir][1][1] - s->mv[dir][0][1]) / 2;
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, 1,
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               2 * block_s, mb_y);
        } else {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               2 * block_s, mb_y >> 1);
        }
        break;
