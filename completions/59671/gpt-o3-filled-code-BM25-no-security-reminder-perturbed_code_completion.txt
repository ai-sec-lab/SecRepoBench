case MV_TYPE_16X8: /* 2 independent 16x8 vectors */
        for (int i = 0; i < 2; i++) {
            const int16_t mv_x = s->mv[dir][i][0];
            const int16_t mv_y = s->mv[dir][i][1];

            /* offset the destination pointers to the top of the corresponding
             * 8-line (block_s) stripe */
            uint8_t *y  = dest_y  + i * block_s * s->linesize;
            uint8_t *cb = dest_cb + (i * (block_s >> s->chroma_y_shift)) * s->uvlinesize;
            uint8_t *cr = dest_cr + (i * (block_s >> s->chroma_y_shift)) * s->uvlinesize;

            mpeg_motion_lowres(s, y, cb, cr,
                               0,                /* field_based   */
                               0,                /* bottom_field  */
                               0,                /* field_select  */
                               ref_picture, pix_op,
                               mv_x, mv_y,
                               block_s,          /* height */
                               macroblocky);
        }
        break;
    case MV_TYPE_FIELD: /* 2 field vectors (top / bottom) */
        for (int i = 0; i < 2; i++) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               1,                 /* field_based  */
                               i,                 /* bottom_field */
                               s->field_select[dir][i],
                               ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1],
                               2 * block_s,       /* full height inside the field */
                               macroblocky);
        }
        break;
    case MV_TYPE_DMV: /* dual-prime – fall back to a single 16×16 vector */
        /* Dual-prime is rarely used; for low-res playback we simply decode it
         * like a normal 16×16 vector.  This preserves binary compatibility
         * while keeping the implementation small. */
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           2 * block_s, macroblocky);
        break;
