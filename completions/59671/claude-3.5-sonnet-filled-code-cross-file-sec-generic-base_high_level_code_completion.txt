case MV_TYPE_FIELD:
    if (s->picture_structure == PICT_FRAME) {
        /* top field */
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 1,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           block_s, mb_y);
        /* bottom field */
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 1, 1,
                           ref_picture, pix_op,
                           s->mv[dir][1][0], s->mv[dir][1][1],
                           block_s, mb_y);
    } else {
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           2 * block_s, mb_y >> 1);
    }
    break;
case MV_TYPE_16X8:
    for (int i = 0; i < 2; i++) {
        uint8_t ** ref2picture = ref_picture;

        if (s->picture_structure == PICT_FRAME) {
            if (s->field_select[dir][i] == 1)
                ref2picture = ref_picture + s->linesize;
        }

        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, i, 0,
                           ref2picture, pix_op,
                           s->mv[dir][i][0], s->mv[dir][i][1] + 2 * i * block_s,
                           block_s, mb_y);

        dest_y  += 2 * block_s * s->linesize;
        dest_cb += (2 * block_s >> s->chroma_y_shift) * s->uvlinesize;
        dest_cr += (2 * block_s >> s->chroma_y_shift) * s->uvlinesize;
    }
    break;
case MV_TYPE_DMV:
    if (s->picture_structure == PICT_FRAME) {
        for (int i = 0; i < 2; i++) {
            int j = i ^ 1;
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, i, 1,
                               ref_picture, pix_op,
                               s->mv[dir][j][0], s->mv[dir][j][1],
                               block_s, mb_y);
        }
    } else {
        for (int i = 0; i < 2; i++) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1],
                               2 * block_s, mb_y >> 1);
        }
    }
    break;
