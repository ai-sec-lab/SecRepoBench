case MV_TYPE_FIELD:
        // Handle field motion vectors
        mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                           0, 0, 0,
                           ref_picture, pix_op,
                           s->mv[dir][0][0], s->mv[dir][0][1],
                           block_s, mb_y);
        break;
    case MV_TYPE_16X8:
        // Handle 16x8 motion vectors
        for (int i = 0; i < 2; i++) {
            hpel_motion_lowres(s, dest_y + (i * 8) * block_s,
                               ref_picture[0], 0, 0,
                               (2 * mb_x) * block_s,
                               (2 * mb_y + i) * block_s,
                               s->width, s->height, s->linesize,
                               s->h_edge_pos >> lowres, s->v_edge_pos >> lowres,
                               block_s, block_s, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1]);
        }
        break;
    case MV_TYPE_DMV:
        // Handle dual-prime motion vectors
        for (int i = 0; i < 2; i++) {
            hpel_motion_lowres(s, dest_y + (i * 8) * block_s,
                               ref_picture[0], 0, 0,
                               (2 * mb_x) * block_s,
                               (2 * mb_y + i) * block_s,
                               s->width, s->height, s->linesize,
                               s->h_edge_pos >> lowres, s->v_edge_pos >> lowres,
                               block_s, block_s, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1]);
        }
        // Average the motion vectors for dual-prime
        mx = (s->mv[dir][0][0] + s->mv[dir][1][0]) / 2;
        my = (s->mv[dir][0][1] + s->mv[dir][1][1]) / 2;
        if (!CONFIG_GRAY || !(s->avctx->flags & AV_CODEC_FLAG_GRAY))
            chroma_4mv_motion_lowres(s, dest_cb, dest_cr, ref_picture,
                                     pix_op, mx, my);
        break;
