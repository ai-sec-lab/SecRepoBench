case MV_TYPE_FIELD:
        if (s->picture_structure == PICT_FRAME) {
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               2 * block_s, mb_y);
        } else {
            int field_select = (s->field_select[mb_x + mb_y * s->mb_stride] >> dir) & 1;
            int field_offset = (s->picture_structure == PICT_BOTTOM_FIELD) ? s->linesize : 0;
            mpeg_motion_lowres(s, dest_y + field_offset * block_s, dest_cb, dest_cr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][0][0], s->mv[dir][0][1],
                               2 * block_s, mb_y);
        }
        break;
    case MV_TYPE_16X8:
        for (int i = 0; i < 2; i++) {
            mpeg_motion_lowres(s, dest_y + i * 8 * s->linesize, dest_cb, dest_cr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               s->mv[dir][i][0], s->mv[dir][i][1],
                               2 * block_s, mb_y + i * 8);
        }
        break;
    case MV_TYPE_DMV:
        {
            int mx = (s->mv[0][0][0] + s->mv[1][0][0]) / 2;
            int my = (s->mv[0][0][1] + s->mv[1][0][1]) / 2;
            mpeg_motion_lowres(s, dest_y, dest_cb, dest_cr,
                               0, 0, 0,
                               ref_picture, pix_op,
                               mx, my,
                               2 * block_s, mb_y);
        }
        break;
