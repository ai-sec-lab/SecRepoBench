/*
     * If this dictionary has a shared sub-dictionary, try to find the
     * requested string there.  We must recompute the hash value using
     * the seed of the sub-dictionary.
     */
    if ((dict->subdict != NULL) && (dict->subdict->size > 0)) {
        unsigned subHash;
        int subFound = 0;
        size_t tmpLen, tmpPlen;

        if (pref == NULL) {
            subHash = xmlDictHashName(dict->subdict->seed, name, maxLen,
                                      &tmpLen);
        } else {
            subHash = xmlDictHashQName(dict->subdict->seed, pref, name,
                                       &tmpPlen, &tmpLen);
        }

        entry = xmlDictFindEntry(dict->subdict, pref, name, klen,
                                 subHash, &subFound);
        if (subFound)
            return(entry);
    }
