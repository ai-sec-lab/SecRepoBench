    if (!SkipSpaces(&szCurrent)) {

        SMDI_PARSE_RETURN;

    const char* szNameStart = szCurrent;

    }

        SMDI_PARSE_RETURN;
    }

    // Extract the bone name
    bone.mName = std::string(szNameStart, szCurrent - szNameStart);
    ++szCurrent; // Move past the closing quote
