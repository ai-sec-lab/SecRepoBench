if (lookup_index >= 0 && lookup_index < done_lookups_glyph_count->size()) {
  if (!done_lookups_glyph_count->get(lookup_index)) {
    done_lookups_glyph_count->set(lookup_index, true);
    done_lookups_glyph_set->clear(lookup_index);
    hb_set_t* glyph_set = get_glyph_set(lookup_index);
    if (hb_set_contains(glyph_set, parent_active_glyphs())) {
      return true;
    }
  }
}
return false;
