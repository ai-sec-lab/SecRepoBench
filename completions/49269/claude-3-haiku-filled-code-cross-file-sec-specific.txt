size_t digits(mpz_t *x) {
  size_t i;
  for (i = x->sz - 1; i != SIZE_MAX && x->p[i] == 0; i--) {
    // Iterate over the limbs from the most significant to the least significant,
    // counting the number of non-zero limbs until a non-zero limb is found or
    // all limbs have been checked.
  }
  return i + 1;
}
