if (x == NULL || x->p == NULL) {
    return 0; // Return 0 if the input pointer is NULL or its internal pointer is NULL
  }

  size_t count = 0;
  for (size_t i = x->sz; i > 0; i--) {
    if (x->p[i - 1] != 0) {
      count = i; // Count the number of significant digits
      break;
    }
  }
  return count;
