size_t i;

	/*
	 * The first byte of the constant (the one that determined the
	 * switch() branch in json_parse()) has already been consumed.
	 * `uc` now points at the second byte and `length` equals the
	 * number of bytes that still have to match (e.g. 4 for "true").
	 * Compare the remaining bytes one-by-one.
	 */
	for (i = 1; i < length; i++, uc++) {
		/* Do not run past the end of the input buffer */
		if (uc == ue || *uc != (unsigned char)str[i]) {
			DPRINTF("Bad const: ", uc, *ucp);
			/* rewind caller's pointer to point at the failure spot */
			*ucp = uc;
			return 0;
		}
	}
