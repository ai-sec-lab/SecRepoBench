/*
		      Save the list of server names found in the certificate so that we can later
		      compare it against the SNI (flow->host_server_name).

		      NOTE
		      ----
		      We store them as a comma separated list as this is lightweight and
		      still allows an easy string comparison in the code below
		      (see the “Certificate mismatch” check).
		    */
		    if(flow->protos.tls_quic.server_names == NULL) {
		      /* First entry */
		      flow->protos.tls_quic.server_names = ndpi_strdup(dNSName);
		    } else {
		      /* Additional entry: append “,<name>” */
		      size_t old_len = strlen(flow->protos.tls_quic.server_names);
		      size_t add_len = strlen(dNSName);
		      char *tmp = ndpi_malloc(old_len + 1 /* comma */ + add_len + 1 /* \0 */);

		      if(tmp != NULL) {
			memcpy(tmp, flow->protos.tls_quic.server_names, old_len);
			tmp[old_len] = ',';
			memcpy(&tmp[old_len + 1], dNSName, add_len);
			tmp[old_len + 1 + add_len] = '\0';

			ndpi_free(flow->protos.tls_quic.server_names);
			flow->protos.tls_quic.server_names = tmp;
		      }
