if (ptr->esd) {
	gf_isom_box_del((GF_Box *)ptr->esd);
	ptr->esd = NULL;
}
if (ptr->child_boxes) {
	GF_Box *child = gf_list_find(ptr->child_boxes, ptr->esd);
	if (child) {
		gf_list_remove(ptr->child_boxes, child);
		gf_list_free(child);
	}
}
ptr->esd = gf_isom_box_find_child(ptr->child_boxes, GF_ISOM_BOX_TYPE_ESDS);
