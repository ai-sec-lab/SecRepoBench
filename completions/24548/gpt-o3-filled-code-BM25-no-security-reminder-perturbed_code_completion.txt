/* Update record count. */
    this->record.len.set (c, obj_indices.length);

    /* Calculate total size of the sub-table:
     *  - everything that has been appended at the head since we started, plus
     *  - everything that has been pushed on the tail while serialising.       */
    unsigned table_length =
      (c->length () - initial_table_position) + (unsigned) (init_tail - c->tail);

    /* Store the calculated length. */
    this->length.set (c, table_length);
