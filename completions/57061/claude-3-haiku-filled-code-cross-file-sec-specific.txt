tmp = (xsltStepOpPtr) xmlRealloc(comp->steps, comp->maxStep * 2 * sizeof(xsltStepOp));

    xsltTransformError(ctxt->ctxt, ctxt->style, ctxt->elem, "xsltCompMatchAdd: memory re-allocation failure.\n");

        ctxt->style->errors++;
    return -1;
}
