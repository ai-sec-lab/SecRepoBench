tmp = (xsltStepOpPtr) xmlRealloc(comp->steps, comp->maxStep * 2 *
                                 sizeof(xsltStepOp));
if (tmp == NULL) {
    xsltGenericError(xsltGenericErrorContext,
                     "xsltCompMatchAdd: memory re-allocation failure.\n");
    if (ctxt->ctxt!= NULL) {
        ctxt->ctxt->error = 1;
    } else {
        ctxt->style->errors++;
    }
    return(-1);
}
comp->steps = tmp;
comp->maxStep *= 2;
