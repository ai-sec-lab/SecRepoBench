if (avctx->codec_id == AV_CODEC_ID_HYMT) {
    if (buf_size < 16 || buf_size - 16 < (width * height + 7) / 8)
        return AVERROR_INVALIDDATA;

    slices_info_offset = buf_size - 16;
    nb_slices = AV_RL32(avpkt->data + slices_info_offset);
    slice_height = AV_RL32(avpkt->data + slices_info_offset + 4);

    if (nb_slices == 0 || slice_height <= 0 || nb_slices > INT_MAX / 8 ||
        slices_info_offset < (unsigned)nb_slices * 8 ||
        height % slice_height != 0 || height / slice_height != nb_slices)
        return AVERROR_INVALIDDATA;
} else {
    nb_slices = 1;
    slice_height = height;
}
