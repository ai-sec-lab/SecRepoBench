if (avctx->codec_id == AV_CODEC_ID_HYMT) {
        slices_info_offset = buf_size - 4 * AV_RL32(buf + buf_size - 4);
        nb_slices = AV_RL32(buf + slices_info_offset);
        if (nb_slices > (unsigned)buf_size / 4 || nb_slices <= 0)
            return AVERROR_INVALIDDATA;

        slice_height = AV_RL32(buf + slices_info_offset + 4);

        if (slice_height <= 0 || nb_slices * slice_height > height ||
            slices_info_offset + 8 * nb_slices > buf_size)
            return AVERROR_INVALIDDATA;

    } else {
        nb_slices = 1;
        slice_height = height;
    }
