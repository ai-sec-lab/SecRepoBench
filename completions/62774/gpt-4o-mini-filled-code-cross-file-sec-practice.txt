// Calculate the byte length of the TupleVariationData
    size_t byte_length = blob->length;

    // Validate input parameters to ensure they are within expected ranges
    if (axis_count == 0 || point_count == 0 || byte_length == 0) {
        return false; // Invalid parameters
    }

    const uint8_t* data = hb_blob_get_data(blob, nullptr);
    if (data == nullptr) {
        return false; // Failed to get data from blob
    }

    // Initialize the tuple iterator
    TupleVariationData::TupleIterator iterator;
    if (!iterator.initialize(data, byte_length, axis_count)) {
        return false; // Initialization failed
    }

    // Prepare a vector to hold shared indices
    std::vector<unsigned> shared_indices;
    shared_indices.reserve(point_count); // Reserve space for efficiency

    // Process the tuple variations
    for (unsigned i = 0; i < point_count; ++i) {
        // Validate the index before accessing the data
        if (i >= byte_length) {
            return false; // Out of bounds access
        }
        // Extract shared index and add to the vector
        shared_indices.push_back(iterator.get_shared_index(i));
