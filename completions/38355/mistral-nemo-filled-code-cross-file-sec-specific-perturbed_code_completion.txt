if (get_type () == Type::Extension && get_edit_count () == 0) {
  const auto& subtables = get_subtables<TSubTable>();
  std::optional<Type> first_type;
  for (const auto& subtable : subtables) {
    if (!first_type.has_value()) {
      first_type = subtable.get_type();
    } else if (subtable.get_type() != first_type.value()) {
      return_trace(false);
    }
  }
}
