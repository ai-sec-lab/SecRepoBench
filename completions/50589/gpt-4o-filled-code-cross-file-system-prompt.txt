buffer.clear();

    while (true) {
        // Check if the cache is exhausted
        if (m_cachePos >= m_cache.size()) {
            if (!readNextBlock()) {
                return false; // End of stream
            }
        }

        // Read characters from the cache until a newline is found
        while (m_cachePos < m_cache.size()) {
            T c = m_cache[m_cachePos++];
            if (c == '\n') {
                return true; // Line read successfully
            }
            if (c != '\r') { // Skip carriage return
                buffer.push_back(c);
            }
        }
    }
