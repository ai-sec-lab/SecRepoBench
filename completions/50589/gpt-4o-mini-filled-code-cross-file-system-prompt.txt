buffer.clear();
    m_cachePos = 0;

    while (true) {
        if (m_cachePos >= m_cache.size()) {
            if (!readNextBlock()) {
                return false;
            }
        }

        // Skip any end-of-line characters
        while (m_cachePos < m_cache.size() && (m_cache[m_cachePos] == '\n' || m_cache[m_cachePos] == '\r')) {
            ++m_cachePos;
        }

        // Check if we reached the end of the cache
        if (m_cachePos >= m_cache.size()) {
            return false;
        }

        // Read until the end of the line
        size_t start = m_cachePos;
        while (m_cachePos < m_cache.size() && m_cache[m_cachePos] != '\n' && m_cache[m_cachePos] != '\r') {
            ++m_cachePos;
        }

        // Copy the line into the buffer
        buffer.assign(m_cache.begin() + start, m_cache.begin() + m_cachePos);
        return true;
    }
