if (m_cachePos >= m_cache.size()) {
        if (!readNextBlock()) {
            return false;
        }
    }

    size_t i = m_cachePos;
    while (i < m_cache.size() && m_cache[i] != '\n' && m_cache[i] != '\r') {
        ++i;
    }

    if (i == m_cachePos) {
        // No data read
        return false;
    }

    buffer.assign(m_cache.begin() + m_cachePos, m_cache.begin() + i);
    m_cachePos = i + 1;
    return true;
