if (m_cachePos >= m_cache.size()) {
        if (!readNextBlock()) {
            return false;
        }
    }

    size_t i = m_cachePos;
    while (i < m_cache.size()) {
        if (m_cache[i] == '\n') {
            break;
        }
        ++i;
    }

    if (i == m_cachePos) {
        if (!readNextBlock()) {
            return false;
        }
        return getNextLine(buffer);
    }

    buffer.resize(i - m_cachePos);
    std::copy(m_cache.begin() + m_cachePos, m_cache.begin() + i, buffer.begin());
    m_cachePos = i + 1;
    return true;
