mrb_int base = 10; // Default base
  mrb_value str_value;
  
  // Check if a base argument is provided
  if (mrb_get_args(mrb, "|i", &base) == 1) {
    if (base < 2 || base > 36) {
      mrb_raise(mrb, E_ARGUMENT_ERROR, "base must be between 2 and 36");
    }
  }

  // Convert the integer to a string representation
  str_value = mrb_str_new_cstr(mrb, mrb_fixnum_to_str(mrb, mrb_fixnum(self), base));

  return str_value;
