static mrb_value
int_to_s(mrb_state *state, mrb_value self)
{
  mrb_int num;
  mrb_int base = 10;
  mrb_get_args(state, "i|i", &num, &base);
  if (base < 2 || base > 36) {
    mrb_raise(state, E_ARGUMENT_ERROR, "base must be between 2 and 36");
  }
  char buffer[64];
  mrb_snprintf(buffer, sizeof(buffer), "%d", num);
  return mrb_str_new_cstr(state, buffer);
}
