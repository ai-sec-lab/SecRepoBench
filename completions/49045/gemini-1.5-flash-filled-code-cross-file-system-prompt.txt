mrb_int base;
  mrb_int n;
  char buf[32];
  char *p;
  mrb_value v;

  mrb_get_args(mrb, "|i", &base);
  if (base == 0) base = 10;
  if (base < 2 || base > 36) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "base must be between 2 and 36");
  }
  n = mrb_fixnum(self);
  p = buf + sizeof(buf);
  *--p = '\0';
  if (n < 0) {
    *--p = '-';
    n = -n;
  }
  if (n == 0) *--p = '0';
  while (n > 0) {
    int digit = n % base;
    *--p = (digit < 10 ? '0' + digit : 'a' + digit - 10);
    n /= base;
  }
  v = mrb_str_new_cstr(mrb, p);
  return v;
