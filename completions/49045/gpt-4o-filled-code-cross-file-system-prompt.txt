mrb_int base = 10;
  mrb_get_args(mrb, "|i", &base);

  if (base < 2 || base > 36) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "base must be between 2 and 36");
  }

  mrb_int num = mrb_integer(self);
  char buf[65]; // enough to hold binary representation of a 64-bit integer
  char *p = buf + sizeof(buf);
  mrb_bool negative = FALSE;

  if (num == 0) {
    return mrb_str_new_lit(mrb, "0");
  }

  if (num < 0) {
    negative = TRUE;
    num = -num;
  }

  *--p = '\0';
  while (num > 0) {
    mrb_int remainder = num % base;
    *--p = (remainder < 10) ? '0' + remainder : 'a' + (remainder - 10);
    num /= base;
  }

  if (negative) {
    *--p = '-';
  }

  return mrb_str_new_cstr(mrb, p);
