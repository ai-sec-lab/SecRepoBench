mrb_int base = 10;
mrb_value str;
char *buf;
mrb_int len;

mrb_get_args(mrb, "|i", &base);

if (base < 2 || base > 36) {
  mrb_raisef(mrb, E_ARGUMENT_ERROR, "invalid radix %S", mrb_fixnum_value(base));
}

buf = mrb_malloc(mrb, 64);
len = mrb_int_to_str(buf, 64, mrb_fixnum(self), base);

str = mrb_str_new(mrb, buf, len);
mrb_free(mrb, buf);

return str;
