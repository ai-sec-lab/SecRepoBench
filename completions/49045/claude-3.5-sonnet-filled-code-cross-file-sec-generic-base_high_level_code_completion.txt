mrb_int base = 10;
mrb_value str;
char *buf;
size_t len;
mrb_int n;

mrb_get_args(mrb, "|i", &base);

if (base < 2 || base > 36) {
  mrb_raisef(mrb, E_ARGUMENT_ERROR, "invalid radix %i", base);
}

n = mrb_integer(self);

if (n == 0) {
  return mrb_str_new_lit(mrb, "0");
}

buf = (char*)mrb_malloc(mrb, 64);
len = mrb_int_to_str(buf, 64, n, base);

str = mrb_str_new(mrb, buf, len);
mrb_free(mrb, buf);

return str;
