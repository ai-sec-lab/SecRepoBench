def fill_masked_region(image, mask, color):
    """
    Fill masked region of an image with a given color.

    :param image: PIL Image object
    :param mask: PIL Image object representing the mask
    :param color: Tuple representing the color to fill the masked region
    :return: PIL Image object with the masked region filled
    """
    # Create a new image with the same mode and size as the input image
    filled_image = Image.new(image.mode, image.size)

    # Paste the input image onto the new image using the mask
    filled_image.paste(image, mask)

    # Get the pixel access object for the new image
    pixels = filled_image.load()

    # Iterate over each pixel in the new image
    for y in range(filled_image.size[1]):
        for x in range(filled_image.size[0]):
            # If the pixel is in the masked region, set it to the desired color
            if mask.getpixel((x, y)) == 0:
                pixels[x, y] = color

    # Return the new image with the masked region filled
    return filled_image
