        while (ch != '\0' && ch != it8->ch && ch != '\n' && ch != '\r') {
            if (it8->str[0] == 0)
                it8->str[0] = ch;
            else

            ch = NextCh(it8);
        }

        if (ch == '\n' || ch == '\r') {

            return;
        }

        if (ch == '\0') {

            return;
        }
