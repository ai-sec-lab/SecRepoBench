char quote = it8->ch;
it8->str[0] = 0;
NextCh(it8);
while (it8->ch != quote && it8->ch != '\n' && it8->ch != '\r') {
    if (it8->ch == 0) {
        SynError(it8, "Unterminated string");
        return;
    }
    if (strlen(it8->str) >= MAXSTR - 1) {
        SynError(it8, "String too long");
        return;
    }
    it8->str[strlen(it8->str) + 1] = 0;
    it8->str[strlen(it8->str)] = it8->ch;
    NextCh(it8);
}
if (it8->ch == quote) {
    it8->sy = SSTRING;
    NextCh(it8);
} else {
    SynError(it8, "Unterminated string");
}
