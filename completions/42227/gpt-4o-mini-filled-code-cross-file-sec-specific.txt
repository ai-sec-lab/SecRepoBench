char buffer[256]; // Output buffer
        size_t index = 0;

        // Clear the buffer
        memset(buffer, 0, sizeof(buffer));

        // Read characters until the closing quote or newline
        while (it8->ch != '\'' && it8->ch != '\"' && it8->ch != '\n' && it8->ch != '\r' && index < sizeof(buffer) - 1) {
            buffer[index++] = it8->ch;
            NextCh(it8);
        }

        // Null-terminate the string
        buffer[index] = '\0';

        // Check for closing quote
        if (it8->ch == '\'' || it8->ch == '\"') {
            NextCh(it8); // Move past the closing quote
            it8->str = strdup(buffer); // Store the parsed string safely
            it8->sy = SSTRING; // Set the current symbol to SSTRING
        } else {
            SynError(it8, "Closing quote expected");
        }
