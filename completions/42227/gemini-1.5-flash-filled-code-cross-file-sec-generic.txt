        it8->str[0] = 0;

        while (ch != '\0' && ch != it8->ch && ch != '\n' && ch != '\r') {
            if (it8->str_ptr >= MAX_IT8_STRING - 1) {
                SynError(it8, "String too long");
                break;
            }
            it8->str[it8->str_ptr++] = ch;
            ch = NextCh(it8);
        }
        it8->str[it8->str_ptr] = 0;
