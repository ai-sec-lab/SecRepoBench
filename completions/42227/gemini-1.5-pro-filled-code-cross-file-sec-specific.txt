char quote = it8->ch;
        cmsIT8UngetChar(it8, quote); // Put the quote char back into the stream
        size_t str_len = 0;
        char str_buffer[MAX_ID_LEN + 1]; // Allocate a buffer with a maximum size

        NextCh(it8); // Consume the initial quote char

        while (it8->ch != 0 && it8->ch != '\n' && it8->ch != '\r') {
            if (it8->ch == quote) {
                NextCh(it8); // Consume the closing quote
                str_buffer[str_len] = '\0'; // Null-terminate the string
                cmsIT8SetString(it8, str_buffer); // Set the parsed string
                it8->sy = SSTRING;
                return;
            }

            if (str_len < MAX_ID_LEN) {
                str_buffer[str_len++] = it8->ch;
            } else {
                // Handle string too long
                SynError(it8, "String too long");
                return;
            }
            NextCh(it8);
        }

        // Handle unterminated string
        SynError(it8, "Unterminated string");
