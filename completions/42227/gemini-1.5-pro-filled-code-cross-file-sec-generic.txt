        cmsIT8UngetChar(it8); // Put back the quote char
        NextCh(it8); // Move to the next char after the quote

        int i = 0;
        while (it8->ch != quote && it8->ch != '\n' && it8->ch != '\r' && it8->ch != EOF && i < sizeof(it8->str) -1) {
            it8->str[i++] = it8->ch;
            NextCh(it8);
        }
        
        if (it8->ch == EOF || it8->ch == '\n' || it8->ch == '\r') {
            SynError(it8, "Unterminated string");
            return;
        }

        if (i == sizeof(it8->str) - 1) {
           SynError(it8, "String too long");
           return;
        }

        if (it8->ch == quote) {
            it8->sy = SSTRING;
            NextCh(it8);
