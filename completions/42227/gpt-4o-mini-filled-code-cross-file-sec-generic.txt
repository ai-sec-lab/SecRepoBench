char buffer[1024]; // Buffer to hold the string being parsed
        size_t index = 0;   // Index for the buffer

        // Store the quote character
        char quote = it8->ch;
        NextCh(it8); // Move past the opening quote

        // Clear the current string buffer
        memset(buffer, 0, sizeof(buffer));

        // Read characters until we find the closing quote or a newline
        while (it8->ch != quote && it8->ch != '\n' && it8->ch != '\r' && index < sizeof(buffer) - 1) {
            buffer[index++] = it8->ch; // Append character to buffer
            NextCh(it8); // Move to the next character
        }

        // Check for closing quote
        if (it8->ch == quote) {
            buffer[index] = '\0'; // Null-terminate the string
            it8->str = strdup(buffer); // Allocate memory and copy the string
            it8->sy = SSTRING; // Set the current symbol to SSTRING
            NextCh(it8); // Move past the closing quote
        } else {
            SynError(it8, "Closing quote expected");
