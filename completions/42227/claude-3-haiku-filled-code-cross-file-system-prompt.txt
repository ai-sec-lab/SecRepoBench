char* start = it8->str;
char* end = start;
char quote = it8->ch;

// Clear the current string buffer
it8->str[0] = '\0';

// Read each character until the closing quote or a newline is encountered
while (1) {
    NextCh(it8);
    if (it8->ch == quote || it8->ch == '\n' || it8->ch == '\r') {
        break;
    }
    end++;
    *end = it8->ch;
    *end = '\0';
}

it8->sy = SSTRING;

// Move to the next character after processing
NextCh(it8);
