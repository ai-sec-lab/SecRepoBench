#include <unordered_map>

class FontTable {
private:
    std::unordered_map<hb_tag_t, unsigned int> table_index_map;

public:
    FontTable() {
        // Initialize the map with some values for demonstration purposes
        table_index_map[HB_TAG('A', 'A', 'A', 'A')] = 0;
        table_index_map[HB_TAG('B', 'B', 'B', 'B')] = 1;
        // ... more entries
    }

    bool find_table_index(hb_tag_t hb_tag, unsigned int *table_index) const {
        auto it = table_index_map.find(hb_tag);
        if (it != table_index_map.end()) {
            *table_index = it->second;
            return true;
        }
        return false;
    }
};
