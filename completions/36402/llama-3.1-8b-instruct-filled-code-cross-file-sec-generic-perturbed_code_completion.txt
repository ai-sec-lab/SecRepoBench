bool find_table_index (hb_tag_t hb_tag, unsigned int *table_index) const
  {
    // Determine the index of the table entry corresponding to the given tag.
    // Use the appropriate search method based on table entry size.
    if (sizeof(hb_tag_t) == 4) {
      // For 32-bit tags, use a binary search
      unsigned int low = 0;
      unsigned int high = table_size - 1;
      while (low <= high) {
        unsigned int mid = (low + high) / 2;
        if (table[mid] == hb_tag) {
          *table_index = mid;
          return true;
        } else if (table[mid] < hb_tag) {
          low = mid + 1;
        } else {
          high = mid - 1;
        }
      }
    } else {
      // For 64-bit tags, use a linear search
      for (unsigned int i = 0; i < table_size; i++) {
        if (table[i] == hb_tag) {
