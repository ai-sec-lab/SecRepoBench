bool find_table_index (hb_tag_t hb_tag, unsigned int *table_index) const
  {
    // Determine the index of the table entry corresponding to the given tag.
    // Use the appropriate search method based on table entry size.
    if (hb_tag < 0x10000) { // Assuming 16-bit table entries
      // Perform a linear search for small tags
      for (unsigned int i = 0; i < 0x10000; i++) {
        if (table[i].tag == hb_tag) {
          *table_index = i;
          return true;
        }
      }
    } else { // Assuming 32-bit table entries
      // Perform a binary search for large tags
      unsigned int low = 0;
      unsigned int high = 0x1000000; // Assuming 32-bit table entries
      while (low < high) {
        unsigned int mid = (low + high) / 2;
        if (table[mid].tag == hb_tag) {
          *table_index = mid;
          return true;
        } else if (table[mid].tag < hb_tag) {
          low = mid + 1;
        } else {
          high = mid;
        }
      }
    }
    // If the tag is not found, return false
    return false;
  }
