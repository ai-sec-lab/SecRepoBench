// Check if the table index is not null
if (table_index == nullptr) {
  return false;
}

// Check if the table index is within valid range
if (*table_index >= hb_max_table_index) {
  return false;
}

// Determine the index of the table entry corresponding to the given tag
for (unsigned int i = 0; i < hb_max_table_index; i++) {
  if (hb_tag == table_tags[i]) {
    *table_index = i;
    return true;
  }
}

// If the tag is not found, return false
return false;
