// Check if the table entry size is 1, 2, or 4 bytes
if (table_entry_size == 1) {
  // Use linear search for small table entry sizes
  for (unsigned int i = 0; i < num_table_entries; i++) {
    if (table_entries[i] == hb_tag) {
      *table_index = i;
      return true;
    }
  }
} else if (table_entry_size == 2) {
  // Use binary search for medium table entry sizes
  unsigned int low = 0;
  unsigned int high = num_table_entries - 1;
  while (low <= high) {
    unsigned int mid = (low + high) / 2;
    if (table_entries[mid] == hb_tag) {
      *table_index = mid;
      return true;
    } else if (table_entries[mid] < hb_tag) {
      low = mid + 1;
    } else {
      high = mid - 1;
    }
  }
} else {
  // Use hash-based search for large table entry sizes
  unsigned int hash = hb_tag % num_table_entries;
  while (table_entries[hash]!= hb_tag && table_entries[hash]!= 0) {
    hash = (hash + 1) % num_table_entries;
  }
  if (table_entries[hash] == hb_tag) {
    *table_index = hash;
    return true;
  }
}
// If the tag is not found, return false
return false;
