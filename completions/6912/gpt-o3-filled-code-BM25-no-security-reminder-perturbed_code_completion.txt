          ssize_t
            number_attributes;

          /*
            Estimate how many coordinate pairs the path might expand to.
            Each alphabetic command designator introduces a new attribute
            which can translate into as many as 6*BezierQuantum points
            when the path is decomposed (worst-case for cubic Bezier curves).
          */
          number_attributes=1;  /* at least one attribute present */
          for (s=t; *s != '\0'; s++)
            if (isalpha((int) *s))
              number_attributes++;
          points_extent=(double) (6*BezierQuantum*number_attributes);
