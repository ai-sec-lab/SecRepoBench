const char *e = p;
  *valp = 0;
  int sign = 1;

  // Skip leading whitespace
  while (e < end && isspace((unsigned char)*e)) {
    e++;
  }

  // Check for a sign
  if (e < end && (*e == '-' || *e == '+')) {
    if (*e == '-') {
      sign = -1;
    }
    e++;
  }

  // Parse digits
  if (e < end && isdigit((unsigned char)*e)) {
    while (e < end && isdigit((unsigned char)*e)) {
      *valp = *valp * 10 + (*e - '0');
      e++;
    }
    *valp *= sign;
  } else {
    return NULL; // No digits found
  }

  return e;
