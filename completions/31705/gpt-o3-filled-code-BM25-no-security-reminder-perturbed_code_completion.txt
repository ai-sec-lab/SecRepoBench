end:
  /* Close the file if it was opened */
  if (fp != NULL) {
    fclose(fp);
  }

  /* If an error occurred propagate it and release resources */
  if (rc < 0) {
    if (*needs_free && *chunk != NULL) {
      free(*chunk);
      *chunk = NULL;
      *needs_free = false;
    }
    return rc;               /* forward the error code */
  }

  /* Success: return the size of the lazy chunk that has been built */
  return lazy_chunk_bytes;
