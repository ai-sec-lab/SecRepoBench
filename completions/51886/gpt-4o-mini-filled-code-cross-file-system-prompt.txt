if (count > 0) {
        const int oldCount = fCount;
        const int newCount = this->calculateSizeDeltaOrDie(count);
        this->resize(newCount, sizeOfT);
        
        if (src != nullptr) {
            memcpy(fStorage + offset(oldCount, sizeOfT), src, mem_size(count, sizeOfT));
        }
    }
    fCount += count;
