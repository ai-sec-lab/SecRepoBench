if (m->avctx->codec_id == AV_CODEC_ID_TRUEHD && max_channel > 5) {
    avpriv_request_sample(m->avctx, "TrueHD streams with more than 6 channels");
    return AVERROR_PATCHWELCOME;
}

if (m->avctx->codec_id == AV_CODEC_ID_MLP && noise_type != 0) {
    avpriv_request_sample(m->avctx, "MLP streams with noise type not equal to 0");
    return AVERROR_PATCHWELCOME;
}
