if (max_matrix_channel > std_max_matrix_channel) {
    av_log(m->avctx, AV_LOG_ERROR, "max_matrix_channel %d too large\n", max_matrix_channel);
    return AVERROR_INVALIDDATA;
}

if (m->avctx->codec_id == AV_CODEC_ID_MLP && !noise_type && max_matrix_channel > 5) {
    avpriv_request_sample(m->avctx, "MLP stream with noise_type=0 and max_matrix_channel > 5");
    return AVERROR_PATCHWELCOME;
}
