if (m->avctx->codec_id == AV_CODEC_ID_TRUEHD && max_matrix_channel > std_max_matrix_channel) {
    if (!noise_type) {
        avpriv_request_sample(m->avctx, "TrueHD stream with more than 6 channels and MLP's noise type");
        return AVERROR_PATCHWELCOME;
    }
} else if (max_matrix_channel > std_max_matrix_channel) {
    av_log(m->avctx, AV_LOG_ERROR, "Invalid max_matrix_channel %d\n", max_matrix_channel);
    return AVERROR_INVALIDDATA;
}
