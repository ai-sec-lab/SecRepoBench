if (m->avctx->codec_id == AV_CODEC_ID_TRUEHD && max_channel + 1 > 6 && noise_type) {
    av_log(m->avctx, AV_LOG_ERROR, "TrueHD streams with more than 6 channels and MLP's noise type are not supported.\n");
    avpriv_request_sample(m->avctx, "TrueHD streams with more than 6 channels and MLP's noise type are not supported");
    return AVERROR_PATCHWELCOME;
}
