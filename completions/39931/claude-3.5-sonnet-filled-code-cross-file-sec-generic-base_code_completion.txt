if (argc < 0) {
  argnum_error(mrb, 0);
  goto L_RAISE;
}
m = mrb_method_search_vm(mrb, &c, mid);
if (MRB_METHOD_UNDEF_P(m)) {
  m = prepare_missing(mrb, recv, mid, &c, a, &c, blk, 0);
  mid = MRB_SYM(method_missing);
}
if (n == 15) {
  mrb_value args = mrb_ary_new_from_values(mrb, argc, argv);
  regs[a+1] = args;
}
else if (argc > n) {
  if (n == 0) {
    mrb_value args = mrb_ary_new_from_values(mrb, argc, argv);
    regs[a+1] = args;
  }
  else {
    value_move(&regs[a+1], argv, n);
    mrb_value args = mrb_ary_new_from_values(mrb, argc-n, argv+n);
    regs[a+n+1] = args;
  }
}
else {
  value_move(&regs[a+1], argv, argc);
  if (argc < n) {
    stack_clear(&regs[a+argc+1], n-argc);
  }
}
mrb_stack_extend(mrb, a+n+2);
