if (argc < 0) {
      mrb_value exc = mrb_exc_new_str(mrb, E_ARGUMENT_ERROR, "negative argument count");
      mrb_exc_set(mrb, exc);
      goto L_RAISE;
    }

    m = mrb_method_search_vm(mrb, &c, mid);
    if (MRB_METHOD_UNDEF_P(m)) {
      mrb_value args = mrb_ary_new_from_values(mrb, argc, argv);
      mrb_ary_unshift(mrb, args, mrb_symbol_value(mid));
      mrb_stack_extend(mrb, argc + 1);
      ci->stack[n + 1] = args;
      argc = 15; // Adjust argc for the method call
    }

    ci = cipush(mrb, n, 0, c, NULL, mid, argc);
    if (MRB_METHOD_PROC_P(m)) {
      struct RProc *p = MRB_METHOD_PROC(m);
      mrb_vm_ci_proc_set(ci, p);
      if (!MRB_PROC_CFUNC_P(p)) {
        mrb_stack_extend(mrb, p->body.irep->nregs + argc);
      }
    }
