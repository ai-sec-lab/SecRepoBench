/* search method */
m = mrb_method_search_vm(mrb, &c, mid);

/* push callinfo */
ci = cipush(mrb, n, 0, c, NULL, mid, argc);

/* prepare stack */
if (voff >= 0) {
  argv = mrb->c->stbase + voff;
}
ci->stack[0] = self;
if (argc < 15) {
  if (argc > 0)
    stack_copy(ci->stack+1, argv, argc);
  ci->stack[argc+1] = blk;
}
else {
  ci->stack[1] = mrb_ary_new_from_values(mrb, argc, argv);
  ci->stack[2] = blk;
}
