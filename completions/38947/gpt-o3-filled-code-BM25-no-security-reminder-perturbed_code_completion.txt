// Move to the element that should contain the file format version.
    ++elementIter;

    // Validate that we really have another element – otherwise the file is malformed.
    if (elementIter == root.children.end() || (*elementIter).tokens[0].empty()) {
        ASSIMP_LOG_ERROR("LWS: Detected invalid LWS file – unable to read file format version");
        throw DeadlyImportError("LWS: Invalid or truncated file, no format version present");
    }
