++it;
if (it == root.children.end() || (*it).tokens[0] != "LWSC") {
    throw DeadlyImportError("LWS: Invalid file format version");
}

++it;
if (it == root.children.end()) {
    throw DeadlyImportError("LWS: Unexpected end of file");
}

version = strtoul10((*it).tokens[0].c_str());
ASSIMP_LOG_INFO("LWS file format version is ", (*it).tokens[0]);

if (version < 1 || version > 5) {
    ASSIMP_LOG_WARN("LWS: Unsupported file format version, continuing with caution");
}
