// Increment the iterator to the next element
    ++it;

    // Check if the iterator has reached the end of the children list
    if (it == root.children.end()) {
        throw DeadlyImportError("LWS: Unexpected end of file while parsing, version information missing.");
    }

    // Retrieve the version number from the current element
    unsigned int version = strtoul10((*it).tokens[0].c_str());

    // Validate the version number
    if (version < 1 || version > 4) {
        throw DeadlyImportError("LWS: Unsupported version number ", std::to_string(version), ".");
    }
