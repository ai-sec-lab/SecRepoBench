// Search for the MPEG start code pattern (0x00 0x00 0x01) followed by "DivX"
            const char *pattern = "\x00\x00\x01DivX";
            char *found = memmem(dcfg + i, vosh_size - i, pattern, strlen(pattern));
            if (found) {
                ctx->forced_packed = GF_TRUE;
                // Replace 'p' with 'n' in the found pattern
                found[4] = 'n'; // Change 'p' to 'n'
            }
            i += 4; // Move to the next byte after the current position
            if (i >= vosh_size) break; // Prevent overflow
