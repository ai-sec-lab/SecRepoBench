i=0;
while (i+4<vosh_size) {
    if (dcfg[i] == 0x00 && dcfg[i+1] == 0x00 && dcfg[i+2] == 0x01) {
        if (dcfg[i+3] == 0xb6) {
            frame = memchr(dcfg + i + 4, 'p', vosh_size - i - 4);
            if (frame) {
                ctx->forced_packed = GF_TRUE;
                frame[0] = 'n';
            }
            break;
        }
    }
    i++;
}
