    if ((i + 3 < vosh_size) && (dcfg[i] == 0x00) && (dcfg[i + 1] == 0x00) && (dcfg[i + 2] == 0x01)) {
        if ((vosh_size - i >= 6) && (memcmp(dcfg + i + 3, "DivX", 4) == 0)) {
            frame = memchr(dcfg + i + 7, 'p', vosh_size - i - 7);
            if (frame) {
                ctx->forced_packed = GF_TRUE;
                frame[0] = 'n';
            }
            break;
        }
    }
    i++;
}
