if (i+4 >= vosh_size) break;

			if ((dcfg[i] == 0) && (dcfg[i+1] == 0) && (dcfg[i+2] == 1) && (dcfg[i+3] == 'D') ) {
				char *frame = memchr(dcfg + i + 4, 'p', vosh_size - i - 4);
				if (frame && !strncmp(frame, "packed", 6)) {
					ctx->is_packed = GF_TRUE;
					frame[0] = 'n';
				}
				if (frame && !strncmp(frame, "vfr", 3)) {
					ctx->is_vfr = GF_TRUE;
				}
				break;
			}
			i++;
