          color->colorspace = sRGBColorspace; // Set colorspace to sRGB

      } else if (sscanf(name, "rgba(%lf,%lf,%lf,%lf)", &color->red, &color->green, &color->blue, &color->alpha) == 4) {
          color->colorspace = sRGBColorspace; // Set colorspace to sRGB
          return MagickTrue; // Successfully parsed RGBA values
      } else {

              OptionWarning, "UnrecognizedColor", "`%s'", name);
