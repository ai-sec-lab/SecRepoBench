char colorspace[MagickPathExtent];
      MagickBooleanType icc_color;
      
      // Initialize the colorspace array and the icc_color flag
      (void) ResetMagickMemory(colorspace, 0, sizeof(colorspace));
      icc_color = MagickFalse;

      // Extract the colorspace name from the input string
      for (i = 0; name[i] != '\0' && name[i] != '('; i++)
        colorspace[i] = name[i];
      colorspace[i] = '\0'; // Null-terminate the colorspace string

      // Check for ICC color space prefix
      if (LocaleNCompare(colorspace, "icc-", 4) == 0)
      {
          icc_color = MagickTrue;
          (void) CopyMagickString(colorspace, colorspace + 4, MagickPathExtent);
      }

      // Validate the colorspace name
      type = ParseCommandOption(MagickColorspaceOptions, MagickFalse, colorspace);
      if (type < 0)
      {
          (void) ThrowMagickException(exception, GetMagickModule(),
              OptionWarning, "UnrecognizedColor", "`%s'", name);
          return(MagickFalse);
      }

      // Set the colorspace in the color structure
      color->colorspace = (ColorspaceType) type;
      if (icc_color == MagickFalse && color->colorspace == RGBColorspace)
      {
          color->colorspace = sRGBColorspace; // Ensure compliance with SVG standard
          color->depth = 8; // Set default depth
      }
