/* Store the original (user-provided) attribute name */
	attr->name = attribute;

	/* Store a lower-cased copy that will be used as the hash key */
	attr->lcname = zend_string_tolower_ex(zend_string_copy(attribute), persistent);

	/* Basic meta-information */
	attr->offset = offset;
	attr->argc   = argc;
	attr->ce     = NULL; /* Will be filled in later where necessary */

	/* Insert the attribute into the hash table */
	zend_hash_update_ptr(*attributes, attr->lcname, attr);

	return attr;
