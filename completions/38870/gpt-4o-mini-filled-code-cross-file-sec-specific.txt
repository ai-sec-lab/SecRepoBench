unz_file_info fileInfo;
        char filename[FileNameSize];
        if (unzGetCurrentFileInfo(m_ZipFileHandle, &fileInfo, filename, sizeof(filename), nullptr, 0, nullptr, 0) != UNZ_OK)
            continue;

        if (fileInfo.uncompressed_size == 0)
            continue;

        // Simplify the filename and store it in the archive map
        std::string simplifiedFilename(filename);
        ReplaceAll(simplifiedFilename, "\\", "/"); // Normalize path separators
        m_ArchiveMap[simplifiedFilename] = ZipFileInfo(m_ZipFileHandle, fileInfo.uncompressed_size);
