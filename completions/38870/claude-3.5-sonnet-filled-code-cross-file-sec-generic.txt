unz_file_info file_info;
if (unzGetCurrentFileInfo(m_ZipFileHandle, &file_info, filename, FileNameSize, nullptr, 0, nullptr, 0) != UNZ_OK) {
    continue;
}

if (file_info.uncompressed_size == 0) {
    continue;
}

std::string simplified_filename(filename);
SimplifyFilename(simplified_filename);

if (simplified_filename.empty()) {
    continue;
}

ZipFileInfo zip_file_info(m_ZipFileHandle, file_info.uncompressed_size);
m_ArchiveMap[simplified_filename] = zip_file_info;
