unz_file_info64 file_info;
        if (unzGetCurrentFileInfo64(m_ZipFileHandle, &file_info, filename, FileNameSize, nullptr, 0, nullptr, 0) != UNZ_OK)
            continue;

        if (file_info.uncompressed_size == 0)
            continue;

        std::string sFilename(filename);
        SimplifyFilename(sFilename);

        ZipFileInfo info = { m_ZipFileHandle, file_info.uncompressed_size };
        m_ArchiveMap.insert(std::make_pair(sFilename, info));
