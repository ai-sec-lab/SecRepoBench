if (unzGetCurrentFileInfo(m_ZipFileHandle, &fileInfo, filename, sizeof(filename), nullptr, 0, nullptr, 0) != UNZ_OK)
            continue;

        // Validate the uncompressed size to ensure it's not zero
        if (fileInfo.uncompressed_size == 0)
            continue;

        // Simplify the filename to remove unnecessary components
        std::string simplifiedFilename(filename);
        SimplifyFilename(simplifiedFilename);

        // Ensure the filename is valid and does not contain any malicious patterns
        if (simplifiedFilename.empty() || simplifiedFilename.find("..") != std::string::npos) {
            continue; // Skip invalid filenames
        }

        // Create a ZipFileInfo instance and store it in the archive map
        ZipFileInfo zipFileInfo;
        zipFileInfo.fileHandle = m_ZipFileHandle; // Store the file handle
        zipFileInfo.uncompressedSize = fileInfo.uncompressed_size; // Store the uncompressed size

        // Insert the simplified filename and its corresponding ZipFileInfo into the archive map
        m_ArchiveMap[simplifiedFilename] = zipFileInfo;
