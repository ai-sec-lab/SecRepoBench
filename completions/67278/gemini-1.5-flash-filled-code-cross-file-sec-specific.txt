u32 read, size;
		if (gf_bs_read_data(bs, (u8*)&acfg, sizeof(acfg), &read) != GF_OK) break;
		if (read!=sizeof(acfg)) break;
		if (acfg.Config.SampleRateIndex>15) {
			nb_skip += sizeof(acfg);
			continue;
		}
		size = gf_bs_get_size(bs) - gf_bs_get_position(bs);
		if (gf_bs_seek(bs, size, GF_BITSTREAM_SEEK_CUR) != GF_OK) break;
		nb_frames++;
