u32 start_pos = gf_bs_get_position(bs);
        if (gf_bs_available(bs) < 3) break;
        
        // Read the first 3 bytes to check for LATM sync word
        u32 sync_word = gf_bs_read_int(bs, 24);
        if (sync_word != 0x2B7) {
            nb_skip++;
            gf_bs_skip_bytes(bs, 1);
            continue;
        }
        
        // Read the AudioMuxElement
        if (!gf_bs_read_bit(bs)) break; // audioMuxVersion
        if (gf_bs_read_bit(bs)) break; // allStreamsSameTimeFraming
        u32 numSubFrames = gf_bs_read_int(bs, 6);
        u32 numProgram = gf_bs_read_int(bs, 4);
        u32 numLayer = gf_bs_read_int(bs, 3);
        
        // For simplicity, we only handle single program and single layer
        if (numProgram != 0 || numLayer != 0) break;
        
        // Read the AudioSpecificConfig
        u32 audioObjectType = gf_bs_read_int(bs, 5);
        u32 sampleRateIndex = gf_bs_read_int(bs, 4);
        
        // Check if the sample rate index is valid
        if (sampleRateIndex >= 13) {
            nb_skip++;
            gf_bs_skip_bytes(bs, 1);
            continue;
        }
        
        nb_frames++;
        
        // Skip the rest of the frame
        u32 frame_length = gf_bs_get_position(bs) - start_pos;
        gf_bs_skip_bytes(bs, frame_length);
        
        if (nb_frames >= 2) break;
