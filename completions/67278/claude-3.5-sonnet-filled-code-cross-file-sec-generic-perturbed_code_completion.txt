u32 hdr, size;
u8 tmp[4];
if (gf_bs_available(bs) < 3) break;
hdr = gf_bs_read_u24(bs);
if (hdr & 0x800000) {	/*audio mux element*/
    if (gf_latm_get_value(bs)) {
        nb_skip++;
        continue;
    }
    gf_bs_read_data(bs, tmp, 4);
    size = ((tmp[0] & 0x1F)<<8) | tmp[1];
    size <<= 8;
    size |= tmp[2];
    size <<= 8;
    size |= tmp[3];
    if (!size || (size > gf_bs_available(bs))) {
        nb_skip++;
        continue;
    }
    memset(&acfg, 0, sizeof(GF_M4ADecSpecInfo));
    gf_m4a_parse_config(bs, &acfg, 0);
    if (acfg.base_sr_index != 15) {
        numframes++;
    } else {
        nb_skip++;
    }
} else {
    nb_skip++;
}
if (numframes==2) break;
