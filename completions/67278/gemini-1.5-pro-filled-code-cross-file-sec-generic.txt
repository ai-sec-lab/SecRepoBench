u32 frame_size;
		GF_Err e = gf_m4a_parse_config(bs, &acfg, 0, &frame_size);
		if (e) {
			if (e==GF_EOS) break;
			nb_skip += gf_bs_get_bit_offset(bs) / 8;
			gf_bs_seek(bs, 0);
			break;
		}
		if (!acfg.base_object_type || !acfg.base_sr_idx) {
			gf_bs_seek(bs, 0);
			break;
		}
		gf_bs_skip_bytes(bs, frame_size);
		nb_frames++;
