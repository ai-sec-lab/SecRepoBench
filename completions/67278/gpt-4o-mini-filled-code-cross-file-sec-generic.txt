// Read the next frame from the bitstream
        if (!gf_bs_read_frame(bs, &acfg)) {
            nb_frames++;
            // Check if the sample rate index is valid
            if (acfg.sample_rate_index < MAX_SAMPLE_RATE_INDEX) {
                // Valid frame found, increment the frame count
                nb_skip = 0; // Reset skip count
            } else {
                // Invalid sample rate, increment skip count
                nb_skip++;
            }
        } else {
            // End of stream or error in reading frame
            break;
        }

        // Optional: Add a limit to avoid infinite loops in case of corrupted data
        if (nb_frames > MAX_FRAMES_TO_PROBE) {
            break;
