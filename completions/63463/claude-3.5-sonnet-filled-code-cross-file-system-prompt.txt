BITCODE_RL obj_size = bit_read_RL (dat);
obj->size = obj_size;
LOG_TRACE ("  size: " FORMAT_RL "\n", obj_size);
if (obj_size > size - 2)
  {
    LOG_ERROR ("Invalid object size " FORMAT_RL " > %u", obj_size,
               (unsigned)(size - 2));
    obj->size = size - 2;
    error |= DWG_ERR_VALUEOUTOFBOUNDS;
  }
if (obj_size < 4)
  {
    LOG_ERROR ("Invalid object size " FORMAT_RL " < 4", obj_size);
    obj->size = 4;
    error |= DWG_ERR_VALUEOUTOFBOUNDS;
  }
if (dat->byte + obj_size > oldpos + size)
  {
    LOG_ERROR ("Invalid object size " FORMAT_RL " > %u", obj_size,
               (unsigned)(oldpos + size - dat->byte));
    obj->size = oldpos + size - dat->byte;
    error |= DWG_ERR_VALUEOUTOFBOUNDS;
  }
if (obj->size < obj_size)
  {
    BITCODE_RL offset = obj_size - obj->size;
    LOG_TRACE (" unknown: %u\n", (unsigned)offset);
    bit_read_fixed (dat, offset);
  }
SINCE (R_11)
{
  BITCODE_RS crc;
  dat->byte += obj_size - 2;
  crc = bit_read_RS (dat);
  LOG_TRACE ("  crc: %04X [RSx]\n", crc);
  dat->byte -= obj_size;
}
