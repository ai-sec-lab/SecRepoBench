obj->size = (dat->byte - oldpos) & 0xFFFFFFFF;
                if (obj->size > dat->size - oldpos)
                  {
                    obj->size = dat->size - oldpos;
                    LOG_ERROR ("Invalid object size %u, adjusted to %u\n",
                               (unsigned)(dat->byte - oldpos), obj->size);
                  }
                if (dat->version >= R_2_0b)
                  {
                    BITCODE_T crc;
                    BITCODE_RL unknown_data_size;
                    size_t unknown_data_pos = dat->byte;
                    crc = bit_read_CRC16 (dat);
                    unknown_data_size = (BITCODE_RL)(dat->byte - unknown_data_pos);
                    if (crc != obj->crc16)
                      {
                        LOG_ERROR ("CRC16 mismatch: expected 0x%x, got 0x%x\n",
                                   obj->crc16, crc);
                        error |= DWG_ERR_CRC16MISMATCH;
                      }
                    if (unknown_data_size)
                      {
                        obj->unknown_data = bit_read_bits (
                            dat, unknown_data_size * 8);
                        LOG_TRACE ("unknown_data [%u bytes]: ",
                                   unknown_data_size);
                        LOG_TRACE_TF (obj->unknown_data, unknown_data_size);
                        LOG_TRACE ("\n");
                      }
                  }
                else if (obj->size != (dat->byte - oldpos))
                  {
                    BITCODE_RL unknown_data_size;
                    size_t unknown_data_pos = dat->byte;
                    unknown_data_size = (BITCODE_RL)(dat->byte - oldpos) - obj->size;
                    if (unknown_data_size)
                      {
                        obj->unknown_data = bit_read_bits (
                            dat, unknown_data_size * 8);
                        LOG_TRACE ("unknown_data [%u bytes]: ",
                                   unknown_data_size);
                        LOG_TRACE_TF (obj->unknown_data, unknown_data_size);
                        LOG_TRACE ("\n");
                      }
                    LOG_WARN ("Invalid object size %u, expected %u\n",
                              (unsigned)obj->size,
                              (unsigned)(dat->byte - oldpos));
                  }
              }
