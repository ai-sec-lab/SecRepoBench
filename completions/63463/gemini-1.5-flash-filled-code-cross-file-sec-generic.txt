if (dat->version < R_11)
                  {
                    if (obj->size != (dat->byte - oldpos))
                      {
                        BITCODE_RL unknown_size = (dat->byte - oldpos) - obj->size;
                        LOG_WARN ("Object size mismatch: expected %" PRIuSIZE
                                  ", got %" PRIuSIZE ", unknown_size %" PRIu64
                                  "\n",
                                  obj->size, (dat->byte - oldpos),
                                  unknown_size);
                        obj->unknown_data = bit_read_bytes (dat, unknown_size);
                        obj->unknown_data_size = unknown_size;
                      }
                  }
                else
                  {
                    BITCODE_U16 crc16;
                    BITCODE_RL extra_data_size = 0;
                    size_t start_byte = dat->byte;
                    crc16 = bit_read_U16 (dat);
                    if (crc16 != obj->crc16)
                      {
                        LOG_WARN ("CRC16 mismatch: expected 0x%x, got 0x%x\n",
                                  obj->crc16, crc16);
                        error |= DWG_ERR_CRC16;
                      }
                    if (obj->size != (dat->byte - oldpos))
                      {
                        extra_data_size = (dat->byte - oldpos) - obj->size;
                        LOG_WARN ("Object size mismatch: expected %" PRIuSIZE
                                  ", got %" PRIuSIZE ", extra_data_size %" PRIu64
                                  "\n",
                                  obj->size, (dat->byte - oldpos),
                                  extra_data_size);
                        obj->unknown_data = bit_read_bytes (dat, extra_data_size);
                        obj->unknown_data_size = extra_data_size;
                      }
                    if (extra_data_size)
                      {
                        BITCODE_U16 crc16_extra = bit_crc16 (
                            obj->unknown_data, extra_data_size);
                        if (crc16_extra != obj->crc16_extra)
                          {
                            LOG_WARN (
                                "CRC16 mismatch for extra data: expected 0x%x,"
                                " got 0x%x\n",
                                obj->crc16_extra, crc16_extra);
                            error |= DWG_ERR_CRC16;
                          }
                      }
                    if (obj->size < 0)
                      {
                        obj->size = (dat->byte - oldpos);
                        LOG_WARN ("Invalid object size: %" PRIuSIZE
                                  ", adjusted to %" PRIuSIZE "\n",
                                  obj->size, (dat->byte - oldpos));
                        error |= DWG_ERR_INVALIDDWG;
                      }
                  }
