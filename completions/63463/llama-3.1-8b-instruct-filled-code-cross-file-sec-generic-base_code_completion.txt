SINCE (R_2_0b)
{
  // This block of code handles the post-processing of objects in a DWG file format,
  // ensuring data consistency and handling extra data and CRC checks.
  // For objects in pre-R11 versions, it checks if the object size matches the expected size.
  // If there is a discrepancy, it calculates the offset of unknown data and stores it.
  // For objects in later versions, it also handles the CRC16 check, adjusting the byte position
  // as needed and logging any errors or discrepancies. If any additional data exists beyond the
  // expected size, it is read and stored as unknown data, and CRC checks are performed to validate
  // the integrity of the data block. If the object size is invalid, it is adjusted and an error is logged.
  if (obj->type!= DWG_TYPE_JUMP_r11)
    {
      // Check if the object size matches the expected size
      if (obj->size!= (dat->byte - oldpos) & 0xFFFFFFFF)
        {
          // Calculate the offset of unknown data
          BITCODE_RL unknown_offset = (dat->byte - oldpos) & 0xFFFFFFFF;
          // Store the unknown data
          obj->unknown_offset = unknown_offset;
          // Log the discrepancy
          LOG_ERROR ("Object size mismatch: expected %d, got %d\n",
                     obj->size, unknown_offset);
        }
      // Check if the object has any unknown bits
      if (dwg_decode_unknown_bits (dat, obj)!= 0)
        {
          // Log the unknown bits
          LOG_ERROR ("Unknown bits found in object\n");
        }
      // Check if the object has any extra data
      if (dat->byte < oldpos + size)
        {
          // Read and store the extra data
          BITCODE_RL extra_data_size = (dat->byte - oldpos) & 0xFFFFFFFF;
          obj->extra_data = bit_read_bits (dat, extra_data_size);
          // Log the extra data
          LOG_TRACE ("Extra data found in object: %d bytes\n",
                     extra_data_size);
        }
      // Check if the object has a valid CRC16
      if (dwg_check_crc16 (dat, obj)!= 0)
        {
          // Log the CRC16 error
          LOG_ERROR ("CRC16 error in object\n");
        }
    }
}
