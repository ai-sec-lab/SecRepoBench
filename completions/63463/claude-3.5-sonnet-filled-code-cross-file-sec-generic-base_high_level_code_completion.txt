BITCODE_RL obj_size = bit_read_RL(dat);
if (obj_size != obj->size)
  {
    if (obj_size > obj->size)
      {
        LOG_ERROR("Object size mismatch: read %u > calculated %u", obj_size, obj->size);
        obj->size = obj_size;
      }
    else
      {
        LOG_WARN("Object size mismatch: read %u < calculated %u", obj_size, obj->size);
      }
  }

BITCODE_RS crc;
if (dat->version >= R_11)
  {
    crc = bit_read_RS(dat);
    LOG_TRACE("crc: 0x%04X [RS]\n", crc);
  }

if (dat->byte - oldpos != obj->size)
  {
    BITCODE_BL unknown_size = obj->size - (dat->byte - oldpos);
    LOG_WARN("Unknown data of %u bytes", (unsigned)unknown_size);
    bit_advance_position(dat, unknown_size * 8);
    if (unknown_size > 0)
      {
        obj->unknown_bits = calloc(1, unknown_size);
        if (obj->unknown_bits)
          {
            memcpy(obj->unknown_bits, &dat->chain[dat->byte - unknown_size], unknown_size);
            obj->num_unknown_bits = unknown_size * 8;
          }
      }
  }
dat->byte = oldpos + obj->size;
