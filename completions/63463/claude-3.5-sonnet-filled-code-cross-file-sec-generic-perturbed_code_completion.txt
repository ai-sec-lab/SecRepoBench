BITCODE_RL obj_size = bit_read_RL (bitchain);
obj->size = obj_size;
LOG_TRACE ("size: " FORMAT_RL " [RL]\n", obj_size);
if (obj_size > size)
  {
    LOG_ERROR ("Invalid object size " FORMAT_RL " > %u", obj_size, size);
    obj->size = size;
    error |= DWG_ERR_VALUEOUTOFBOUNDS;
  }
else if (obj_size + 2 < size)
  {
    BITCODE_RS crc;
    BITCODE_RL offset = obj_size - 2;
    LOG_TRACE ("UNKNOWN_BITS pos: " FORMAT_RL "\n", offset);
    bit_set_position (bitchain, (bitchain->byte - oldpos) * 8);
    if (offset > 0 && offset < size)
      {
        obj->unknown_bits = bit_read_TF (bitchain, offset);
        LOG_TRACE_TF (obj->unknown_bits, offset);
      }
    crc = bit_read_RS (bitchain);
    LOG_TRACE ("crc: %04X [RS]\n", crc);
    if (crc != bit_calc_CRC (0xC0C1, &bitchain->chain[oldpos], obj_size - 2))
      {
        LOG_ERROR ("Wrong CRC %04X", crc);
        error |= DWG_ERR_WRONGCRC;
      }
  }
else
  {
    bit_advance_position (bitchain, (obj_size - 2) * 8);
    bit_read_RS (bitchain); // crc
  }
