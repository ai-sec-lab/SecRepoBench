BITCODE_RL obj_size = bit_read_RL (dat);
obj->size = obj_size;
LOG_TRACE ("size: " FORMAT_RL " [RL]\n", obj_size);
if (obj_size > size - 2)
  {
    LOG_ERROR ("Invalid object size " FORMAT_RL " > %d", obj_size, size - 2);
    obj->size = size - 2;
    error |= DWG_ERR_VALUEOUTOFBOUNDS;
  }
if (dat->version >= R_11)
  {
    BITCODE_RS crc;
    dat->byte += obj_size;
    crc = bit_read_RS (dat);
    LOG_TRACE ("crc: %04X [RS]\n", crc);
    if (dat->byte - 2 == oldpos + obj_size)
      {
        BITCODE_RS calc_crc = bit_calc_CRC (0xC0C1, &dat->chain[oldpos],
                                            obj_size);
        if (crc != calc_crc)
          {
            LOG_ERROR ("Wrong CRC %04X, expected %04X", crc, calc_crc);
            error |= DWG_ERR_WRONGCRC;
          }
      }
    else
      {
        LOG_ERROR ("Wrong object size " FORMAT_RL ", expected %ld",
                   obj_size, dat->byte - 2 - oldpos);
        error |= DWG_ERR_VALUEOUTOFBOUNDS;
      }
    dat->byte -= 2;
    if (dat->byte - oldpos < obj_size)
      {
        BITCODE_RL offset = obj_size - (dat->byte - oldpos);
        LOG_TRACE ("offset: " FORMAT_RL " [RL]\n", offset);
        bit_read_fixed (dat, offset);
      }
  }
if (dat->byte - oldpos > obj_size)
  {
    BITCODE_RL offset = (dat->byte - oldpos) - obj_size;
    LOG_HANDLE ("unknown: %d\n", offset);
    obj->unknown_bits = (unsigned char *)calloc (offset, 1);
    if (!obj->unknown_bits)
      {
        LOG_ERROR ("Out of memory");
        return DWG_ERR_OUTOFMEM;
      }
    memcpy (obj->unknown_bits, &dat->chain[dat->byte - offset], offset);
    obj->num_unknown_bits = offset * 8;
    LOG_TRACE_TF (obj->unknown_bits, offset);
    LOG_TRACE ("\n");
  }
