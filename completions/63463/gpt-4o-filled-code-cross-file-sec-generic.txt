{
  BITCODE_RL obj_size;
  BITCODE_RL unknown_size;
  BITCODE_RL crc16;
  BITCODE_RL expected_crc16;

  // Calculate the object size
  obj_size = bit_read_RL(dat);
  LOG_TRACE("object size: 0x%x\n", obj_size);

  // Check if the object size is valid
  if (obj_size != (dat->byte - oldpos))
  {
    LOG_WARN("Object size mismatch: expected 0x%x, got 0x%x\n", obj_size, (dat->byte - oldpos));
    unknown_size = obj_size - (dat->byte - oldpos);
    if (unknown_size > 0)
    {
      // Read unknown data
      obj->unknown_bits = bit_read_bits(dat, unknown_size * 8);
      obj->num_unknown_bits = unknown_size * 8;
      LOG_TRACE("Read unknown data of size: 0x%x\n", unknown_size);
    }
  }

  // Read and verify CRC16
  crc16 = bit_read_RL(dat);
  expected_crc16 = calculate_crc16(dat, oldpos, obj_size);
  if (crc16 != expected_crc16)
  {
    LOG_ERROR("CRC16 mismatch: expected 0x%x, got 0x%x\n", expected_crc16, crc16);
    error |= DWG_ERR_CRC;
  }
}
