BITCODE_RL obj_size = bit_read_RL(dat);
if (obj_size != obj->size)
  {
    BITCODE_RL offset = obj_size - obj->size;
    if (offset > 0 && offset < 0x1000)
      {
        obj->unknown_bits = bit_read_TF(dat, offset);
        obj->num_unknown_bits = offset * 8;
        LOG_TRACE("unknown_bits: %d [TF]\n", offset);
      }
    else
      {
        LOG_ERROR("Invalid object size %d != %d", obj_size, obj->size);
        obj->size = obj_size;
        error |= DWG_ERR_VALUEOUTOFBOUNDS;
      }
  }
SINCE(R_11)
{
  BITCODE_RS crc;
  long pos = dat->byte;
  long end_address = obj->address + obj->size;
  if (dat->byte + 2 <= end_address)
    {
      crc = bit_read_RS(dat);
      LOG_TRACE("crc: %04X [RSx]\n", crc);
      if (dat->byte != end_address)
        {
          LOG_ERROR("Wrong object size: %ld + %d != %ld",
                    obj->address, obj->size, dat->byte);
          dat->byte = end_address;
          error |= DWG_ERR_VALUEOUTOFBOUNDS;
        }
    }
  else
    {
      LOG_ERROR("Wrong object size: %ld + %d > %ld",
                obj->address, obj->size, dat->size);
      error |= DWG_ERR_VALUEOUTOFBOUNDS;
    }
  dat->byte = pos;
}
