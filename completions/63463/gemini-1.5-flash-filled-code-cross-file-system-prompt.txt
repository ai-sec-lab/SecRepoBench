BITCODE_RL unknown_data_offset;
                if (obj->size < 0)
                  {
                    obj->size = 0;
                    LOG_ERROR ("Invalid object size %" PRId64 "\n", obj->size);
                  }
                if (dat->version < R_11)
                  {
                    if (obj->size != (dat->byte - oldpos))
                      {
                        unknown_data_offset = (dat->byte - oldpos) - obj->size;
                        UNKNOWN_UNTIL (unknown_data_offset);
                        obj->size = (dat->byte - oldpos);
                      }
                  }
                else
                  {
                    BITCODE_T crc16;
                    size_t crc_pos = dat->byte;
                    crc16 = bit_read_T (dat);
                    if (crc16 != bit_crc16 (obj->unknown_bits, obj->size))
                      {
                        LOG_ERROR ("CRC16 mismatch for object at 0x%zx\n",
                                   obj->address);
                        error |= DWG_ERR_CRC16MISMATCH;
                      }
                    if (obj->size != (dat->byte - oldpos))
                      {

                        UNKNOWN_UNTIL (unknown_data_offset);
                        obj->size = (dat->byte - oldpos);
                      }
