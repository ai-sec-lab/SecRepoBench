BITCODE_RL obj_size = bit_read_RL(dat);
if (obj_size != obj->size)
  {
    BITCODE_RL offset = obj_size - obj->size;
    if (offset > 0 && offset < obj_size)
      {
        obj->unknown_size = offset;
        obj->unknown_bits = bit_read_TF(dat, offset);
        LOG_TRACE("unknown_bits: %d [TF]\n", offset);
      }
    else
      {
        LOG_ERROR("Invalid object size %d != %d", obj_size, obj->size);
        error |= DWG_ERR_VALUEOUTOFBOUNDS;
      }
  }
SINCE(R_11)
{
  BITCODE_RS crc;
  dat->byte += 2;
  crc = bit_calc_CRC(0xC0C1, dat->chain + oldpos, obj_size);
  if (crc != bit_read_RS(dat))
    {
      LOG_ERROR("Wrong CRC %04X", crc);
      error |= DWG_ERR_WRONGCRC;
    }
}
if (dat->byte - oldpos != obj_size)
  {
    long pos = dat->byte;
    dat->byte = oldpos + obj_size;
    if (dat->byte > dat->size)
      {
        LOG_ERROR("Invalid object size %d > %d", obj_size, (int)dat->size);
        error |= DWG_ERR_VALUEOUTOFBOUNDS;
        dat->byte = pos;
      }
  }
