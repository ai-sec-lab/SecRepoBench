    if (!hb_serialize_allocate(c, sizeof(VarRegionList), &dest)) {
      return false;

    dest->axis_count = src->axis_count;
    dest->region_count = regionmapping.size();
    size_t size = dest->axis_count * dest->region_count * sizeof(hb_serialize_axis_t);
    if (!hb_serialize_allocate(c, size, &dest->axis_data)) {
      return false;
    }
    size_t offset = 0;
    for (const auto &mapping : regionmapping) {
      const VarRegion *src_region = src->regions[mapping.first];
      hb_serialize_axis_t *dest_axis = (hb_serialize_axis_t *)((uint8_t *)dest->axis_data + offset);
      memcpy(dest_axis, src_region->axis_data, src_region->axis_count * sizeof(hb_serialize_axis_t));
      offset += src_region->axis_count * sizeof(hb_serialize_axis_t);
    }
