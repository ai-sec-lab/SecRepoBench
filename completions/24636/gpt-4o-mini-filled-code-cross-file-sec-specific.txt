uint32_t tp_header_size;
if (bytestream2_get_bytes_left(&s->packed_headers_stream) < 4) {
    av_log(s->avctx, AV_LOG_ERROR, "Insufficient data for tile-part header size\n");
    return AVERROR_INVALIDDATA;
}
tp_header_size = bytestream2_get_be32u(&s->packed_headers_stream);
if (tp_header_size > MAX_TILE_PART_HEADER_SIZE) {
    av_log(s->avctx, AV_LOG_ERROR, "Tile-part header size exceeds maximum allowed size\n");
    return AVERROR_INVALIDDATA;
}
