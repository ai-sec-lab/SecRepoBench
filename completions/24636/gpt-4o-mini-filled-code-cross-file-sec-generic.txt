uint32_t tp_header_size = bytestream2_get_be32u(&s->g); // Read the next 32 bits for the tile-part header size
                if (tp_header_size < 4 || tp_header_size > bytestream2_get_bytes_left(&s->g)) {
                    av_log(s->avctx, AV_LOG_ERROR, "Invalid tile-part header size: %u\n", tp_header_size);
                    return AVERROR_INVALIDDATA;
                }
