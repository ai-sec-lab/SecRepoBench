unsigned int tp_header_size = bytestream2_get_be32u(&s->packed_headers_stream);
                if (tp_header_size < 4 || tp_header_size > bytestream2_get_bytes_left(&s->packed_headers_stream) + 4) {
                    av_log(s->avctx, AV_LOG_ERROR, "Invalid tile-part header size: %u.\n", tp_header_size);
                    return AVERROR_INVALIDDATA;
                }
