uint32_t tp_header_size;
                if (bytestream2_get_bytes_left(&s->g) < 4) {
                    av_log(s->avctx, AV_LOG_ERROR, "Insufficient data for tile-part header size\n");
                    return AVERROR_INVALIDDATA;
                }
                tp_header_size = bytestream2_get_be32u(&s->g);
                if (tp_header_size < 1 || tp_header_size > MAX_TILE_PART_HEADER_SIZE) {
                    av_log(s->avctx, AV_LOG_ERROR, "Invalid tile-part header size: %u\n", tp_header_size);
                    return AVERROR_INVALIDDATA;
