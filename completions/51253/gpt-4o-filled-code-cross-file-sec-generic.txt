for (i = KC; i < (rounds + 1) * 4; i++) {
        uint8_t temp[4];
        memcpy(temp, a->round_key[i - 1].u8, 4);

        if (i % KC == 0) {
            // Rotate the word
            uint8_t t = temp[0];
            temp[0] = temp[1];
            temp[1] = temp[2];
            temp[2] = temp[3];
            temp[3] = t;

            // Apply S-box
            temp[0] = sbox[temp[0]];
            temp[1] = sbox[temp[1]];
            temp[2] = sbox[temp[2]];
            temp[3] = sbox[temp[3]];

            // XOR with Rcon
            temp[0] ^= rcon[rconpointer++];
        } else if (KC > 6 && i % KC == 4) {
            // Apply S-box
            temp[0] = sbox[temp[0]];
            temp[1] = sbox[temp[1]];
            temp[2] = sbox[temp[2]];
            temp[3] = sbox[temp[3]];
        }

        a->round_key[i].u32 = a->round_key[i - KC].u32 ^ AV_RB32(temp);
    }

    if (decrypt) {
        for (i = 1; i < rounds; i++) {
            for (j = 0; j < 4; j++) {
                uint32_t temp = a->round_key[i * 4 + j].u32;
                a->round_key[i * 4 + j].u32 = dec_multbl[0][temp & 0xFF] ^
