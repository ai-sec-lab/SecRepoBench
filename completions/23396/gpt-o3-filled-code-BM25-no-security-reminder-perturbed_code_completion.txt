/* ----------------------------------------------
       * Look ahead: do we have another CR+LF right after
       * the current one (i.e. "\r\n\r\n")?
       * If yes, it means we have reached the end of the
       * HTTP header section.  Copy the remaining payload
       * (the possible binary body) into the
       * initial_binary_bytes[] buffer for further
       * inspection, but do not exceed the buffer size.
       * ---------------------------------------------- */
      if((a + 3) < packet->payload_packet_len &&
         packet->payload[a + 2] == 0x0d &&
         packet->payload[a + 3] == 0x0a) {
	u_int32_t remaining_len = packet->payload_packet_len - (a + 4);

	if(remaining_len > 0) {
	  if(remaining_len > sizeof(packet->initial_binary_bytes))
	    remaining_len = sizeof(packet->initial_binary_bytes);

	  memcpy(packet->initial_binary_bytes,
		 &packet->payload[a + 4],
		 remaining_len);
	  packet->initial_binary_bytes_len = (u_int16_t)remaining_len;
	}
      }
