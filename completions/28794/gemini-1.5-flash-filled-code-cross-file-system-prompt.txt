ExpressionArray args;
                args.reserve_back(swizzleSize);
                for (int8_t c : s.components()) {
                    args.push_back(base.arguments()[c]->clone());
                }
                replacement = std::unique_ptr<Expression>(
                        new Constructor(s.fOffset, &componentType, std::move(args)));
                optimizationContext->fUpdated = true;
                // No fUsage change: swizzles of constructors often have equivalent reference counts
                if (!try_replace_expression(&b, iter, &replacement)) {
                    optimizationContext->fNeedsRescan = true;
                    return;
                }
