ExpressionArray reordered;
                std::vector<int> argumentIndices(swizzleSize);
                std::vector<bool> used(base.arguments().size(), false);
                int nextIndex = 0;
                for (int component : s.components()) {
                    argumentIndices[nextIndex++] = component;
                    used[component] = true;
                }
                for (size_t i = 0; i < base.arguments().size(); ++i) {
                    if (used[i]) {
                        reordered.push_back(base.arguments()[i]->clone());
                    }
                }
                replacement = std::unique_ptr<Expression>(
                        new Constructor(base.fOffset, &componentType, std::move(reordered)));
                optimizationContext->fUpdated = true;
                // No fUsage change: swizzles of constructors have equivalent reference counts
                if (!try_replace_expression(&b, iter, &replacement)) {
                    optimizationContext->fNeedsRescan = true;
                    return;
                }
