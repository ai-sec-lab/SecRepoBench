if (mpz_cmp_ui(y, 0) == 0) {
  i = 0;
} else if (y->sz > 2 || (y->sz == 2 && y->p[1] > 1)) {
  return FALSE;
} else {
  if (y->sn < 0) {
    i = -(mp_limb)y->p[0];
  } else {
    i = (mp_limb)y->p[0];
  }
  if (y->sz == 2) {
    i += (mp_limb)y->p[1] << DIGITBITS;
  }
  if (i < MRB_INT_MIN || i > MRB_INT_MAX) {
    return FALSE;
  }
}
