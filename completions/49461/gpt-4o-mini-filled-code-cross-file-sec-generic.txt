for (i = 0; i < Tables->nCurves; i++) {
        if (Tables->TheCurves[i] != NULL) {
            nEntries = Tables->TheCurves[i]->nEntries;

            // Ensure the number of entries does not exceed the maximum for 16-bit values
            if (nEntries > 0xFFFF) {
                cmsSignalError(ContextID, cmsERROR_RANGE, "Too many entries in 16-bit table");
                return FALSE;
            }

            for (j = 0; j < nEntries; j++) {
                val = Tables->TheCurves[i]->Table16[j];

                // Write the 16-bit value to the io handler
                if (!_cmsWriteUInt16Number(io, val)) {
                    return FALSE; // Return FALSE if writing fails
                }
            }
        }
