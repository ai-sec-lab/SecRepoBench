for (i = 0; i < Tables->nStages; i++) {
    nEntries = Tables->TheCurves[i]->nEntries;

    if (nEntries > 0xFFFF) {
        return FALSE;  // Prevent integer overflow
    }

    for (j = 0; j < nEntries; j++) {
        val = Tables->TheCurves[i]->Table16[j];
        if (!_cmsWriteUInt16Number(io, val)) {
            return FALSE;
        }
    }
}
