// Safety check: if there are no tables, nothing to write
    if (Tables == NULL)
        return TRUE;

    // Walk through every curve contained in the Tables structure
    for (j = 0; j < Tables->nCurves; j++) {

        // For each curve, iterate across its entries
        for (index = 0; index < Tables->TheCurves[j]->nEntries; index++) {

            val = Tables->TheCurves[j]->Table16[index];

            // Write the 16-bit value.  Abort on failure.
            if (!_cmsWriteUInt16Number(io, val))
                return FALSE;
        }
    }

    return TRUE;
