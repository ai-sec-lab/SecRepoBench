const char
    *p;

  assert(expressionstr != (const char *) NULL);
  assert(name != (const char *) NULL);

  /*  Does the expression start with the requested function name?  */
  if (LocaleNCompare(expressionstr,name,length) != 0)
    return(MagickFalse);

  /*  Skip over the function name and any white-space that follows.  */
  p=expressionstr+length;
  while (isspace((int) ((unsigned char) *p)) != 0)
    p++;

  /*  A valid function call must be followed by an opening parenthesis.  */
  return(*p == '(' ? MagickTrue : MagickFalse);
