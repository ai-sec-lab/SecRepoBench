    n = _cmsSearchTag(Icc, tagSignature, TRUE);
    if (n < 0) {
        _cmsUnlockMutex(Icc->ContextID, Icc ->UsrMutex); // Tag not found
        return NULL;
    }

    // Has this tag already been read or written?
    if (Icc->TagPtrs[n] != NULL) {
        _cmsUnlockMutex(Icc->ContextID, Icc ->UsrMutex);
        return Icc->TagPtrs[n];                       // Cached copy
