unsigned int numIndices;
        sz = buffer;
        SkipSpaces(&sz);
        numIndices = strtoul10(sz, &sz);
        if (numIndices > 256) {
            ASSIMP_LOG_ERROR("OFF: Too many indices per face");
            numIndices = 3;
        }
        if (numIndices < 3) {
            ASSIMP_LOG_ERROR("OFF: Faces with less than 3 indices are not supported");
            continue;
        }
        aiFace& face = faces[i++];
        face.mNumIndices = numIndices;
        face.mIndices = new unsigned int[numIndices];
        for (unsigned int j = 0; j < numIndices; ++j) {
            SkipSpaces(&sz);
            unsigned int index = strtoul10(sz, &sz);
            if (index >= numVertices) {
                ASSIMP_LOG_ERROR("OFF: Index out of range");
                index = 0;
            }
            face.mIndices[j] = index;
        }
