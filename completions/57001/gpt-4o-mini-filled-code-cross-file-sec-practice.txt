// load faces with their indices
    faces = mesh->mFaces;
    for (unsigned int i = 0; i < numFaces; ) {
        // Read the number of vertices for the current face
        if (!GetNextLine(buffer, line)) {
            ASSIMP_LOG_ERROR("OFF: The number of faces in the header is incorrect");
            break;
        }
        const char *sz = line;
        SkipSpaces(&sz);
        if (IsLineEnd(*sz)) {
            ASSIMP_LOG_ERROR("OFF: Empty face line encountered");
            break;
        }

        unsigned int numIndices = strtoul10(sz, &sz);
        if (numIndices < 3) {
            ASSIMP_LOG_ERROR("OFF: A face must have at least 3 indices");
            continue; // Skip this face
        }

        if (i >= numFaces) {
            ASSIMP_LOG_ERROR("OFF: Too many faces specified");
            break; // Prevent overflow
        }

        aiFace& face = faces[i++];
        face.mNumIndices = numIndices;
        face.mIndices = new unsigned int[numIndices];

        // Read the vertex indices for the face
        for (unsigned int j = 0; j < numIndices; ++j) {
            unsigned int index = strtoul10(sz, &sz);
            if (index >= numVertices) {
                ASSIMP_LOG_ERROR("OFF: Index out of range");
                index = 0; // Set to a default value to avoid crashes
            }
            face.mIndices[j] = index;
        }
    }

    if (i != numFaces) {
        ASSIMP_LOG_ERROR("OFF: Not enough faces read");
