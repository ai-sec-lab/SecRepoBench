// Read face data
        for (; i < numFaces; ++i) {
            if (!GetNextLine(buffer, line)) {
                ASSIMP_LOG_ERROR("OFF: The number of faces in the header is incorrect");
                break;
            }
            sz = line;
            SkipSpaces(&sz);

            // Read the number of vertices for the face
            unsigned int numIndices = strtoul10(sz, &sz);
            if (numIndices < 3) {
                ASSIMP_LOG_ERROR("OFF: Faces with less than 3 vertices are not supported");
                continue;
            }

            aiFace& face = faces[i];
            face.mNumIndices = numIndices;
            face.mIndices = new unsigned int[numIndices];

            // Read the vertex indices
            for (unsigned int j = 0; j < numIndices; ++j) {
                SkipSpaces(&sz);
                unsigned int index = strtoul10(sz, &sz);
                if (index >= numVertices) {
                    ASSIMP_LOG_ERROR("OFF: Vertex index out of range");
                    index = 0; // Set to a valid index to avoid crashes
                }
                face.mIndices[j] = index;
            }
        }
