unsigned int numIndices = strtoul10(car, &car);
if (numIndices > AI_MAX_FACE_INDICES) {
    ASSIMP_LOG_WARN("OFF: Skipping face with more than AI_MAX_FACE_INDICES indices");
    SkipLine(&car);
    continue;
}
if (numIndices < 3) {
    ASSIMP_LOG_WARN("OFF: Skipping face with less than 3 indices");
    SkipLine(&car);
    continue;
}

faces->mNumIndices = numIndices;
faces->mIndices = new unsigned int[numIndices];

for (unsigned int idx = 0; idx < numIndices; ++idx) {
    NextToken(&car, end);
    if (car >= end) {
        ASSIMP_LOG_ERROR("OFF: Unexpected end of file while reading face indices");
        break;
    }
    unsigned int tmp = strtoul10(car, &car);
    if (tmp >= numVertices) {
        ASSIMP_LOG_ERROR("OFF: Vertex index out of range");
        tmp = 0;
    }
    faces->mIndices[idx] = tmp;
}

NextToken(&car, end);
++faces;
++i;
