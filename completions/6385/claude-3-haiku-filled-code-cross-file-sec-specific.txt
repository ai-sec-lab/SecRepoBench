// Update the image's horizontal and vertical resolution based on parsed geometry information.
// The approach may vary depending on whether specific flags (RhoValue and SigmaValue) are set.
if ((flags & RhoValue) != 0)
{
  image->resolution.x = geometry_info.rho;
}
if ((flags & SigmaValue) != 0)
{
  image->resolution.y = geometry_info.sigma;
}
else
{
  image->resolution.x = geometry_info.rho;
  image->resolution.y = geometry_info.rho;
}
