for (i = 0; i < 3; i++) {
                int list_idx_c = cand_lists[i];
                int j;

                for (j = 0; j < s->rps[list_idx_c].nb_refs &&
                               rpl_tmp.nb_refs < HEVC_MAX_REFS;
                     j++) {
                    if (s->rps[list_idx_c].used[j]) {
                        rpl_tmp.list[rpl_tmp.nb_refs]       = s->rps[list_idx_c].list[j];
                        rpl_tmp.ref[rpl_tmp.nb_refs]        = s->rps[list_idx_c].ref[j];
                        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = s->rps[list_idx_c].isLongTerm[j];
                        rpl_tmp.nb_refs++;
                    }
                }
            }
            if (s->ps.pps->pps_curr_pic_ref_enabled_flag) {
                rpl_tmp.list[rpl_tmp.nb_refs]       = s->ref->poc;
                rpl_tmp.ref[rpl_tmp.nb_refs]        = s->ref;
                rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
                rpl_tmp.nb_refs++;
            }
