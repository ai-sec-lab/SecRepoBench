for (j = 0; j < 3 && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; j++) {
    int type = cand_lists[j];
    for (i = 0; i < s->rps[type].nb_refs && rpl_tmp.nb_refs < sh->nb_refs[list_idx] && rpl_tmp.nb_refs < HEVC_MAX_REFS; i++) {
        if (s->rps[type].used[i]) {
            rpl_tmp.list[rpl_tmp.nb_refs]       = s->rps[type].poc[i];
            rpl_tmp.ref[rpl_tmp.nb_refs]        = s->rps[type].ref[i];
            rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = s->rps[type].isLongTerm[i];
            rpl_tmp.nb_refs++;
        }
    }
}

if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < sh->nb_refs[list_idx] && rpl_tmp.nb_refs < HEVC_MAX_REFS) {
    rpl_tmp.list[rpl_tmp.nb_refs]       = s->poc;
    rpl_tmp.ref[rpl_tmp.nb_refs]        = s->ref;
    rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
    rpl_tmp.nb_refs++;
}
