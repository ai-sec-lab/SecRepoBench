for (int i = 0; i < 3 && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; i++) {
    for (int j = 0; j < s->rps[cand_lists[i]].nb_refs && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; j++) {
        if (s->rps[cand_lists[i]].ref[j]) {
            rpl_tmp.list[rpl_tmp.nb_refs] = s->rps[cand_lists[i]].ref[j]->poc;
            rpl_tmp.ref[rpl_tmp.nb_refs] = s->rps[cand_lists[i]].ref[j];
            rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = (cand_lists[i] == LT_CURR);
            rpl_tmp.nb_refs++;
        }
    }
}

if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
    rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->poc;
    rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref;
    rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 1;
    rpl_tmp.nb_refs++;
}
