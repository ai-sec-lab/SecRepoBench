for (i = 0; i < 3; i++) {
                int idx = cand_lists[i];
                if (s->rps[idx].nb_refs > 0) {
                    for (j = 0; j < s->rps[idx].nb_refs; j++) {
                        if (rpl_tmp.nb_refs < HEVC_MAX_REFS) {
                            rpl_tmp.list[rpl_tmp.nb_refs] = s->rps[idx].poc[j];
                            rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref->ref[s->rps[idx].used[j]];
                            rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = s->rps[idx].isLongTerm[j];
                            rpl_tmp.nb_refs++;
                        } else {
                            av_log(s->avctx, AV_LOG_WARNING, "Maximum reference count exceeded.\n");
                            break; // Prevent exceeding the maximum number of references
                        }
                    }
                }
            }

            // If current picture reference is enabled, add it to the reference list
            if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < HEVC_MAX_REFS) {
                rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->poc;
                rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref;
                rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0; // Current picture is not long-term
                rpl_tmp.nb_refs++;
            }
