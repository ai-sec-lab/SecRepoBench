for (int j = 0; j < 3 && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; j++) {
    for (int i = 0; i < s->rps[cand_lists[j]].nb_refs && rpl_tmp.nb_refs < HEVC_MAX_REFS; i++) {
        rpl_tmp.list[rpl_tmp.nb_refs] = s->rps[cand_lists[j]].list[i];
        rpl_tmp.ref[rpl_tmp.nb_refs] = s->rps[cand_lists[j]].ref[i];
        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = cand_lists[j] == LT_CURR;
        rpl_tmp.nb_refs++;
    }
}

if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < HEVC_MAX_REFS) {
    rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->poc;
    rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref;
    rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
    rpl_tmp.nb_refs++;
}
