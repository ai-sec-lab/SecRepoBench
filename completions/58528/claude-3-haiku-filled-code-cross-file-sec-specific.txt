while (rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
    for (i = 0; i < 3; i++) {
        int cand_list = cand_lists[i];
        int nb_refs = s->rps[cand_list].nb_refs;
        for (j = 0; j < nb_refs; j++) {
            if (rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
                rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->refPicList[cand_list].list[j];
                rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref->refPicList[cand_list].ref[j];
                rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = s->ref->refPicList[cand_list].isLongTerm[j];
                rpl_tmp.nb_refs++;
            }
        }
    }
    if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
        rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->poc;
        rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref;
        rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
        rpl_tmp.nb_refs++;
    }
