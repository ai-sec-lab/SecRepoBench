for (j = 0; j < 3; j++) {
                int cand_idx = cand_lists[j];
                if (s->rps[cand_idx].nb_refs > 0) {
                    for (i = 0; i < s->rps[cand_idx].nb_refs; i++) {
                        if (rpl_tmp.nb_refs < HEVC_MAX_REFS) {
                            rpl_tmp.list[rpl_tmp.nb_refs] = s->rps[cand_idx].poc[i];
                            rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref->ref[s->rps[cand_idx].used[i]];
                            rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = s->rps[cand_idx].used[i];
                            rpl_tmp.nb_refs++;
                        }
                    }
                }
            }
            if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < HEVC_MAX_REFS) {
                rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->poc;
                rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref;
                rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
                rpl_tmp.nb_refs++;
            }
