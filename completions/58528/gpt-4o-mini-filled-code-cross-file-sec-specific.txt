for (i = 0; i < 3; i++) {
                int cand_idx = cand_lists[i];
                if (s->rps[cand_idx].nb_refs > 0) {
                    for (j = 0; j < s->rps[cand_idx].nb_refs; j++) {
                        if (rpl_tmp.nb_refs < HEVC_MAX_REFS) {
                            rpl_tmp.list[rpl_tmp.nb_refs] = s->rps[cand_idx].poc[j];
                            rpl_tmp.ref[rpl_tmp.nb_refs] = s->rps[cand_idx].ref[j];
                            rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = s->rps[cand_idx].isLongTerm[j];
                            rpl_tmp.nb_refs++;
                        }
                    }
                }
            }
            if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < HEVC_MAX_REFS) {
                rpl_tmp.list[rpl_tmp.nb_refs] = s->ref->poc;
                rpl_tmp.ref[rpl_tmp.nb_refs] = s->ref;
                rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0; // Assuming current picture is not long-term
                rpl_tmp.nb_refs++;
            }
