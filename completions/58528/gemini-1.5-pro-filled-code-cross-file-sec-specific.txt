for (i = 0; i < 3 && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; i++) {
    int type = cand_lists[i];
    ShortTermRPS *st_rps = &s->rps[type];

    for (j = 0; j < st_rps->nb_refs && rpl_tmp.nb_refs < sh->nb_refs[list_idx]; j++) {
        if (st_rps->used[j]) {
            rpl_tmp.list[rpl_tmp.nb_refs]       = st_rps->poc[j];
            rpl_tmp.ref[rpl_tmp.nb_refs]        = st_rps->ref[j];
            rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = st_rps->is_long_term[j];
            rpl_tmp.nb_refs++;
        }
    }
}

if (s->ps.pps->pps_curr_pic_ref_enabled_flag && rpl_tmp.nb_refs < sh->nb_refs[list_idx]) {
    rpl_tmp.list[rpl_tmp.nb_refs]       = s->poc;
    rpl_tmp.ref[rpl_tmp.nb_refs]        = s->ref;
    rpl_tmp.isLongTerm[rpl_tmp.nb_refs] = 0;
    rpl_tmp.nb_refs++;
}
