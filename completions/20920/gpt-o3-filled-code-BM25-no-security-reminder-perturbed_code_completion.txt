/*
        The IPTC profile is stored as an array of 32–bit words (type LONG).
        If the file is byte–swapped we must correct the byte order before
        we pass the data to the generic profile reader.  The first four
        bytes contain a Photoshop length field—skip it.
      */
      if (TIFFIsByteSwapped(tiff) != 0)
        TIFFSwabArrayOfLong((uint32 *) profile,length/4);

      if (length > 4)
        status=ReadProfile(img,"iptc",profile+4,(ssize_t) (length-4),exception);
      else
        status=ReadProfile(img,"iptc",profile,(ssize_t) length,exception);
