if (s->codec_id == AV_CODEC_ID_H264 ||
        s->codec_id == AV_CODEC_ID_VP8 ||
        s->codec_id == AV_CODEC_ID_VP9 ||
        s->codec_id == AV_CODEC_ID_HEVC ||
        s->width < 32 || s->height < 32) {
        h_align += 2;
    }

    *width = FFALIGN(*width, w_align);
    *height = FFALIGN(*height, h_align);
