  size_t y_digits = y->sz;

  if (y_digits < x_digits) {
    return -1;
  } else if (y_digits > x_digits) {
    return 1;
  }

  // If the number of digits is the same, compare each digit starting from the most significant
  for (size_t i = x_digits; i > 0; i--) {
    if (y->p[i - 1] > x->p[i - 1]) {
      return 1;
    } else if (y->p[i - 1] < x->p[i - 1]) {
      return -1;
