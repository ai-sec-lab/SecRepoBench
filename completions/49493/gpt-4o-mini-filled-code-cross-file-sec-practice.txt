size_t x_size = mpz_sizeinbase(x, 10); // Assuming base 10 for digit comparison
  size_t y_size = mpz_sizeinbase(y, 10);

  if (y_size < x_size) {
    return -1; // y is smaller
  } else if (y_size > x_size) {
    return 1; // y is larger
  }

  // If they have the same number of digits, compare digit by digit
  for (size_t i = x_size; i > 0; i--) {
    if (y->p[i - 1] > x->p[i - 1]) {
      return 1; // y's digit is greater
    } else if (y->p[i - 1] < x->p[i - 1]) {
      return -1; // y's digit is smaller
    }
  }

  return 0; // They are equal
