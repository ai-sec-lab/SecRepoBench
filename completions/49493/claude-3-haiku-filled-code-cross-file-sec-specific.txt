if (x->sz == 0 || y->sz == 0) {
  return 0;
}

int xdigits = digits(x);
int ydigits = digits(y);

if (ydigits < xdigits) {
  return -1;
} else if (ydigits > xdigits) {
  return 1;
}

for (int i = xdigits - 1; i >= 0; i--) {
  if (x->p[i] < y->p[i]) {
    return -1;
  } else if (x->p[i] > y->p[i]) {
    return 1;
  }
}

return 0;
