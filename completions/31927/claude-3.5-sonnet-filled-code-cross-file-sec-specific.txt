switch (ptr->sampleSize) {
    case 4:
        for (i = 0; i < ptr->sampleCount; i += 2) {
            u8 b = gf_bs_read_u8(bs);
            ptr->sizes[i] = b >> 4;
            if (i + 1 < ptr->sampleCount) {
                ptr->sizes[i + 1] = b & 0x0F;
            }
            if (ptr->max_size < ptr->sizes[i])
                ptr->max_size = ptr->sizes[i];
            ptr->total_size += ptr->sizes[i];
            ptr->total_samples++;
            if (i + 1 < ptr->sampleCount) {
                if (ptr->max_size < ptr->sizes[i + 1])
                    ptr->max_size = ptr->sizes[i + 1];
                ptr->total_size += ptr->sizes[i + 1];
                ptr->total_samples++;
            }
        }
        break;
    case 8:
        for (i = 0; i < ptr->sampleCount; i++) {
            ptr->sizes[i] = gf_bs_read_u8(bs);
            if (ptr->max_size < ptr->sizes[i])
                ptr->max_size = ptr->sizes[i];
            ptr->total_size += ptr->sizes[i];
            ptr->total_samples++;
        }
        break;
    case 16:
        for (i = 0; i < ptr->sampleCount; i++) {
            ptr->sizes[i] = gf_bs_read_u16(bs);
            if (ptr->max_size < ptr->sizes[i])
                ptr->max_size = ptr->sizes[i];
            ptr->total_size += ptr->sizes[i];
            ptr->total_samples++;
        }
        break;
}
