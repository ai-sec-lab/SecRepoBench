for (i = 0; i < ptr->sampleCount; ) {
    if (ptr->sampleSize == 4) {
        u8 val = gf_bs_read_int(bs, 4);
        ptr->sizes[i++] = val;
        if (i < ptr->sampleCount) {
            val = gf_bs_read_int(bs, 4);
            ptr->sizes[i++] = val;
        }
    } else {
        ptr->sizes[i] = gf_bs_read_int(bs, ptr->sampleSize);
        i++;
    }
    ptr->total_size += ptr->sizes[i-1];
    ptr->total_samples++;
    if (ptr->sizes[i-1] > ptr->max_size)
        ptr->max_size = ptr->sizes[i-1];
}
