for (i = 0; i < ptr->sampleCount; ) {
    if (ptr->sampleSize == 4) {
        // Compact sample size (4-bit)
        u8 sample_size = gf_bs_read_int(bs, 4);
        ptr->sizes[i++] = sample_size;
        if (i < ptr->sampleCount) {
            sample_size = gf_bs_read_int(bs, 4);
            ptr->sizes[i++] = sample_size;
        }
    } else {
        // Regular sample size
        ptr->sizes[i] = gf_bs_read_int(bs, ptr->sampleSize);
        ptr->max_size = MAX(ptr->max_size, ptr->sizes[i]);
        ptr->total_size += ptr->sizes[i];
        ptr->total_samples++;
        i++;
    }
}
