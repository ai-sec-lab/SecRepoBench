u32 val;
			if (ptr->sampleSize==4) {
				val = gf_bs_read_u8(bs);
				ptr->sizes[i] = val >> 4;
				if (ptr->max_size < ptr->sizes[i]) ptr->max_size = ptr->sizes[i];
				ptr->total_size += ptr->sizes[i];
				ptr->total_samples++;
				i++;
				if (i<ptr->sampleCount) {
					ptr->sizes[i] = val & 0xf;
					if (ptr->max_size < ptr->sizes[i]) ptr->max_size = ptr->sizes[i];
					ptr->total_size += ptr->sizes[i];
					ptr->total_samples++;
					i++;
				}
			} else {
				switch (ptr->sampleSize) {
				case 8:
					ptr->sizes[i] = gf_bs_read_u8(bs);
					break;
				case 16:
					ptr->sizes[i] = gf_bs_read_u16(bs);
					break;
				}
				if (ptr->max_size < ptr->sizes[i]) ptr->max_size = ptr->sizes[i];
				ptr->total_size += ptr->sizes[i];
				ptr->total_samples++;
				i++;
