// Read sample sizes from the bitstream
			if (ptr->sampleSize == 4) {
				// Read two samples per byte for compact sizes
				for (u32 byteIndex = 0; byteIndex < (ptr->sampleCount + 1) / 2; byteIndex++) {
					u8 byte = gf_bs_read_u8(bs);
					if (byteIndex * 2 < ptr->sampleCount) {
						ptr->sizes[byteIndex * 2] = byte & 0x0F; // lower nibble
					}
					if (byteIndex * 2 + 1 < ptr->sampleCount) {
						ptr->sizes[byteIndex * 2 + 1] = (byte >> 4) & 0x0F; // upper nibble
					}
				}
			} else {
				// Read regular sample sizes
				for (i = 0; i < ptr->sampleCount; i++) {
					ptr->sizes[i] = gf_bs_read_u32(bs);
					if (ptr->max_size < ptr->sizes[i]) {
						ptr->max_size = ptr->sizes[i];
					}
					ptr->total_size += ptr->sizes[i];
					ptr->total_samples++;
				}
			}
