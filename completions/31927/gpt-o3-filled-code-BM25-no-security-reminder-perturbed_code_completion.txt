				u8 v = gf_bs_read_u8(bs);

				/* first sample (high nibble) */
				ptr->sizes[index] = (v >> 4) & 0x0F;
				if (ptr->max_size < ptr->sizes[index]) ptr->max_size = ptr->sizes[index];
				ptr->total_size   += ptr->sizes[index];
				ptr->total_samples++;
				index++;

				/* second sample (low nibble) â€“ only if we still have data */
				if (index < ptr->sampleCount) {
					ptr->sizes[index] = v & 0x0F;
					if (ptr->max_size < ptr->sizes[index]) ptr->max_size = ptr->sizes[index];
					ptr->total_size   += ptr->sizes[index];
					ptr->total_samples++;
					index++;
				}
			} else if (ptr->sampleSize == 8) {
				ptr->sizes[index] = gf_bs_read_u8(bs);
				if (ptr->max_size < ptr->sizes[index]) ptr->max_size = ptr->sizes[index];
				ptr->total_size   += ptr->sizes[index];
				ptr->total_samples++;
				index++;
			} else if (ptr->sampleSize == 16) {
				ptr->sizes[index] = gf_bs_read_u16(bs);
				if (ptr->max_size < ptr->sizes[index]) ptr->max_size = ptr->sizes[index];
				ptr->total_size   += ptr->sizes[index];
				ptr->total_samples++;
				index++;
