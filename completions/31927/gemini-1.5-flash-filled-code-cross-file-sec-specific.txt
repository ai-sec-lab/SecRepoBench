u32 sampleSize = ptr->sampleSize;
			if (sampleSize == 4) {
				for (; i < ptr->sampleCount; i += 2) {
					u8 val = gf_bs_read_u8(bs);
					ptr->sizes[i] = (val >> 4) & 0xF;
					if (i + 1 < ptr->sampleCount) {
						ptr->sizes[i + 1] = val & 0xF;
					}
					if (ptr->max_size < ptr->sizes[i]) ptr->max_size = ptr->sizes[i];
					ptr->total_size += ptr->sizes[i];
					ptr->total_samples++;
					if (i + 1 < ptr->sampleCount) {
						if (ptr->max_size < ptr->sizes[i + 1]) ptr->max_size = ptr->sizes[i + 1];
						ptr->total_size += ptr->sizes[i + 1];
						ptr->total_samples++;
					}
				}
			} else {
				for (; i < ptr->sampleCount; i++) {
					ptr->sizes[i] = gf_bs_read_u32(bs);
					if (ptr->max_size < ptr->sizes[i]) ptr->max_size = ptr->sizes[i];
					ptr->total_size += ptr->sizes[i];
					ptr->total_samples++;
				}
