switch (ptr->sampleSize) {
    case 4:
        if (i + 1 < ptr->sampleCount) {
            u8 val = gf_bs_read_u8(bs);
            ptr->sizes[i] = val >> 4;
            ptr->sizes[i+1] = val & 0x0F;
            i += 2;
        } else {
            ptr->sizes[i] = gf_bs_read_int(bs, 4);
            i++;
        }
        break;
    case 8:
        ptr->sizes[i] = gf_bs_read_u8(bs);
        i++;
        break;
    case 16:
        ptr->sizes[i] = gf_bs_read_u16(bs);
        i++;
        break;
    default:
        return GF_ISOM_INVALID_FILE;
}
if (ptr->max_size < ptr->sizes[i-1])
    ptr->max_size = ptr->sizes[i-1];
ptr->total_size += ptr->sizes[i-1];
ptr->total_samples++;
