u32 val;

				ptr->sizes[i] = val >> 4;

				ptr->total_size += ptr->sizes[i];

					ptr->total_size += ptr->sizes[i];
					ptr->total_samples++;
					i++;
				}

				val = gf_bs_read_u8(bs);
				if (ptr->sampleSize==8) {
					ptr->sizes[i] = val;
				} else {
					ptr->sizes[i] = (val<<8) | gf_bs_read_u8(bs);
