/* Release the original packed buffer if it differs from the buffer we
     * will continue working with (this happens when a decoder generates
     * a new buffer).
     */
    if (mp_buf && mp_buf != tmp_out_buf) {
        flb_free(mp_buf);
    }

    /* Resolve the time key to be used */
    if (parser->time_key) {
        time_key = parser->time_key;
    }
    else {
        time_key = "time";
    }
    slen = strlen(time_key);

    /* Iterate over the map looking for the time key */
    map_size = map.via.map.size;
    skip      = -1;          /* index to skip if we remove the time key   */
    k         = NULL;
    time_value = NULL;

    for (i = 0; i < map_size; i++) {
        msgpack_object *key_obj = &map.via.map.ptr[i].key;

        /* Only compare string keys */
        if (key_obj->type != MSGPACK_OBJECT_STR) {
            continue;
        }

        /* Quick length check before doing a memory compare */
        if (key_obj->via.str.size != (uint32_t) slen) {
            continue;
        }

        if (memcmp(key_obj->via.str.ptr, time_key, slen) == 0) {
            /* Found the desired time key */
            k = key_obj;
            time_value = &map.via.map.ptr[i].val;

            if (parser->time_keep == FLB_FALSE) {
                skip = i;           /* mark this entry to be skipped later */
            }
            break;
        }
    }
