static const struct {
    int class;
    int index;
    const uint8_t *bits;
    const uint8_t *val;
} default_huffman_table_param[] = {
    {0, 0, avpriv_mjpeg_bits_dc_luminance, avpriv_mjpeg_val_dc},
    {0, 1, avpriv_mjpeg_bits_dc_chrominance, avpriv_mjpeg_val_dc},
    {1, 0, avpriv_mjpeg_bits_ac_luminance, avpriv_mjpeg_val_ac_luminance},
    {1, 1, avpriv_mjpeg_bits_ac_chrominance, avpriv_mjpeg_val_ac_chrominance},
};

for (i = 0; i < FF_ARRAY_ELEMS(default_huffman_table_param); i++) {
    int class = default_huffman_table_param[i].class;
    int index = default_huffman_table_param[i].index;

    if (ff_mjpeg_build_vlc(&s->vlcs[class][index],
                           default_huffman_table_param[i].bits,
                           default_huffman_table_param[i].val,
                           class, s->avctx) < 0) {
        return AVERROR_INVALIDDATA;
    }

    if (class < 2) {
        memcpy(s->bits_data[class][index], default_huffman_table_param[i].bits, 16);
        memcpy(s->val_data[class][index], default_huffman_table_param[i].val, 256);
    }
}
