        { 0, 0, 12, 0x0000000000000000ULL },
        { 1, 0, 12, 0x0000000000000000ULL },
        { 0, 1, 12, 0x0000000000000000ULL },
        { 1, 1, 12, 0x0000000000000000ULL },
        { 0, 0, 11, 0x0000000000000000ULL },
        { 1, 0, 11, 0x0000000000000000ULL },
        { 0, 1, 11, 0x0000000000000000ULL },
        { 1, 1, 11, 0x0000000000000000ULL },
        { 0, 0, 10, 0x0000000000000000ULL },
        { 1, 0, 10, 0x0000000000000000ULL },
        { 0, 1, 10, 0x0000000000000000ULL },
        { 1, 1, 10, 0x0000000000000000ULL },
    };
    int i;
    for (i = 0; i < 12; i++) {
        int ret = build_vlc(&s->vlcs[0][i], ht[i].bits, 1 << ht[i].bits,
                            ht[i].values, 1, 1, 0);
        if (ret < 0)
            return ret;
        if (ht[i].class < 2) {
            s->huff_len[i] = ht[i].bits;
            s->huff_val[i] = ht[i].values;
        }
