int i;
for (i = 0; i < 4; i++) {
    int class = ht[i].class;
    int index = ht[i].index;
    int bits  = ht[i].bits;
    int huffval = ht[i].huffval;

    if (class < 2) {
        s->dc_bits[class][index]  = bits;
        s->dc_value[class][index] = huffval;
    } else {
        ff_mjpeg_build_huffman_codes(s->ac_vlc[class-2][index].bits,
                                     s->ac_vlc[class-2][index].huffval,
                                     bits, huffval);
    }
}

return 0;
