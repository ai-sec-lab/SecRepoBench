contour_points.reserve (num_contours * 2 + (phantom_only? 0 : num_points));
if (phantom_only)
  {
    for (HBUINT16 i = 0; i < num_contours; i++)
      {
        contour_points.emplace_back (endPtsOfContours[i] & 0xFFFF);
        contour_points.emplace_back (endPtsOfContours[i] >> 16);
      }
  }
else
  {
    for (HBUINT16 i = 0; i < num_contours; i++)
      {
        contour_points.emplace_back (endPtsOfContours[i] & 0xFFFF);
        contour_points.emplace_back (endPtsOfContours[i] >> 16);
        contour_points.emplace_back (endPtsOfContours[i + 1] & 0xFFFF);
        contour_points.emplace_back (endPtsOfContours[i + 1] >> 16);
      }
  }
