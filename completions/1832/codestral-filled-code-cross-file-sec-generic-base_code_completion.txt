if (s->num_primitive_matrices > max_primitive_matrices) {
    av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds the maximum allowed.\n");
    return AVERROR_INVALIDDATA;
}

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    s->matrix_params[mat].output_channel = get_bits(gbp, 4);
    s->matrix_params[mat].fractional_bits = get_bits(gbp, 4);
    s->matrix_params[mat].bypass_lsbs = get_bits1(gbp);

    if (s->matrix_params[mat].output_channel > s->max_matrix_channel) {
        av_log(m->avctx, AV_LOG_ERROR, "Output channel exceeds the maximum matrix channel.\n");
        return AVERROR_INVALIDDATA;
    }

    if (s->matrix_params[mat].fractional_bits > 14) {
        av_log(m->avctx, AV_LOG_ERROR, "Fractional bits exceed the maximum allowed.\n");
        return AVERROR_INVALIDDATA;
    }

    for (ch = 0; ch < s->max_matrix_channel; ch++) {
        s->matrix_params[mat].coeff[ch] = get_sbits(gbp, 16) * (1 << s->matrix_params[mat].fractional_bits);
    }

    if (s->noise_type) {
        s->matrix_params[mat].noise_shift = get_bits(gbp, 4);
    } else {
        s->matrix_params[mat].noise_shift = 0;
        s->max_matrix_channel += 2;
    }
}
