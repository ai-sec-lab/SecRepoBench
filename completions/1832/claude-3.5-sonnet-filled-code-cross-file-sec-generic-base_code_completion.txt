if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d (max %d)\n",
               s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        int32_t coeff_mask = 0;
        s->matrix_out_ch[mat] = get_bits(gbp, 4);
        s->matrix_frac_bits[mat] = get_bits(gbp, 4);
        s->lsb_bypass[mat] = get_bits1(gbp);

        if (s->matrix_out_ch[mat] >= MAX_CHANNELS) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix output channel: %d\n", s->matrix_out_ch[mat]);
            return AVERROR_INVALIDDATA;
        }

        if (s->matrix_frac_bits[mat] > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix fractional bits: %d\n", s->matrix_frac_bits[mat]);
            return AVERROR_INVALIDDATA;
        }

        if (s->matrix_out_ch[mat] > s->max_matrix_channel)
            s->max_matrix_channel = s->matrix_out_ch[mat];
    }

    if (!s->noise_type)
        s->max_matrix_channel += 2;

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        for (ch = 0; ch <= s->max_matrix_channel; ch++) {
            int32_t coeff = get_sbits(gbp, 16);
            s->matrix_coeff[mat][ch] = coeff * (1 << s->matrix_frac_bits[mat]);
        }
    }

    if (s->noise_type) {
        for (mat = 0; mat < s->num_primitive_matrices; mat++)
            s->matrix_noise_shift[mat] = get_bits(gbp, 4);
    } else {
        for (mat = 0; mat < s->num_primitive_matrices; mat++)
            s->matrix_noise_shift[mat] = 0;
    }

    return 0;
