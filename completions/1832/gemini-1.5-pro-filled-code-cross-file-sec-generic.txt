or the number of primitive matrices is invalid, return an error code.

    if (s->num_primitive_matrices > max_primitive_matrices || !s->num_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Invalid number of primitive matrices: %d.\n",
               s->num_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->matrix_out_ch[mat]    = get_bits(gbp, 4);
        s->matrix_frac_bits[mat] = get_bits(gbp, 4);
        s->matrix_lsb_bypass[mat] = get_bits1(gbp);

        if (s->matrix_out_ch[mat] >= m->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix output channel: %d.\n",
                   s->matrix_out_ch[mat]);
            return AVERROR_INVALIDDATA;
        }
        if (s->matrix_frac_bits[mat] > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix fractional bits: %d.\n",
                   s->matrix_frac_bits[mat]);
            return AVERROR_INVALIDDATA;
        }
    }

    if (!m->noise_type)
        m->max_matrix_channel = FFMAX(m->max_matrix_channel, s->max_matrix_channel + 2);

    for (ch = 0; ch < m->max_matrix_channel; ch++) {
        for (mat = 0; mat < s->num_primitive_matrices; mat++) {
            int coeff = get_sbits(gbp, 14);
            s->matrix_coeff[mat][ch] = coeff * (1 << s->matrix_frac_bits[mat]);
        }
    }

    if (m->noise_type) {
        s->noise_shift = get_bits(gbp, 4);
    } else {
        s->noise_shift = 0;
    }


    return 0;
