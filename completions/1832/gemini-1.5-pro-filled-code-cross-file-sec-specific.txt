or the number of primitive matrices exceeds the allowed maximum, return an error.

    if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d > %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->matrix[mat].output_channel = get_bits(gbp, 4);
        s->matrix[mat].fractional_bits = get_bits(gbp, 4);
        s->matrix[mat].lsb_bypass = get_bits1(gbp);

        if (s->matrix[mat].output_channel > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid output channel %d.\n",
                   s->matrix[mat].output_channel);
            return AVERROR_INVALIDDATA;

        }
        if (s->matrix[mat].fractional_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid fractional bits %d.\n",
                   s->matrix[mat].fractional_bits);
            return AVERROR_INVALIDDATA;
        }


        for (ch = 0; ch <= s->max_matrix_channel; ch++) {
            int coeff = get_sbits(gbp, 16);
            s->matrix_coeff[mat][ch] = coeff * (1 << s->matrix[mat].fractional_bits);
        }

        if (s->noise_type) {
            s->matrix_noise_shift[mat] = get_bits(gbp, 2);
        } else {
            s->matrix_noise_shift[mat] = 0;
        }
    }

    return 0;
