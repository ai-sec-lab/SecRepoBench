if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds maximum allowed.\n");
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        ch = get_bits(gbp, 4); // Read output channel
        if (ch > max_primitive_matrices) {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel %d exceeds maximum %d.\n", ch, max_primitive_matrices);
            return AVERROR_INVALIDDATA;
        }

        int fractional_bits = get_bits(gbp, 4); // Read fractional bits
        if (fractional_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits %d exceed maximum of 14.\n", fractional_bits);
            return AVERROR_INVALIDDATA;
        }

        // Check for noise type and adjust max channel if necessary
        if (s->noise_type) {
            s->max_matrix_channel = ch + 2; // Adjust for noise channels
        } else {
            s->max_matrix_channel = ch + 1;
        }

        // Read coefficients
        for (unsigned int i = 0; i < s->max_matrix_channel; i++) {
            int coeff = get_sbits(gbp, fractional_bits); // Read coefficient value
            s->coefficients[mat][i] = coeff * (1 << fractional_bits); // Scale coefficient
        }

        // Read noise shift value if noise type is set
        if (s->noise_type) {
            s->noise_shift = get_bits(gbp, 4);
        } else {
            s->noise_shift = 0;
        }
    }
