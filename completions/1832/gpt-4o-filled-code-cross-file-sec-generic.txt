if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d\n", s->num_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->matrix_out_ch[mat] = get_bits(gbp, 4);
        s->matrix_out_shift[mat] = get_bits(gbp, 4);
        s->matrix_out_lsb[mat] = get_bits1(gbp);

        if (s->matrix_out_ch[mat] > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel number too high: %d\n", s->matrix_out_ch[mat]);
            return AVERROR_INVALIDDATA;
        }

        if (s->matrix_out_shift[mat] > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Too many fractional bits: %d\n", s->matrix_out_shift[mat]);
            return AVERROR_INVALIDDATA;
        }

        for (ch = 0; ch <= s->max_matrix_channel; ch++) {
            s->matrix_coeff[mat][ch] = get_sbits(gbp, 9) * (1 << s->matrix_out_shift[mat]);
        }
    }

    if (s->noise_type) {
        s->matrix_noise_shift = get_bits(gbp, 4);
    } else {
        s->matrix_noise_shift = 0;
    }

    return 0;
