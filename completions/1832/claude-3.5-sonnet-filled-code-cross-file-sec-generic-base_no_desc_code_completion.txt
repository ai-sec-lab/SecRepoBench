if (s->num_primitive_matrices > max_primitive_matrices) {
    av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d\n", s->num_primitive_matrices);
    return AVERROR_INVALIDDATA;
}

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    int32_t *matrix = s->matrix_coeff[mat];
    int32_t coeff_mask = (1 << get_bits(gbp, 4)) - 1;
    int bypass_flag = get_bits1(gbp);
    int lsb_bypass = get_bits1(gbp);
    int sign_shift = get_bits(gbp, 4);

    if (sign_shift > 15) {
        av_log(m->avctx, AV_LOG_ERROR, "Invalid sign shift: %d\n", sign_shift);
        return AVERROR_INVALIDDATA;
    }

    for (ch = 0; ch <= s->max_matrix_channel; ch++) {
        matrix[ch] = get_sbits(gbp, 8) * (1 << sign_shift);
        if (!bypass_flag && !lsb_bypass)
            matrix[ch] &= coeff_mask;
    }
}

return 0;
