or the number of primitive matrices is invalid, return an error code.

    if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Too many primitive matrices: %d > %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->matrix_out_ch[mat]    = get_bits(gbp, 4);
        s->matrix_frac_bits[mat] = get_bits(gbp, 4);
        s->matrix_bypass[mat]    = get_bits1(gbp);

        ch = s->matrix_out_ch[mat];
        if (ch >= s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix output channel.\n");
            return AVERROR_INVALIDDATA;
        }
        if (s->matrix_frac_bits[mat] > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix fractional bits.\n");
            return AVERROR_INVALIDDATA;
        }

        if (!m->noise_type)
            s->max_matrix_channel = FFMAX(s->max_matrix_channel,
                                          ch + s->num_output_channels - 1);

        for (ch = 0; ch < s->num_output_channels; ch++)
            s->matrix_coeff[mat][ch] = get_bits(gbp, 5 + s->matrix_frac_bits[mat]) *
                                       (1 << s->matrix_frac_bits[mat]);
    }

    if (m->noise_type) {
        s->noise_shift = get_bits(gbp, 4);
        if (s->noise_shift > 11) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid noise shift value.\n");
            return AVERROR_INVALIDDATA;
        }
    } else {
        s->noise_shift = 0;
    }

    return 0;
