if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices %d exceeds maximum %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        ch = get_bits(gbp, 4); // Read output channel
        if (ch > max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel %d exceeds maximum %d.\n", ch, max_matrix_channel);
            return AVERROR_INVALIDDATA;
        }

        int fractional_bits = get_bits(gbp, 4); // Read fractional bits
        if (fractional_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits %d exceed maximum 14.\n", fractional_bits);
            return AVERROR_INVALIDDATA;
        }

        // Check if noise type is set and adjust max channel accordingly
        if (s->param_presence_flags & PARAM_NOISE) {
            max_channel = s->max_matrix_channel + 2; // Adjust for noise channels
        }

        // Read coefficient value
        int32_t coefficient = get_sbits(gbp, 16); // Assuming 16 bits for coefficient
        coefficient *= (1 << fractional_bits); // Scale by fractional bits

        // Store the coefficient in the appropriate structure
        s->channel_params[ch].coeff[mat] = coefficient;

        // Read noise shift value if noise type is set
        if (s->param_presence_flags & PARAM_NOISE) {
            s->channel_params[ch].noise_shift = get_bits(gbp, 4);
        } else {
            s->channel_params[ch].noise_shift = 0; // Set to zero if not applicable
        }
    }

    return 0; // Indicate success
