if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds maximum allowed.\n");
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        ch = get_bits(gbp, 4); // Read output channel
        if (ch >= s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel %u exceeds max channel %u.\n", ch, s->max_matrix_channel);
            return AVERROR_INVALIDDATA;
        }

        int fractional_bits = get_bits(gbp, 4); // Read fractional bits
        if (fractional_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits %d exceed maximum of 14.\n", fractional_bits);
            return AVERROR_INVALIDDATA;
        }

        // Read whether LSBs should be bypassed
        int bypass_lsb = get_bits1(gbp);
        
        // Read coefficient value
        int32_t coefficient = get_sbits(gbp, fractional_bits);
        coefficient *= (1 << fractional_bits); // Scale coefficient

        // Store the coefficient in the appropriate data structure
        s->channel_params[ch].coeff[mat] = coefficient;

        // If noise type is set, read and store the matrix noise shift value
        if (get_bits1(gbp)) {
            s->channel_params[ch].noise_shift = get_bits(gbp, 4);
        } else {
            s->channel_params[ch].noise_shift = 0; // Set to zero if not set
        }
    }

    return 0; // Success
