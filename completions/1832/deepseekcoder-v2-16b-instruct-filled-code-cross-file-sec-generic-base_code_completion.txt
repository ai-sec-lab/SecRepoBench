for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        ch = get_bits(gbp, 4);
        if (ch > m->avctx->channels) {
            av_log(m->avctx, AV_LOG_ERROR, "Channel index %d exceeds maximum allowed %d.\n", ch, m->avctx->channels);
            return AVERROR_INVALIDDATA;
        }

        int32_t coeff_value = get_bits(gbp, 14);
        if (coeff_value >= (1 << 14)) {
            av_log(m->avctx, AV_LOG_ERROR, "Coefficient value exceeds maximum allowed.\n");
            return AVERROR_INVALIDDATA;
        }

        int bypass = get_bits1(gbp);
        if (bypass && (coeff_value & 1)) {
            av_log(m->avctx, AV_LOG_ERROR, "LSBs should not be bypassed for non-zero coefficient values.\n");
            return AVERROR_INVALIDDATA;
        }

        if (s->noise_type) {
            s->matrix_noise_shift[mat] = get_bits(gbp, 4);
        } else {
            s->matrix_noise_shift[mat] = 0;
        }

        // Read channel parameters for each channel
        if ((ret = read_channel_params(m, substr, gbp, ch)) < 0) {
            return ret;
        }
    }

    // Validate the total number of primitive matrices
    if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices %d exceeds maximum allowed %d.\n", s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    // Adjust max matrix channel if noise type is set
    if (s->noise_type) {
        s->max_matrix_channel += 2; // Add 2 for noise channels
    }

    return 0;
}
