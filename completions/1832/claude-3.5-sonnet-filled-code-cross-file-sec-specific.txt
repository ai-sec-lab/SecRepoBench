if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d (max %d)\n",
               s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        int out_ch = get_bits(gbp, 4);
        int max_channel = m->avctx->codec_id == AV_CODEC_ID_MLP ? MAX_CHANNELS_MLP : MAX_CHANNELS_TRUEHD;

        if (out_ch >= max_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid output channel: %d\n", out_ch);
            return AVERROR_INVALIDDATA;
        }

        s->matrix_out_ch[mat] = out_ch;
        s->matrix_frac_bits[mat] = get_bits(gbp, 4);
        s->lsb_bypass[mat] = get_bits1(gbp);

        if (s->matrix_frac_bits[mat] > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid fractional bits: %d\n", s->matrix_frac_bits[mat]);
            return AVERROR_INVALIDDATA;
        }

        if (!s->noise_type)
            max_channel += 2;

        for (ch = 0; ch < max_channel; ch++) {
            int coeff = get_sbits(gbp, 16);
            s->matrix_coeff[mat][ch] = coeff * (1 << s->matrix_frac_bits[mat]);
        }

        s->matrix_noise_shift[mat] = s->noise_type ? get_bits(gbp, 4) : 0;
    }

    return 0;
