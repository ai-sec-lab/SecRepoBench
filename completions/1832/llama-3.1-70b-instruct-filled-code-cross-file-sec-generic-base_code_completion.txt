if (s->num_primitive_matrices > max_primitive_matrices) {
    av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds maximum allowed.\n");
    return AVERROR_INVALIDDATA;
}

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    unsigned int out_ch = get_bits(gbp, 5);
    unsigned int frac_bits = get_bits(gbp, 4);
    unsigned int bypass_lsb = get_bits1(gbp);

    if (out_ch >= s->max_matrix_channel) {
        if (!m->noise_type) {
            s->max_matrix_channel = out_ch + 1;
            if (s->max_matrix_channel > MAX_CHANNELS) {
                av_log(m->avctx, AV_LOG_ERROR, "Matrix channel exceeds maximum allowed.\n");
                return AVERROR_INVALIDDATA;
            }
        } else {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel exceeds maximum allowed.\n");
            return AVERROR_INVALIDDATA;
        }
    }

    if (frac_bits > 14) {
        av_log(m->avctx, AV_LOG_ERROR, "Fractional bits exceed maximum allowed.\n");
        return AVERROR_INVALIDDATA;
    }

    for (ch = 0; ch < s->max_matrix_channel; ch++) {
        int32_t coeff = get_sbits(gbp, 16) * (1 << frac_bits);
        s->matrix[mat][ch] = coeff;
    }

    if (m->noise_type) {
        s->matrix_noise_shift[mat] = get_bits(gbp, 4);
    } else {
        s->matrix_noise_shift[mat] = 0;
    }
}
