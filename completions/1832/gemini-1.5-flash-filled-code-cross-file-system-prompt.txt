return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        ch = get_bits(gbp, 5);
        if (ch > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Output channel exceeds maximum channel.\n");
            return AVERROR_INVALIDDATA;
        }
        s->primitive_matrix[mat].channel = ch;
        s->primitive_matrix[mat].frac_bits = get_bits(gbp, 4);
        if (s->primitive_matrix[mat].frac_bits > 14) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits exceed maximum.\n");
            return AVERROR_INVALIDDATA;
        }
        s->primitive_matrix[mat].bypass_lsbs = get_bits1(gbp);
        for (int i = 0; i <= ch; i++) {
            int coeff = get_sbits(gbp, 16);
            s->primitive_matrix[mat].coeff[i] = coeff * (1 << s->primitive_matrix[mat].frac_bits);
        }
        if (s->noise_type) {
            s->primitive_matrix[mat].noise_shift = get_bits(gbp, 3);
        } else {
            s->primitive_matrix[mat].noise_shift = 0;
        }
    }
    return 0;
