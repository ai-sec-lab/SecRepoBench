while (p < end && ISDIGIT(*p)) {
  if (nextnumber > MRB_INT_MAX / 10) {
    mrb_raise(mrb, E_RANGE_ERROR, "integer overflow");
  }
  nextnumber *= 10;
  int digit = *p - '0';
  if (nextnumber > MRB_INT_MAX - digit) {
    mrb_raise(mrb, E_RANGE_ERROR, "integer overflow");
  }
  nextnumber += digit;
  p++;
}
