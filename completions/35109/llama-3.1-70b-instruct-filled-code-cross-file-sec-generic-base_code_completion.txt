while (p < end && *p >= '0' && *p <= '9') {
  mrb_int d = *p - '0';
  if (mrb_int_mul_overflow(next_n, 10, &next_n) || mrb_int_add_overflow(next_n, d, &next_n)) {
    return NULL;
  }
  if (next_n > MRB_INT_MAX) {
    return NULL;
  }
  p++;
}
if (p > end) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "malformed format string");
}
