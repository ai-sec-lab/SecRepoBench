int overflow;
next_n = mrb_int_mul_overflow((mrb_int)*valp, 10, &overflow);
if (overflow || next_n > MRB_INT_MAX) {
  return NULL;
}
while (p < end && isdigit((unsigned char)*p)) {
  next_n = mrb_int_mul_overflow(next_n, 10, &overflow);
  if (overflow || next_n > MRB_INT_MAX) {
    return NULL;
  }
  next_n += *p++ - '0';
}
if (p == end) {
  mrb_raisef(mrb, E_ARGUMENT_ERROR, "malformed format string");
}
