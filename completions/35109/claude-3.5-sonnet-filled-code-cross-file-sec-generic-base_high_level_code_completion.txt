while (p < end && ISDIGIT(*p)) {
  mrb_int digit = *p - '0';
  if (next_n > (MRB_INT_MAX - digit) / 10) {
    return NULL;  // Overflow
  }
  next_n = next_n * 10 + digit;
  p++;
}

if (p == end) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "string not terminated");
}

if (p == (const char*)*valp) {
  return NULL;  // No digits found
}
