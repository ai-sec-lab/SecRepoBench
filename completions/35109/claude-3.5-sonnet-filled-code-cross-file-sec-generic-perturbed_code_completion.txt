mrb_int next_n = 0;
for (; p < end && ISDIGIT(*p); p++) {
  if (mrb_int_mul_overflow(next_n, 10, &next_n)) {
    return NULL;
  }
  if (next_n > MRB_INT_MAX / 10) {
    return NULL;
  }
  next_n += *p - '0';
  if (next_n < 0 || next_n > MRB_INT_MAX) {
    return NULL;
  }
}
if (p >= end) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "malformed format string - %%n");
}
nextnumber = next_n;
