while (p < limit && ISDIGIT(*p)) {
  mrb_int digit = *p - '0';
  if (mrb_int_mul_overflow(next_n, 10, &next_n) || 
      mrb_int_add_overflow(next_n, digit, &next_n) ||
      next_n > MRB_INT_MAX) {
    return NULL;
  }
  p++;
}

if (p == limit) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "malformed format string - %%*[0-9]");
}
