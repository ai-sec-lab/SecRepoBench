while (p < end && ISDIGIT(*p)) {
  if (mrb_int_mul_overflow(next_n, 10, &next_n)) {
    return NULL;
  }
  int digit = *p - '0';
  if (next_n > MRB_INT_MAX / 10 || (next_n == MRB_INT_MAX / 10 && digit > MRB_INT_MAX % 10)) {
    return NULL;
  }
  next_n = next_n * 10 + digit;
  p++;
}

if (p > end) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "malformed format string - %%*[0-9]");
}
