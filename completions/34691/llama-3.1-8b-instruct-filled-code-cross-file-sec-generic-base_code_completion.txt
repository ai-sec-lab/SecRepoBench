if (len > ARY_SHIFT_SHARED_MIN) {
  ary_make_shared(mrb, a);
  goto L_SHIFT;
}
else if (n == 0) {
  ARY_SET_LEN(a, 0);
  return mrb_nil_value();
}
else {
  mrb_value *ptr = ARY_PTR(a);
  mrb_int size = len-n;

  while (size--) {
    *ptr = *(ptr+n);
    ++ptr;
  }
  ARY_SET_LEN(a, len-n);
  return ARY_PTR(a)[0];
}
