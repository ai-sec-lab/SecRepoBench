/* argument check */
  if (n < 0) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "negative array size");
  }

  /* nothing to shift or empty array */
  if (n == 0 || len == 0) {
    return mrb_ary_new(mrb);
  }

  if (n > len) {
    n = len;
  }

  /* prepare modification */
  ary_modify(mrb, ary);

  /* build return value */
  if (ARY_SHARED_P(ary)) {
    val = ary_subseq(mrb, ary, 0, n);
  }
  else {
    val = mrb_ary_new_from_values(mrb, n, ARY_PTR(ary));
  }

  /* shrink self */
  if (len == n) {           /* shifted out everything */
    ARY_SET_LEN(ary, 0);
    return val;
  }

  if (len > ARY_SHIFT_SHARED_MIN) {
    /* make shared then just move the start pointer */
    ary_make_shared(mrb, ary);
    ary->as.heap.ptr += n;
    ARY_SET_LEN(ary, len - n);
    return val;
  }

  /* small array â€“ move the remaining elements */
  {
    mrb_value *ptr = ARY_PTR(ary);
    mrb_int  rest  = len - n;

    while (rest--) {
      *ptr = *(ptr + n);
      ++ptr;
    }
    ARY_SET_LEN(ary, len - n);
  }
  return val;
