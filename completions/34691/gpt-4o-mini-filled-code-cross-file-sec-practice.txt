mrb_get_args(mrb, "|i", &n); // Get the number of elements to shift, default to 1 if not provided
  if (n < 0) {
    mrb_raise(mrb, E_ARGUMENT_ERROR, "negative shift count");
  }
  
  if (n == 0) {
    return self; // No shift needed
  }

  if (n > len) {
    n = len; // Adjust n to not exceed the array length
  }

  val = ARY_PTR(a)[0]; // Store the first element to return
  if (ARY_SHARED_P(a)) {
    ary_modify(mrb, a); // Ensure we can modify the array
  }

  // Shift elements
  mrb_value *ptr = ARY_PTR(a);
  mrb_int size = len - n;

  while (size--) {
    *ptr = *(ptr + n);
    ++ptr;
  }
  ARY_SET_LEN(a, len - n); // Update the length of the array

  return val; // Return the shifted value
