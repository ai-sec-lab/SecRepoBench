// Iterate through the array to find the appropriate segment where the value should be mapped.
// If the value is beyond or equal to the current segment's fromCoord, adjust the value using the current segment's toCoord.
// Handle cases where two fromCoords are equal by using the previous toCoord for mapping.
// Calculate the denominator for linear interpolation between two segments.
int i;
for (i = 0; i < len; i++)
{
  if (inputvalue <= arrayZ[i].fromCoord)
  {
    return arrayZ[i].toCoord;
  }
  else if (i == len - 1)
  {
    return arrayZ[i].toCoord +
           ((arrayZ[i].toCoord - arrayZ[i-1].toCoord) *
            (inputvalue - arrayZ[i-1].fromCoord)) / (arrayZ[i].fromCoord - arrayZ[i-1].fromCoord);
  }
  else if (arrayZ[i].fromCoord == arrayZ[i+1].fromCoord)
  {
    return arrayZ[i-1].toCoord +
           ((arrayZ[i].toCoord - arrayZ[i-1].toCoord) *
            (inputvalue - arrayZ[i-1].fromCoord)) / (arrayZ[i].fromCoord - arrayZ[i-1].fromCoord);
  }
  else if (inputvalue >= arrayZ[i+1].fromCoord)
  {
    denom = arrayZ[i+1].fromCoord - arrayZ[i].fromCoord;
    return arrayZ[i].toCoord +
           ((arrayZ[i+1].toCoord - arrayZ[i].toCoord) *
            (inputvalue - arrayZ[i].fromCoord)) / denom;
  }
  else
  {
    denom = arrayZ[i+1].fromCoord - arrayZ[i].fromCoord;
    return arrayZ[i].toCoord +
           ((arrayZ[i+1].toCoord - arrayZ[i].toCoord) *
            (inputvalue - arrayZ[i].fromCoord) + (arrayZ[i+1].toCoord - arrayZ[i].toCoord) *
            (arrayZ[i+1].fromCoord - inputvalue)) / denom;
  }
}
