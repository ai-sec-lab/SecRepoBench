// Validate the argument count
    if (argc < 0) {
        mrb_value exc = mrb_exc_new_str(mrb, E_ARGUMENT_ERROR, mrb_format(mrb, "negative argument count: %d", argc));
        mrb_exc_set(mrb, exc);
        goto L_RAISE;
    }

    // Locate the method associated with the given symbol and class
    m = mrb_method_search_vm(mrb, &c, mid);
    if (MRB_METHOD_UNDEF_P(m)) {
        mrb_value args = mrb_ary_new_from_values(mrb, argc, argv);
        mrb_method_missing(mrb, mid, self, args);
    }

    // Check if the method is a C function or a Ruby method
    if (MRB_METHOD_CFUNC_P(m)) {
        ci->cci = CINFO_DIRECT;
        val = MRB_METHOD_CFUNC(m)(mrb, self);
    } else {
        ci->cci = CINFO_SKIP;
        val = mrb_run(mrb, MRB_METHOD_PROC(m), self);
    }
