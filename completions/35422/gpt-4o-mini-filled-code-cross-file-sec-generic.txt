// Allocate memory for the concatenated result
  string_info->datum = (unsigned char *) ResizeQuantumMemory(string_info->datum, length + source->length + MagickPathExtent, sizeof(*string_info->datum));
  if (string_info->datum == (unsigned char *) NULL)
    ThrowFatalException(ResourceLimitFatalError, "MemoryAllocationFailed");

  // If the current data is NULL, initialize it
  if (string_info->length == 0)
    {
      string_info->length = source->length;
      (void) memcpy(string_info->datum, source->datum, source->length);
    }
  else
    {
      // Copy the source data to the end of the destination's data
      (void) memcpy(string_info->datum + length, source->datum, source->length);
      string_info->length += source->length;
    }

  // Ensure the new length is set correctly
  SetStringInfoLength(string_info, string_info->length);
