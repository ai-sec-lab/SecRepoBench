// Calculate the number of bytes required for the video data
    int required_video_size = aligned_width * avctx->height * c->bpp / 8;

    // Check if the video size is sufficient
    if (c->video_size < required_video_size) {
        av_log(avctx, AV_LOG_ERROR, "Insufficient video size: %d < %d\n", c->video_size, required_video_size);
        return AVERROR_INVALIDDATA;
    }

    // Determine the pixel format based on bits per pixel and format
    switch (c->bpp) {
    case 8:
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
        break;
    case 24:
        avctx->pix_fmt = AV_PIX_FMT_BGR24;
        break;
    default:
        avpriv_request_sample(avctx, "Unsupported bits per pixel: %d", c->bpp);
        return AVERROR_PATCHWELCOME;
    }
