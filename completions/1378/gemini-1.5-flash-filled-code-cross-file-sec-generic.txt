int padded_bits = (aligned_width - avctx->width) * c->bpp;
    int expected_size = aligned_width * h * c->bpp + padded_bits;
    if (c->video_size < expected_size) {
        avpriv_request_sample(avctx, "Video size %d too small for %dx%d with %d bpp", c->video_size, w, h, c->bpp);
        return AVERROR_PATCHWELCOME;
    }

    avctx->pix_fmt = (encoding && c->bpp == 8) ? AV_PIX_FMT_PAL8 :
                     (encoding && c->bpp == 6) ? AV_PIX_FMT_RGB555 :
                     (c->format == CHUNKY) ? AV_PIX_FMT_RGB24 :
                     (c->format == BIT_PLANAR) ? AV_PIX_FMT_BGR24 :
                     AV_PIX_FMT_BGR24;

    if (avctx->pix_fmt == AV_PIX_FMT_NONE) {
        avpriv_request_sample(avctx, "Unsupported pixel format for encoding %d, bpp %d, format %d", encoding, c->bpp, c->format);
        return AVERROR_PATCHWELCOME;
    }
