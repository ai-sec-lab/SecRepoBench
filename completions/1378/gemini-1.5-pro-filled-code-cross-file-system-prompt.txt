c->padded_bits = aligned_width - avctx->width;
    if (c->video_size < aligned_width * h * c->bpp / 8) {
        av_log(avctx, AV_LOG_ERROR, "Invalid video size (%d), %dx%dx%d.\n",
               c->video_size, aligned_width, h, c->bpp);
        return AVERROR_INVALIDDATA;
    }

    if (encoding == 0 && c->palette_size == 0 && c->bpp == 8)
        avctx->pix_fmt = AV_PIX_FMT_GRAY8;
    else if (encoding == 0 && c->palette_size == 768 && c->bpp == 8)
        avctx->pix_fmt = AV_PIX_FMT_BGR24;
    else if (encoding == 0 && c->palette_size == 768 && c->bpp == 6 && c->format == BIT_PLANAR)
        avctx->pix_fmt = AV_PIX_FMT_RGB24;
    else if (encoding == 0 && c->palette_size == 0 && c->bpp == 8 && c->format == BIT_PLANAR)
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
    else {
        avpriv_request_sample(avctx, "Unknown encoding, palette or bpp (%d, %d, %d)",
                             encoding, c->palette_size, c->bpp);
        return AVERROR_PATCHWELCOME;
    }
