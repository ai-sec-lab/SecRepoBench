// Calculate the number of padded bits needed for video decoding
    int padded_bits = aligned_width * avctx->height * c->bpp;
    if (c->video_size < padded_bits / 8) {
        av_log(avctx, AV_LOG_ERROR, "Insufficient video size for decoding\n");
        return AVERROR_INVALIDDATA;
    }

    // Determine the appropriate pixel format
    switch (c->bpp) {
    case 8:
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
        break;
    case 24:
        avctx->pix_fmt = AV_PIX_FMT_BGR24;
        break;
    default:
        avpriv_request_sample(avctx, "Unsupported bits per pixel: %d", c->bpp);
        return AVERROR_PATCHWELCOME;
    }
