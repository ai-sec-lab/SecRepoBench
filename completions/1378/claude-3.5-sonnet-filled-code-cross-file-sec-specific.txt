int padded_bits = aligned_width - avctx->width;
int64_t required_size = (int64_t)aligned_width * avctx->height * c->bpp / 8;

if (required_size > c->video_size || required_size < 0)
    return AVERROR_INVALIDDATA;

if (c->palette_size && c->bpp <= 8) {
    avctx->pix_fmt = AV_PIX_FMT_PAL8;
} else if (encoding) {
    if (c->bpp == 6 || c->bpp == 8) {
        avctx->pix_fmt = AV_PIX_FMT_BGR24;
    } else {
        avpriv_request_sample(avctx, "Encoding %d with %d bpp", encoding, c->bpp);
        return AVERROR_PATCHWELCOME;
    }
} else if (c->bpp == 24) {
    avctx->pix_fmt = AV_PIX_FMT_RGB24;
} else {
    avpriv_request_sample(avctx, "Bit depth %d", c->bpp);
    return AVERROR_PATCHWELCOME;
}
