int padded_bits = aligned_width * c->bpp - avctx->width * c->bpp;
int required_size = (aligned_width * c->bpp * avctx->height + 7) >> 3;

if (c->video_size < required_size)
    return AVERROR_INVALIDDATA;

if (encoding == 0 && c->palette_size && c->bpp <= 8) {
    avctx->pix_fmt = AV_PIX_FMT_PAL8;
} else if (encoding == 0 && c->bpp == 24) {
    avctx->pix_fmt = AV_PIX_FMT_RGB24;
} else if (encoding == 1 && c->bpp == 6) {
    avctx->pix_fmt = AV_PIX_FMT_RGB24;
} else if (encoding == 1 && c->bpp == 8) {
    avctx->pix_fmt = AV_PIX_FMT_RGB24;
} else {
    avpriv_request_sample(avctx, "Encoding %d with %d bpp", encoding, c->bpp);
    return AVERROR_PATCHWELCOME;
}
