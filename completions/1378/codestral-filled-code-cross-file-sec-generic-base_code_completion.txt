int padded_bits = (aligned_width - avctx->width) * 8;
if (c->video_size < (h * aligned_width * c->bpp + 7) / 8) {
    avpriv_request_sample(avctx, "Insufficient video size for given parameters");
    return AVERROR_PATCHWELCOME;
}

if (c->format == BIT_LINE) {
    if (c->bpp == 8) {
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
    } else if (c->bpp == 24) {
        avctx->pix_fmt = AV_PIX_FMT_BGR24;
    } else if (c->bpp == 32) {
        avctx->pix_fmt = AV_PIX_FMT_RGB24;
    } else {
        avpriv_request_sample(avctx, "Unsupported bits per pixel for BIT_LINE format");
        return AVERROR_PATCHWELCOME;
    }
} else if (c->format == BIT_PLANAR || c->format == CHUNKY) {
    if (c->bpp != 8) {
        avpriv_request_sample(avctx, "Unsupported bits per pixel for BIT_PLANAR or CHUNKY format");
        return AVERROR_PATCHWELCOME;
    }
}
