int padded_bits = aligned_width - avctx->width;
int required_size = (aligned_width * c->bpp * avctx->height + 7) >> 3;

if (c->video_size < required_size)
    return AVERROR_INVALIDDATA;

if (encoding == 0 && (c->palette_size != 0 || c->bpp != 8)) {
    avpriv_request_sample(avctx, "Encoding 0 with palette size %d and bpp %d", c->palette_size, c->bpp);
    return AVERROR_PATCHWELCOME;
}

switch (c->bpp) {
    case 1:
    case 2:
    case 4:
    case 8:
        avctx->pix_fmt = AV_PIX_FMT_PAL8;
        break;
    case 24:
        avctx->pix_fmt = AV_PIX_FMT_RGB24;
        break;
    default:
        avpriv_request_sample(avctx, "Bits per pixel %d", c->bpp);
        return AVERROR_PATCHWELCOME;
}

if (c->palette_size) {
    const uint8_t *pal = c->palette;
    int pal_size = FFMIN(c->palette_size, 256 * 3);
    int i;

    for (i = 0; i < pal_size / 3; i++) {
        c->pal[i] = 0xFF000000 | AV_RB24(pal);
        pal += 3;
    }
}
