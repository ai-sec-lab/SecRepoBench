            }
            p->format = AV_PIX_FMT_PAL8;

            if (c->video_size != aligned_width * h * 3) {
                avpriv_request_sample(avctx, "Video size mismatch");
                return AVERROR_PATCHWELCOME;
            }
            p->format = AV_PIX_FMT_BGR24;
        } else {
            avpriv_request_sample(avctx, "Unsupported bits per pixel");
            return AVERROR_PATCHWELCOME;
        }
    } else {

                return AVERROR_PATCHWELCOME;
            }
            p->format = AV_PIX_FMT_PAL8;
        } else {
            avpriv_request_sample(avctx, "Unsupported bits per pixel");
            return AVERROR_PATCHWELCOME;
        }
    }

        } else {
            avpriv_request_sample(avctx, "Unsupported bits per pixel");
