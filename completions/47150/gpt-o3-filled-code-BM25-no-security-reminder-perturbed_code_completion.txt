/*  NULL or empty: not a full path */
    if (!file_path || !*file_path) {
        return FALSE;
    }

    /*  Unix / Posix or UNC style (or Windows with leading '\')
        First character is the directory join character. */
    if (file_path[0] == joinchar) {
        /*  Examples: "/usr/bin", "\\server\share"  */
        return TRUE;
    }

    /*  Windows drive-letter style:  <alpha ':' ...>  */
    if ( ((file_path[0] >= 'A' && file_path[0] <= 'Z') ||
          (file_path[0] >= 'a' && file_path[0] <= 'z')) &&
         file_path[1] == ':' ) {
        /*  Examples: "C:\foo\bar", "D:/baz" */
        return TRUE;
    }
