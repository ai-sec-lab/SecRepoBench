if (codeLength < (uint64_t) SHORT_ZEROCODE_RUN)
        {
            /* literal code length (may be zero) */
            if (codeLength != 0)
            {
                /* sanity-check */
                if (codeLength > (uint64_t) MAX_CODE_LEN)
                {
                    if (pctxt)
                        pctxt->print_error (
                            pctxt,
                            EXR_ERR_CORRUPT_CHUNK,
                            "Error decoding Huffman table (Invalid code length).");
                    return EXR_ERR_CORRUPT_CHUNK;
                }

                /* track min / max lengths */
                if (codeLength < (uint64_t) fhd->_minCodeLength)
                    fhd->_minCodeLength = (unsigned char) codeLength;
                if (codeLength > (uint64_t) fhd->_maxCodeLength)
                    fhd->_maxCodeLength = (unsigned char) codeLength;

                /* update count of codes with this length */
                codeCount[codeLength]++;
            }
            /* if codeLength == 0 nothing else to do for this symbol */
        }
        else if (codeLength == (uint64_t) LONG_ZEROCODE_RUN)
        {
            /* long run of zero code lengths */
            uint64_t run =
                fasthuf_read_bits (8, &currBits, &currBitCount, &currByte) +
                SHORTEST_LONG_RUN;
            symbol += run - 1; /* -1 because the for-loop will ++ symbol */
        }
        else /* short run of zero code lengths (59-62) */
        {
            uint64_t run = codeLength - SHORT_ZEROCODE_RUN + 2;
            symbol += run - 1; /* -1 to compensate for loop increment */
        }
