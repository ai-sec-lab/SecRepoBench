  if ((count > 0) && (text != (char *) NULL))
    {
      char
        *value;

      value=AcquireString(text);
      if (value != (char *) NULL)
        {
          value[count]='\0';
          (void) SetImageProperty(image,"tiff:image-id",value,exception);
          value=DestroyString(value);
        }
