if (TIFFGetField(tiff,TIFFTAG_ARTIST, &text) != 0)
    (void) SetImageProperty(image,"artist",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_COPYRIGHT, &text) != 0)
    (void) SetImageProperty(image,"copyright",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_DOCUMENTNAME, &text) != 0)
    (void) SetImageProperty(image,"document",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_IMAGEDESCRIPTION, &text) != 0)
    (void) SetImageProperty(image,"comment",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_HOSTCOMPUTER, &text) != 0)
    (void) SetImageProperty(image,"hostcomputer",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_MAKE, &text) != 0)
    (void) SetImageProperty(image,"make",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_MODEL, &text) != 0)
    (void) SetImageProperty(image,"model",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_PAGENAME, &text) != 0)
    (void) SetImageProperty(image,"label",text,exception);
  if (TIFFGetField(tiff,TIFFTAG_SOFTWARE, &text) != 0)
    (void) SetImageProperty(image,"software",text,exception);
  if (TIFFGetField(tiff,33423,&count,&type,&text) == 1)
    {
      if ((type == TIFF_ASCII) && (count != 0))
        (void) SetImageProperty(image,"kodak-33423",text,exception);
    }
  if (TIFFGetField(tiff,36867,&count,&type,&text) == 1)
    {
      if ((type == TIFF_ASCII) && (count != 0))
        (void) SetImageProperty(image,"kodak-36867",text,exception);
    }
  if (TIFFGetField(tiff,TIFFTAG_OPIIMAGEID,&count,&type,&text) == 1)
    {
      if (count > 1)  /* for some reason count is 1 and text is NULL sometimes */
        {
          if (type == TIFF_ASCII)
            (void) SetImageProperty(image,"OPI",text,exception);
          else if (type == TIFF_BYTE)
            {
              (void) memset(message, 0, MagickPathExtent);
              (void) memcpy(message, text, MagickMin(count, MagickPathExtent - 1));
              (void) SetImageProperty(image, "OPI", message, exception);
            }
        }
    }

  if (TIFFGetField(tiff,TIFFTAG_SUBFILETYPE,&type) != 0)
    {
      (void) FormatLocaleString(message,MagickPathExtent,"%u",type);
      (void) SetImageProperty(image,"tiff:subfiletype",message,exception);
      switch (type)
      {
        case FILETYPE_REDUCEDIMAGE:
        {
          (void) SetImageProperty(image,"tiff:subfiletype","REDUCEDIMAGE",
            exception);
          break;
        }
        case FILETYPE_PAGE:
        {
          (void) SetImageProperty(image,"tiff:subfiletype","PAGE",exception);
          break;
        }
        case FILETYPE_MASK:
        {
          (void) SetImageProperty(image,"tiff:subfiletype","MASK",exception);
          break;
        }
        default:
          break;
      }
    }
