// Retrieve and set image properties from TIFF tags.
    TIFFGetField(tiff, TIFFTAG_ARTIST, &text);
    if (text != NULL)
        SetImageProperty(image, "artist", text, exception);

    TIFFGetField(tiff, TIFFTAG_COPYRIGHT, &text);
    if (text != NULL)
        SetImageProperty(image, "copyright", text, exception);

    TIFFGetField(tiff, TIFFTAG_DATETIME, &text);
    if (text != NULL)
        SetImageProperty(image, "timestamp", text, exception);

    TIFFGetField(tiff, TIFFTAG_DOCUMENTNAME, &text);
    if (text != NULL)
        SetImageProperty(image, "document", text, exception);

    TIFFGetField(tiff, TIFFTAG_HOSTCOMPUTER, &text);
    if (text != NULL)
        SetImageProperty(image, "hostcomputer", text, exception);

    TIFFGetField(tiff, TIFFTAG_MAKE, &text);
    if (text != NULL)
        SetImageProperty(image, "make", text, exception);

    TIFFGetField(tiff, TIFFTAG_MODEL, &text);
    if (text != NULL)
        SetImageProperty(image, "model", text, exception);

    TIFFGetField(tiff, TIFFTAG_SOFTWARE, &text);
    if (text != NULL)
        SetImageProperty(image, "software", text, exception);

    // Handle specific TIFF tags
    TIFFGetField(tiff, TIFFTAG_SUBFILETYPE, &type);
    if (type == FILETYPE_REDUCEDIMAGE || type == FILETYPE_PAGE || type == FILETYPE_MASK)
        SetImageProperty(image, "tiff:subfiletype", (type == FILETYPE_REDUCEDIMAGE) ? "REDUCEDIMAGE" :
            (type == FILETYPE_PAGE) ? "PAGE" : "MASK", exception);
