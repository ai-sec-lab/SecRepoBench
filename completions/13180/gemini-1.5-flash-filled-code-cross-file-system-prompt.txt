for (count=TIFFGetField(tiff,TIFFTAG_ARTIST,&text); count != 0; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(image,"tiff:artist",text,exception);
    break;
  }
  for (count=TIFFGetField(tiff,TIFFTAG_COPYRIGHT,&text); count != 0; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(image,"tiff:copyright",text,exception);
    break;
  }
  for (count=TIFFGetField(tiff,TIFFTAG_DOCUMENTNAME,&text); count != 0; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(image,"tiff:document",text,exception);
    break;
  }
  for (count=TIFFGetField(tiff,TIFFTAG_IMAGEDESCRIPTION,&text); count != 0; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(image,"comment",text,exception);
    break;
  }
  for (count=TIFFGetField(tiff,TIFFTAG_MAKE,&text); count != 0; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(image,"tiff:make",text,exception);
    break;
  }
  for (count=TIFFGetField(tiff,TIFFTAG_MODEL,&text); count != 0; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(image,"tiff:model",text,exception);
    break;
  }
  for (count=TIFFGetField(tiff,TIFFTAG_PAGENAME,&text); count != 0; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(image,"label",text,exception);
    break;
  }
  for (count=TIFFGetField(tiff,TIFFTAG_SOFTWARE,&text); count != 0; )
  {
    if (text != (char *) NULL)
      (void) SetImageProperty(image,"tiff:software",text,exception);
    break;
  }
  for (count=TIFFGetField(tiff,TIFFTAG_OPIIMAGEID,&text); count != 0; )
  {
    if (text != (char *) NULL)
      {
        length=strlen(text);
        if (length < MagickPathExtent)
          (void) CopyMagickString(message,text,MagickPathExtent);
        else
          (void) CopyMagickString(message,text,MagickPathExtent-1);
        (void) SetImageProperty(image,"tiff:OPI-imageid",message,exception);
      }
    break;
  }
  for (count=TIFFGetField(tiff,TIFFTAG_SUBFILETYPE,&type); count != 0; )
  {
    if (type == FILETYPE_REDUCEDIMAGE)
      (void) SetImageProperty(image,"tiff:subfiletype","REDUCEDIMAGE",exception);
    else
      if (type == FILETYPE_PAGE)
        (void) SetImageProperty(image,"tiff:subfiletype","PAGE",exception);
      else
        if (type == FILETYPE_MASK)
          (void) SetImageProperty(image,"tiff:subfiletype","MASK",exception);
    break;
  }
