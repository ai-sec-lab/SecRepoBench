        av_log(avctx, AV_LOG_ERROR, "Too many reference location offsets: %d\n",
               pps->num_ref_loc_offsets);
        return AVERROR_INVALIDDATA;
    }
    for (int i = 0; i < pps->num_ref_loc_offsets; i++) {
        pps->ref_loc_offset[i].ref_layer_id = get_bits(gb, 6);
        if (pps->ref_loc_offset[i].scaled_ref_layer_offset_present_flag = get_bits1(gb)) {
            pps->ref_loc_offset[i].scaled_ref_layer_left_offset  = get_se_golomb(gb);
            pps->ref_loc_offset[i].scaled_ref_layer_top_offset   = get_se_golomb(gb);
            pps->ref_loc_offset[i].scaled_ref_layer_right_offset = get_se_golomb(gb);
            pps->ref_loc_offset[i].scaled_ref_layer_bottom_offset= get_se_golomb(gb);
        }
        if (pps->ref_loc_offset[i].ref_region_offset_present_flag = get_bits1(gb)) {
            pps->ref_loc_offset[i].ref_region_left_offset  = get_se_golomb(gb);
            pps->ref_loc_offset[i].ref_region_top_offset   = get_se_golomb(gb);
            pps->ref_loc_offset[i].ref_region_right_offset = get_se_golomb(gb);
            pps->ref_loc_offset[i].ref_region_bottom_offset= get_se_golomb(gb);
        }
        if (pps->ref_loc_offset[i].resample_phase_set_present_flag = get_bits1(gb)) {
            pps->ref_loc_offset[i].phase_hor_luma = get_bits(gb, 2);
            pps->ref_loc_offset[i].phase_ver_luma = get_bits(gb, 2);
            pps->ref_loc_offset[i].phase_hor_chroma = get_bits(gb, 2);
            pps->ref_loc_offset[i].phase_ver_chroma = get_bits(gb, 2);
        }
