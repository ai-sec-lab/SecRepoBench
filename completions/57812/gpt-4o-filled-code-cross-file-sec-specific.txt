    }

        pps->resample_phase_set_present_flag[i] = get_bits1(gb);

        if (pps->scaled_ref_layer_offset_present_flag[i]) {
            pps->scaled_ref_layer_left_offset[i] = get_se_golomb_long(gb);

            pps->scaled_ref_layer_bottom_offset[i] = get_se_golomb_long(gb);
        }

        if (pps->ref_region_offset_present_flag[i]) {
            pps->ref_region_left_offset[i] = get_se_golomb_long(gb);
            pps->ref_region_top_offset[i] = get_se_golomb_long(gb);
            pps->ref_region_right_offset[i] = get_se_golomb_long(gb);

        }

            pps->resample_phase_set_id[i] = get_ue_golomb_long(gb);
