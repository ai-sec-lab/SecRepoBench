if (pps->num_ref_loc_offsets > 0) {
    pps->ref_loc_offset = av_calloc(pps->num_ref_loc_offsets, sizeof(*pps->ref_loc_offset));
    if (!pps->ref_loc_offset)
        return AVERROR(ENOMEM);

    for (int i = 0; i < pps->num_ref_loc_offsets; i++) {
        pps->ref_loc_offset[i].ref_loc_offset_layer_id = get_bits(gb, 6);
        pps->ref_loc_offset[i].scaled_ref_layer_offset_present_flag = get_bits1(gb);
        if (pps->ref_loc_offset[i].scaled_ref_layer_offset_present_flag) {
            pps->ref_loc_offset[i].scaled_ref_layer_left_offset = get_se_golomb_long(gb);
            pps->ref_loc_offset[i].scaled_ref_layer_top_offset = get_se_golomb_long(gb);
            pps->ref_loc_offset[i].scaled_ref_layer_right_offset = get_se_golomb_long(gb);
            pps->ref_loc_offset[i].scaled_ref_layer_bottom_offset = get_se_golomb_long(gb);
        }
        pps->ref_loc_offset[i].ref_region_offset_present_flag = get_bits1(gb);
        if (pps->ref_loc_offset[i].ref_region_offset_present_flag) {
            pps->ref_loc_offset[i].ref_region_left_offset = get_se_golomb_long(gb);
            pps->ref_loc_offset[i].ref_region_top_offset = get_se_golomb_long(gb);
            pps->ref_loc_offset[i].ref_region_right_offset = get_se_golomb_long(gb);
            pps->ref_loc_offset[i].ref_region_bottom_offset = get_se_golomb_long(gb);
        }
        pps->ref_loc_offset[i].resample_phase_set_present_flag = get_bits1(gb);
        if (pps->ref_loc_offset[i].resample_phase_set_present_flag) {
            pps->ref_loc_offset[i].phase_hor_luma = get_ue_golomb_long(gb);
            pps->ref_loc_offset[i].phase_ver_luma = get_ue_golomb_long(gb);
            pps->ref_loc_offset[i].phase_hor_chroma_plus8 = get_ue_golomb_long(gb);
            pps->ref_loc_offset[i].phase_ver_chroma_plus8 = get_ue_golomb_long(gb);
        }
    }
