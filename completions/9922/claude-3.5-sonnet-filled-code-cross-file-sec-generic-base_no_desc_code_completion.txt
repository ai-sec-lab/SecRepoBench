if (uc >= ue)
    goto out;
switch (*uc++) {
case '"':
case '\\':
case '/':
case 'b':
case 'f':
case 'n':
case 'r':
case 't':
    break;
case 'u':
    if (uc + 4 > ue)
        goto out;
    for (i = 0; i < 4; i++) {
        if (!isxdigit((unsigned char)uc[i]))
            goto out;
    }
    uc += 4;
    break;
default:
    goto out;
}
