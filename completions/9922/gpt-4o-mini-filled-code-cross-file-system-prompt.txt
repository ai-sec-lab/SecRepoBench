case '\"':
			case '\\':
			case '/':
			case 'b':
			case 'f':
			case 'n':
			case 'r':
			case 't':
				continue; // Valid escape sequences
			case 'u': {
				// Handle Unicode escape sequence
				if (uc + 4 > ue) {
					goto out; // Not enough data for Unicode
				}
				for (i = 0; i < 4; i++) {
					if (!json_isxdigit(*uc++)) {
						goto out; // Invalid hex digit
					}
				}
				continue;
			}
			default:
				goto out; // Invalid escape sequence
