			switch (*uc) {
			case '\\':
			case '"':
			case '/':
				uc++;
				break;
			case 'b':
				uc++;
				break;
			case 'f':
				uc++;
				break;
			case 'n':
				uc++;
				break;
			case 'r':
				uc++;
				break;
			case 't':
				uc++;
				break;
			case 'u':
				for (i = 0; i < 4 && uc < ue; i++) {
					if (!isxdigit(*++uc))
						goto out;
				}
				if (i != 4)
					goto out;

				break;
			default:
				goto out;
