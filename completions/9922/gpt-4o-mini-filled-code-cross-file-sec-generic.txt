// Handle escape sequences
			if (uc >= ue) {
				DPRINTF("Unexpected end of string: ", uc, *ucp);
				goto out;
			}
			switch (*uc++) {
			case '\\':
			case '\"':
			case '/':
			case 'b':
			case 'f':
			case 'n':
			case 'r':
			case 't':
				// Valid escape sequences, continue parsing
				break;
			case 'u':
				// Handle Unicode escape sequence
				if (uc + 4 > ue) {
					DPRINTF("Invalid Unicode escape sequence: ", uc, *ucp);
					goto out;
				}
				for (i = 0; i < 4; i++) {
					if (!json_isxdigit(*uc++)) {
						DPRINTF("Invalid hex digit in Unicode escape: ", uc, *ucp);
						goto out;
					}
				}
				break;
			default:
				DPRINTF("Invalid escape sequence: ", uc, *ucp);
				goto out;
			}
