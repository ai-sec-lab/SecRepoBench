case '\\':
    uc++;
    if (uc >= ue) {
        DPRINTF("Unexpected end of data", uc, *ucp);
        return 0;
    }
    switch (*uc++) {
    case '\\':
    case '"':
    case '/':
    case 'b':
    case 'f':
    case 'n':
    case 'r':
    case 't':
        break;
    case 'u':
        for (i = 0; i < 4 && uc < ue; i++) {
            if (!isxdigit(*uc)) {
                DPRINTF("Invalid escape sequence", uc, *ucp);
                return 0;
            }
            uc++;
        }
        break;
    default:
        DPRINTF("Invalid escape sequence", uc, *ucp);
        return 0;
    }
    break;
