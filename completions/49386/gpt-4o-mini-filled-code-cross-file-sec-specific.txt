for (i = 0; i < TagCount; i++) {
        cmsTagEntry tag;
        cmsUInt32Number tagSize;

        // Read the tag entry
        if (io->Read(io, &tag, sizeof(cmsTagEntry), 1) != 1) {
            cmsSignalError(Icc->ContextID, cmsERROR_READ, "Failed to read tag entry");
            return FALSE;
        }

        // Adjust endianness for the tag entry
        tag.sig = _cmsAdjustEndianess32(tag.sig);
        tag.offset = _cmsAdjustEndianess32(tag.offset);
        tag.size = _cmsAdjustEndianess32(tag.size);

        // Validate tag size
        if (tag.size > Icc->IOhandler->ReportedSize - tag.offset) {
            cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Tag size exceeds available data");
            return FALSE;
        }

        // Store the tag information
        Icc->TagOffsets[Icc->TagCount] = tag.offset;
        Icc->TagSizes[Icc->TagCount] = tag.size;
        Icc->TagSignatures[Icc->TagCount] = tag.sig;
        Icc->TagCount++;

        // Check for maximum tag count
        if (Icc->TagCount >= MAX_TABLE_TAG) {
            cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Exceeded maximum tag count");
            return FALSE;
        }
    }

    return TRUE;
