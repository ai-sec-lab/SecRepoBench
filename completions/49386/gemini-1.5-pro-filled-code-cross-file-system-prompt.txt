for (i = 0; i < TagCount; i++) {

        if (!io -> Read(io, &Tag, sizeof(cmsTagEntry), 1)) return FALSE;

        // Adjust endianness
        Tag.sig    = _cmsAdjustEndianess32(Tag.sig);
        Tag.offset = _cmsAdjustEndianess32(Tag.offset);
        Tag.size   = _cmsAdjustEndianess32(Tag.size);

        // Basic sanity checks
        if (Tag.offset + Tag.size > HeaderSize || Tag.offset < sizeof(cmsICCHeader) || Tag.size <= 0) {

            char tmp[5];
            _cmsTagSignature2String(tmp, Tag.sig);
            cmsSignalError(Icc ->ContextID, cmsERROR_BAD_TAG, "Tag '%s' has invalid offset/size", tmp);
            return FALSE;
        }

        Icc -> TagSigs[Icc -> TagCount] = Tag.sig;
        Icc -> TagOffsets[Icc -> TagCount] = Tag.offset;
        Icc -> TagSizes[Icc -> TagCount] = Tag.size;

        // Check for links between tags.
        for (j = 0; j < Icc->TagCount; j++) {
            if (Icc->TagOffsets[j] == Tag.offset &&
                Icc->TagSizes[j] == Tag.size &&
                Icc->TagSigs[j] != Tag.sig) {
                Icc->TagLinked[Icc->TagCount] = Icc->TagSigs[j];
                break;
            }
        }

        Icc -> TagCount++;
    }

    return TRUE;
