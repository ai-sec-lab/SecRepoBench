for (i = 0; i < TagCount; i++) {
        cmsTagEntry TagEntry;

        // Read the tag entry
        if (io->Read(io, &TagEntry, sizeof(cmsTagEntry), 1) != 1) {
            cmsSignalError(Icc->ContextID, cmsERROR_READ, "Failed to read tag entry");
            return FALSE;
        }

        // Validate tag entry
        TagEntry.signature = _cmsAdjustEndianess32(TagEntry.signature);
        TagEntry.offset = _cmsAdjustEndianess32(TagEntry.offset);
        TagEntry.size = _cmsAdjustEndianess32(TagEntry.size);

        // Ensure the tag offset and size are within the bounds of the profile
        if (TagEntry.offset + TagEntry.size > HeaderSize) {
            cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Tag entry out of bounds");
            return FALSE;
        }

        // Store the tag information in the profile structure
        Icc->TagSignatures[Icc->TagCount] = TagEntry.signature;
        Icc->TagOffsets[Icc->TagCount] = TagEntry.offset;
        Icc->TagSizes[Icc->TagCount] = TagEntry.size;

        // Check for linked tags
        if (TagEntry.linkedTag != 0) {
            Icc->TagLinked[Icc->TagCount] = _cmsAdjustEndianess32(TagEntry.linkedTag);
        } else {
            Icc->TagLinked[Icc->TagCount] = 0;
        }

        Icc->TagCount++;

        // Ensure we do not exceed the maximum number of tags
        if (Icc->TagCount >= MAX_TABLE_TAG) {
            cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Exceeded maximum number of tags");
            return FALSE;
        }
    }

    // Set links between tags if applicable
    if (!SetLinks(Icc)) {
        cmsSignalError(Icc->ContextID, cmsERROR_LINK, "Failed to set tag links");
        return FALSE;
    }

    return TRUE;
