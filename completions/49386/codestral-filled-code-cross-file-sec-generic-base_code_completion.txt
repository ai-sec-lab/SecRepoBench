for (i = 0; i < TagCount; i++) {
    if (io->Read(io, &Tag, sizeof(cmsTagEntry), 1) != 1) {
        return FALSE;
    }

    cmsTagSignature sig = (cmsTagSignature)_cmsAdjustEndianess32(Tag.sig);
    cmsUInt32Number offset = _cmsAdjustEndianess32(Tag.offset);
    cmsUInt32Number size = _cmsAdjustEndianess32(Tag.size);

    if (offset + size > HeaderSize) {
        cmsSignalError(Icc->ContextID, cmsERROR_RANGE, "Tag out of bounds");
        return FALSE;
    }

    Icc->TagSignatures[Icc->TagCount] = sig;
    Icc->TagOffsets[Icc->TagCount] = offset;
    Icc->TagSizes[Icc->TagCount] = size;

    // Check for linked tags
    if (i < TagCount - 1) {
        cmsTagEntry nextTag;
        if (io->Read(io, &nextTag, sizeof(cmsTagEntry), 1) == 1) {
            cmsUInt32Number nextOffset = _cmsAdjustEndianess32(nextTag.offset);
            cmsUInt32Number nextSize = _cmsAdjustEndianess32(nextTag.size);

            if (offset == nextOffset && size == nextSize) {
                Icc->TagLinked[Icc->TagCount] = (cmsTagSignature)_cmsAdjustEndianess32(nextTag.sig);
            }
        }
        io->Seek(io, -sizeof(cmsTagEntry), SEEK_CUR); // Rewind to read the next tag in the next iteration
    }

    Icc->TagCount++;
}

// Set links for any tags that were found to be linked
SetLinks(Icc);

return TRUE;
