// Check if the element already exists in the map to prevent duplicates
        if (elements.find(str) != elements.end()) {
            ParseError("duplicate key found: " + str, n);
        }

        // Insert the newly created element into the elements map
        elements.emplace(str, element);

        // If the scope is at the top level, we can return early
        if (topLevel) {
            return;
        }

        // Advance to the next token
        n = parser.AdvanceToNextToken();
        if (n == nullptr) {
            // If we reach the end of the file unexpectedly, delete the element and trigger an error
            delete_Element(element);
            ParseError("unexpected end of file while parsing scope", n);
        }
