{   
                     _cmsStageMatrixData* Data1 = (_cmsStageMatrixData*) cmsStageData(Matrix1);
                     _cmsStageMatrixData* Data2 = (_cmsStageMatrixData*) cmsStageData(Matrix2);

                     // The input offset of the first matrix should be zero, otherwise we cannot optimize
                     if (Data1->Offset != NULL)
                            return FALSE;

                     // Multiply both 3x3 matrices (Matrix2 * Matrix1) to obtain a single resulting matrix
                     _cmsMAT3per(&res, (cmsMAT3*) Data2->Double, (cmsMAT3*) Data1->Double);

                     // Keep the output offset of the second matrix (may be NULL which means zero offset)
                     Offset = Data2->Offset;

                     // If the resulting matrix is identity AND the offset is NULL, we can drop the matrix altogether
                     isIdentityMatrix = FALSE;
                     if (_cmsMAT3isIdentity(&res) && Offset == NULL) {
                            isIdentityMatrix = TRUE;
                     }
