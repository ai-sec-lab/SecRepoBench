              Data1 = (_cmsStageMatrixData*)cmsStageData(Matrix1);
              Data2 = (_cmsStageMatrixData*)cmsStageData(Matrix2);

              // Ensure the input offset of the first matrix is zero
              if (Data1->Offset != NULL) return FALSE;

              // Multiply the two matrices to compute the resultant matrix
              _cmsMAT3per(&res, (cmsMAT3*)Data2->Double, (cmsMAT3*)Data1->Double);

              // Check if the resultant matrix, combined with the second matrix's offset, is an identity matrix
              IdentityMat = FALSE;
              if (_cmsMAT3isIdentity(&res) && Data2->Offset == NULL) {
                  // We can get rid of the full matrix
                  IdentityMat = TRUE;
