// Check if the size of the current box exceeds the size of the parent container box.
// Log a warning if the current box is larger than the container.
// If the current box is larger, handle it by setting the parent container size to zero.
// Otherwise, reduce the parent container's size by the size of the current box.
if (a->size > parent->size) {
    if (!skip_logs) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" is larger than its parent container\n", gf_4cc_to_str(a->type)));
    }
    parent->size = 0;
} else {
    parent->size -= a->size;
}
