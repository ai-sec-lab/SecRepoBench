if (a->size > parent->size) {
    if (!skip_logs) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" size %d exceeds parent box \"%s\" remaining size %d\n",
            gf_4cc_to_str(a->type), a->size, gf_4cc_to_str(parent->type), parent->size));
    }
    parent->size = 0;
} else {
    parent->size -= a->size;
}
