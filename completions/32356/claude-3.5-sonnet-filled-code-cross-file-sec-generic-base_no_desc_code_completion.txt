if (parent->size < a->size) {
    if (!skip_logs) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" size %d larger than remaining parent size %d\n", gf_4cc_to_str(a->type), a->size, parent->size));
    }
    gf_isom_box_del(a);
    return GF_ISOM_INVALID_FILE;
}
parent->size -= a->size;
