if (a->size > parent->size) {
    if (!skip_logs) {
        GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" exceeds parent container size\n", gf_4cc_to_str(a->type)));
    }
    parent->size = 0; // Set parent size to zero if current box exceeds it
} else {
    parent->size -= a->size; // Reduce parent size by the size of the current box
}
