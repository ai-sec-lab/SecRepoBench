if (a->size > parent->size) {
            // Log a warning about the size mismatch
            if (!skip_logs) {
                GF_LOG(GF_LOG_WARNING, GF_LOG_CONTAINER, ("[iso file] Box \"%s\" exceeds parent box size. Current box size: %u, Parent box size: %u\n", 
                    gf_4cc_to_str(a->type), a->size, parent->size));
            }
            // Handle the size mismatch by setting the parent size to zero
            parent->size = 0;
        } else {
            // Reduce the parent container's size by the size of the current box
            parent->size -= a->size;
        }
