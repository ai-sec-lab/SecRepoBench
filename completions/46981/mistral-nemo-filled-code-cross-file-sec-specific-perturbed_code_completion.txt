{
  if (mng_info->have_global_bkgd)
    {
      if (logging != MagickFalse)
        (void) LogMagickEvent(CoderEvent,GetMagickModule(),
          "  Have global background color");

      current_image->background_color=mng_info->mng_global_bkgd;
      (void) SetImageBackgroundColor(current_image,exception);
    }
  else
    {
      if (logging != MagickFalse)
        (void) LogMagickEvent(CoderEvent,GetMagickModule(),
          "  No global background color");

      current_image->background_color=TransparentColor;
      (void) SetImageBackgroundColor(current_image,exception);
    }
}
