/*
      Validate image returned by ReadOnePNGImage()/ReadOneJNGImage().
    */
    if (image == (Image *) NULL)
      {
        if (logging != MagickFalse)
          (void) LogMagickEvent(CoderEvent,GetMagickModule(),
            "  Decoder returned NULL image pointer.");
        return((Image *) NULL);
      }

    /*  Reject images that have no pixels.  */
    if ((image->columns == 0) || (image->rows == 0))
      {
        if (logging != MagickFalse)
          (void) LogMagickEvent(CoderEvent,GetMagickModule(),
            "  Image with zero width or height encountered "
            "(%.20g x %.20g) â€“ aborting.",
            (double) image->columns,(double) image->rows);
        (void) CloseBlob(image);
        return(DestroyImageList(image));
      }

    /*  Keep mng_data in sync with the image just read.  */
    mng_data->image=image;
