/*
    Allocate storage for the q-path and setup helper to expand the
    allocation as required while we add points to either path.
  */
  path_q=(PointInfo *) AcquireQuantumMemory(path_extent,sizeof(*path_q));
  /*
    Enlarge the path arrays as required.
  */
#define CheckPathExtent(offset)                                           \
{                                                                          \
  if (((ssize_t) (p+q+(offset))) >= (ssize_t) path_extent)                 \
    {                                                                      \
      while (((ssize_t) (p+q+(offset))) >= (ssize_t) path_extent)          \
        path_extent <<= 1;                                                 \
      path_p=(PointInfo *) RenewQuantumMemory(path_p,(size_t) path_extent, \
        sizeof(*path_p));                                                  \
      path_q=(PointInfo *) RenewQuantumMemory(path_q,(size_t) path_extent, \
        sizeof(*path_q));                                                  \
      if ((path_p == (PointInfo *) NULL) || (path_q == (PointInfo *) NULL))\
        {                                                                  \
          if (path_p != (PointInfo *) NULL)                                \
            path_p=(PointInfo *) RelinquishMagickMemory(path_p);           \
          if (path_q != (PointInfo *) NULL)                                \
            path_q=(PointInfo *) RelinquishMagickMemory(path_q);           \
          polygon_primitive=(PrimitiveInfo *)                              \
            RelinquishMagickMemory(polygon_primitive);                     \
          return((PrimitiveInfo *) NULL);                                  \
        }                                                                  \
    }                                                                      \
