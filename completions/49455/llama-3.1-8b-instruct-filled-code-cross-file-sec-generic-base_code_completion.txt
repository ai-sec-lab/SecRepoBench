for (rv = 1; b > 0; b--) {
    if (Dims[b - 1] <= 0) {
        cmsSignalError(self->ContextID, cmsERROR_INVALID_ARGUMENT, "Invalid dimension size");
        return 0;
    }
    if (rv > UINT_MAX / Dims[b - 1]) {
        cmsSignalError(self->ContextID, cmsERROR_OVERFLOW, "Overflow in dimension calculation");
        return 0;
    }
    rv *= Dims[b - 1];
}
