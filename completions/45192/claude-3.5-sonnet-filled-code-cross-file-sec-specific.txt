for (i = 0; i < srclen; i++) {
  unsigned char c = src[i];
  if (n > (UINT32_MAX >> 7)) {
    mrb_raise(mrb, E_RANGE_ERROR, "integer overflow in BER unpacking");
  }
  n = (n << 7) | (c & 0x7f);
  if (!(c & 0x80)) {
    i++;
    break;
  }
}
if (i > srclen) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "BER unpacking failed");
}
