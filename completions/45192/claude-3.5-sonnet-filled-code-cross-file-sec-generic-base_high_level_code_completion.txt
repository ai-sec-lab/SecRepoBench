for (i = 0; i < srclen; i++) {
  unsigned char byte = src[i];
  if (n > (MRB_INT_MAX >> 7)) {
    mrb_raise(mrb, E_RANGE_ERROR, "integer overflow in BER decoding");
  }
  n = (n << 7) | (byte & 0x7f);
  if (!(byte & 0x80)) {
    i++;
    break;
  }
}
if (i == srclen && (src[srclen-1] & 0x80)) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "incomplete BER encoding");
}
