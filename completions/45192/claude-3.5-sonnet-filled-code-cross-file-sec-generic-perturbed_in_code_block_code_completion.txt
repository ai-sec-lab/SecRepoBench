for (i = 0; i < srclen; i++) {
  unsigned char byte = sourcedata[i];
  if (n > (UINT32_MAX >> 7)) {
    mrb_raise(mrb, E_RANGE_ERROR, "integer overflow in BER unpacking");
  }
  n = (n << 7) | (byte & 0x7f);
  if (!(byte & 0x80)) {
    break;
  }
}
if (i == srclen) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "BER unpacking failed: incomplete integer");
}
