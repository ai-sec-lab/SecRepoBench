for (i = 0; i < srclen; i++) {
  unsigned char b = src[i];
  if (n > (MRB_INT_MAX >> 7)) {
    mrb_raise(mrb, E_RANGE_ERROR, "integer overflow in BER unpacking");
  }
  n = (n << 7) | (b & 0x7f);
  if (!(b & 0x80)) {
    i++;
    break;
  }
}
if (i == srclen && (src[srclen-1] & 0x80)) {
  mrb_raise(mrb, E_ARGUMENT_ERROR, "BER unpacking failed");
}
