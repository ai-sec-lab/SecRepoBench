    unsigned char c = src[i];

    n = (n << 7) | (c & 0x7f);
    if (!(c & 0x80)) {
      if (i + 1 < srclen) {
        if (flags & MRB_ASN1_OCTET_STRING_FLAG) {
          break;
        }
      }
      if ((mrb_int)n < MRB_INT_MIN || (mrb_int)n > MRB_INT_MAX) {
        mrb_raise(mrb, E_RANGE_ERROR, "BER integer too big");
      }
      break;
    }
