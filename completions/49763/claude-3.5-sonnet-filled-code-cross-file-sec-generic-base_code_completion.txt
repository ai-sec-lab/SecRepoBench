for (i = xd - yd; i >= 0; i--) {
  mp_limb qhat = (x.p[i+yd] << DIG_SIZE | x.p[i+yd-1]) / z;
  if (qhat > DIG_BASE - 1) qhat = DIG_BASE - 1;
  mp_dbl_limb borrow = 0;
  mp_dbl_limb carry = 0;
  for (size_t j = 0; j < yd; j++) {
    mp_dbl_limb prod = (mp_dbl_limb)qhat * y.p[j];
    mp_dbl_limb diff = (mp_dbl_limb)x.p[i+j] - (prod & DIG_BASE_MASK) - borrow;
    x.p[i+j] = diff & DIG_BASE_MASK;
    borrow = (prod >> DIG_SIZE) - (diff >> DIG_SIZE);
  }
  if (x.p[i+yd] < borrow) {
    qhat--;
    carry = 0;
    for (size_t j = 0; j < yd; j++) {
      mp_dbl_limb sum = (mp_dbl_limb)x.p[i+j] + y.p[j] + carry;
      x.p[i+j] = sum & DIG_BASE_MASK;
      carry = sum >> DIG_SIZE;
    }
    x.p[i+yd] += carry;
  }
  q.p[i] = qhat;
}
