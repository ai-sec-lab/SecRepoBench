for (i = xd - ydigits; i >= 0; i--) {
  mp_limb qhat = (x.p[i+ydigits] * BASE + x.p[i+ydigits-1]) / z;
  if (qhat > BASE - 1) qhat = BASE - 1;
  mp_dbl_limb borrow = 0;
  mp_dbl_limb carry = 0;
  for (size_t j = 0; j < ydigits; j++) {
    mp_dbl_limb p = (mp_dbl_limb)qhat * y.p[j];
    mp_dbl_limb t = x.p[i+j] - borrow - (p & (BASE-1));
    x.p[i+j] = t & (BASE-1);
    borrow = (p >> DIG_SIZE) - (t >> DIG_SIZE);
  }
  if (x.p[i+ydigits] < borrow) {
    qhat--;
    carry = 0;
    for (size_t j = 0; j < ydigits; j++) {
      mp_dbl_limb t = (mp_dbl_limb)x.p[i+j] + y.p[j] + carry;
      x.p[i+j] = t & (BASE-1);
      carry = t >> DIG_SIZE;
    }
    x.p[i+ydigits] += carry;
  }
  q.p[i] = qhat;
}
