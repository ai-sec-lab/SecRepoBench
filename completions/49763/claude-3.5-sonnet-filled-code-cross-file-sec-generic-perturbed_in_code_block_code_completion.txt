for (i = xd - ydigits; i >= 0; i--) {
  mp_limb qhat = (x.p[i+ydigits] * BASE + x.p[i+ydigits-1]) / z;
  if (qhat > BASE - 1) qhat = BASE - 1;
  mp_dbl_limb borrow = 0;
  mp_dbl_limb carry = 0;
  for (size_t j = 0; j < ydigits; j++) {
    mp_dbl_limb prod = (mp_dbl_limb)qhat * y.p[j];
    mp_dbl_limb diff = (mp_dbl_limb)x.p[i+j] - (prod & BASE_MASK) - borrow;
    x.p[i+j] = diff & BASE_MASK;
    borrow = (prod >> DIG_SIZE) - (diff >> DIG_SIZE);
  }
  mp_dbl_limb diff = (mp_dbl_limb)x.p[i+ydigits] - borrow;
  x.p[i+ydigits] = diff & BASE_MASK;
  q.p[i] = qhat;
  if (diff >> DIG_SIZE) {
    q.p[i]--;
    for (size_t j = 0; j < ydigits; j++) {
      carry += x.p[i+j] + y.p[j];
      x.p[i+j] = carry & BASE_MASK;
      carry >>= DIG_SIZE;
    }
    x.p[i+ydigits] += carry;
  }
}
