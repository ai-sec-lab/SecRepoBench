for (i = xd - yd; i >= 0; i--) {
  mp_limb qhat = (x.p[i+yd] << DIG_SIZE | x.p[i+yd-1]) / z;
  if (qhat > MP_LIMB_MAX) qhat = MP_LIMB_MAX;
  mp_dbl_limb borrow = 0;
  for (size_t j = 0; j < yd; j++) {
    mp_dbl_limb prod = (mp_dbl_limb)qhat * y.p[j];
    mp_dbl_limb diff = (mp_dbl_limb)x.p[i+j] - borrow - (prod & MP_LIMB_MAX);
    x.p[i+j] = diff & MP_LIMB_MAX;
    borrow = (prod >> DIG_SIZE) - (diff >> DIG_SIZE);
  }
  mp_dbl_limb diff = (mp_dbl_limb)x.p[i+yd] - borrow;
  x.p[i+yd] = diff & MP_LIMB_MAX;
  q.p[i] = qhat;
  if (diff >> DIG_SIZE) {
    q.p[i]--;
    mp_limb carry = 0;
    for (size_t j = 0; j < yd; j++) {
      mp_dbl_limb sum = (mp_dbl_limb)x.p[i+j] + y.p[j] + carry;
      x.p[i+j] = sum & MP_LIMB_MAX;
      carry = sum >> DIG_SIZE;
    }
    x.p[i+yd] += carry;
  }
}
