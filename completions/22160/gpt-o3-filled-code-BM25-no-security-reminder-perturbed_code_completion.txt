/*
    Skip to start of image data.
  */
  if (dpx.file.image_offset != 0)
    {
      if (offset < (MagickOffsetType) dpx.file.image_offset)
        {
          while (offset < (MagickOffsetType) dpx.file.image_offset)
          {
            if (ReadBlobByte(image) == EOF)
              break;
            offset++;
          }
        }
      else
        offset=SeekBlob(image,(MagickOffsetType) dpx.file.image_offset,SEEK_SET);
      if (offset != (MagickOffsetType) dpx.file.image_offset)
        ThrowReaderException(CorruptImageError,"UnexpectedEndOfFile");
    }

  /*
    If this is a ping request, we're done after parsing the header.
  */
  if (image_info->ping != MagickFalse)
    {
      (void) CloseBlob(image);
      return(GetFirstImageInList(image));
    }

  /*
    Initialize the image pixel cache now that the geometry is known.
  */
  operationsuccess=SetImageExtent(image,image->columns,image->rows,exception);
