int err = MATIO_E_NO_ERROR;

    /* Open the attribute that stores the field names */
    attr_id = H5Aopen_by_name(dset_id, ".", "MATLAB_fields", H5P_DEFAULT, H5P_DEFAULT);
    if ( attr_id < 0 ) {
        return MATIO_E_GENERIC_READ_ERROR;
    }

    /* Obtain the dataspace of the attribute                  */
    dataspace_id = H5Aget_space(attr_id);
    if ( dataspace_id < 0 ) {
        H5Aclose(attr_id);
        return MATIO_E_GENERIC_READ_ERROR;
    }

    /* Determine how many fields are stored in the attribute  */
    {
        hsize_t dims[1] = {0};
        if ( H5Sget_simple_extent_dims(dataspace_id, dims, NULL) < 0 ) {
            H5Sclose(dataspace_id);
            H5Aclose(attr_id);
            return MATIO_E_GENERIC_READ_ERROR;
        }
        *nfields = dims[0]; /* store number of fields for caller */
        if ( 0 == *nfields ) {
            H5Sclose(dataspace_id);
            H5Aclose(attr_id);
            return MATIO_E_GENERIC_READ_ERROR;
        }
    }

    /* Allocate buffer that will receive the variable-length strings */
    fieldnames_vl = (hvl_t *)malloc((size_t)(*nfields) * sizeof(*fieldnames_vl));
