if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds the maximum allowed: %d > %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->primitive_matrix[mat].output_channel = get_bits(gbp, 4);
        s->primitive_matrix[mat].frac_bits = get_bits(gbp, 4);
        if (s->primitive_matrix[mat].output_channel > s->max_matrix_channel) {
            av_log(m->avctx, AV_LOG_ERROR, "Matrix output channel exceeds the maximum matrix channel: %d > %d.\n",
                   s->primitive_matrix[mat].output_channel, s->max_matrix_channel);
            return AVERROR_INVALIDDATA;
        }
        if (s->primitive_matrix[mat].frac_bits > 15) {
            av_log(m->avctx, AV_LOG_ERROR, "Fractional bits exceed the maximum allowed: %d > 15.\n",
                   s->primitive_matrix[mat].frac_bits);
            return AVERROR_INVALIDDATA;
        }

        for (ch = 0; ch <= s->primitive_matrix[mat].output_channel; ch++) {
            s->primitive_matrix[mat].coeff[ch] = get_bits(gbp, 16);
        }

        if (s->noise_present) {
            s->primitive_matrix[mat].noise_shift = get_bits(gbp, 4);
        } else {
            s->primitive_matrix[mat].noise_shift = 0;
        }
    }
