if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds maximum allowed.\n");
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        // Read matrix output channel
        ch = get_bits(gbp, 4);
        if (ch >= MAX_CHANNELS) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix output channel: %u\n", ch);
            return AVERROR_INVALIDDATA;
        }
        s->primitive_matrix[mat].output_channel = ch;

        // Read number of fractional bits
        s->primitive_matrix[mat].fractional_bits = get_bits(gbp, 4);
        if (s->primitive_matrix[mat].fractional_bits > MAX_FRACTIONAL_BITS) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid number of fractional bits: %d\n", s->primitive_matrix[mat].fractional_bits);
            return AVERROR_INVALIDDATA;
        }

        // Read LSB bypass flag
        s->primitive_matrix[mat].lsb_bypass = get_bits1(gbp);

        // Read matrix coefficients
        for (unsigned int i = 0; i < MAX_MATRIX_CHANNEL; i++) {
            s->primitive_matrix[mat].coefficients[i] = get_sbits(gbp, s->primitive_matrix[mat].fractional_bits);
        }

        // Check for noise presence and read noise shift value
        if (s->primitive_matrix[mat].noise_present) {
            s->primitive_matrix[mat].noise_shift = get_bits(gbp, 4);
        } else {
            s->primitive_matrix[mat].noise_shift = 0;
        }
    }
