// Check if the number of primitive matrices exceeds the maximum allowed.
if (s->num_primitive_matrices > max_primitive_matrices) {
    av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds the maximum allowed.\n");
    return AVERROR_INVALIDDATA;
}

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    // Read the matrix output channel
    ch = get_bits(gbp, 4);
    if (ch >= m->avctx->channels) {
        av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix output channel: %u\n", ch);
        return AVERROR_INVALIDDATA;
    }
    s->matrix_params[mat].output_channel = ch;

    // Read the number of fractional bits
    s->matrix_params[mat].frac_bits = get_bits(gbp, 4);

    // Read whether LSB bypass is enabled
    s->matrix_params[mat].lsb_bypass = get_bits1(gbp);

    // Read and store the matrix coefficient values
    for (ch = 0; ch <= s->matrix_params[mat].output_channel; ch++) {
        int i;
        for (i = 0; i < 8; i++) {
            s->matrix_params[mat].coeff[ch][i] = get_sbits(gbp, 8);
        }
    }

    // If noise is present, read the matrix noise shift value
    if (s->noise_present) {
        s->matrix_params[mat].noise_shift = get_bits(gbp, 4);
    } else {
        s->matrix_params[mat].noise_shift = 0;
    }
}
