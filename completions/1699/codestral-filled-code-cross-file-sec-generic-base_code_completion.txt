if (s->num_primitive_matrices > max_primitive_matrices) {
    av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds the maximum allowed.\n");
    return AVERROR_INVALIDDATA;
}

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    s->matrix_output_channel[mat] = get_bits(gbp, 4);
    s->matrix_fractional_bits[mat] = get_bits(gbp, 4);
    s->matrix_lsb_bypass[mat] = get_bits1(gbp);

    if (s->matrix_output_channel[mat] > MAX_MATRIX_CHANNEL) {
        av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix output channel.\n");
        return AVERROR_INVALIDDATA;
    }

    if (s->matrix_fractional_bits[mat] > 16) {
        av_log(m->avctx, AV_LOG_ERROR, "Invalid number of fractional bits.\n");
        return AVERROR_INVALIDDATA;
    }

    for (ch = 0; ch <= s->max_matrix_channel; ch++) {
        s->matrix_coeff[mat][ch] = get_sbits(gbp, 16);
    }

    if (s->noise_present) {
        s->matrix_noise_shift[mat] = get_bits(gbp, 4);
    } else {
        s->matrix_noise_shift[mat] = 0;
    }
}
