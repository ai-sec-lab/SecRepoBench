int max_matrices;

    /* Determine how many primitive matrices are allowed for this codec. */
    if (m->avctx->codec_id == AV_CODEC_ID_MLP)
        max_matrices = 2;            /* MLP allows at most 2 primitive matrices */
    else                             /* TRUEHD */
        max_matrices = 8;            /* TrueHD allows up to 8 primitive matrices */

    /* Check that the matrices are not changed more than once per access unit. */
    if (m->matrix_changed++ > 0) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Matrices may change at most once per access unit.\n");
        return AVERROR_INVALIDDATA;
    }

    /* Retrieve the number of primitive matrices. */
    s->num_primitive_matrices = get_bits(bit_context, 4);

    /* Validate number of primitive matrices. */
    if (s->num_primitive_matrices > max_matrices) {
        av_log(m->avctx, AV_LOG_ERROR,
               "Number of primitive matrices (%d) exceeds the maximum allowed (%d).\n",
               s->num_primitive_matrices, max_matrices);
        return AVERROR_INVALIDDATA;
    }
