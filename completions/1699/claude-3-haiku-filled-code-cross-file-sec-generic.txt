if (s->num_primitive_matrices > max_primitive_matrices) {
    av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds the maximum allowed.\n");
    return AVERROR_INVALIDDATA;
}

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    ch = get_bits(gbp, 4);
    if (ch >= m->avctx->channels) {
        av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix output channel: %d.\n", ch);
        return AVERROR_INVALIDDATA;
    }

    s->matrix_precision[mat] = get_bits(gbp, 4);
    if (s->matrix_precision[mat] > 15) {
        av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix precision: %d.\n", s->matrix_precision[mat]);
        return AVERROR_INVALIDDATA;
    }

    s->matrix_lsb_bypass[mat] = get_bits1(gbp);

    for (int i = 0; i <= ch; i++) {
        s->matrix_coeff[mat][i] = get_sbits(gbp, s->matrix_precision[mat] + 1);
    }

    if (s->noise_shift_present) {
        s->matrix_noise_shift[mat] = get_bits(gbp, 4);
    } else {
        s->matrix_noise_shift[mat] = 0;
    }
}
