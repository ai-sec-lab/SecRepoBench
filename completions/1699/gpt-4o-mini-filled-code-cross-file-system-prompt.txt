if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices exceeds maximum allowed.\n");
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->primitive_matrix[mat].output_channel = get_bits(gbp, 4);
        s->primitive_matrix[mat].fractional_bits = get_bits(gbp, 4);
        s->primitive_matrix[mat].lsb_bypass = get_bits1(gbp);

        if (s->primitive_matrix[mat].output_channel >= MAX_CHANNELS) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid output channel for matrix %d.\n", mat);
            return AVERROR_INVALIDDATA;
        }

        if (s->primitive_matrix[mat].fractional_bits > MAX_FRACTIONAL_BITS) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid fractional bits for matrix %d.\n", mat);
            return AVERROR_INVALIDDATA;
        }

        for (ch = 0; ch < MAX_MATRIX_CHANNEL; ch++) {
            s->primitive_matrix[mat].coeff[ch] = get_sbits(gbp, 16);
        }

        if (get_bits1(gbp)) {
            s->primitive_matrix[mat].noise_shift = get_bits(gbp, 4);
        } else {
            s->primitive_matrix[mat].noise_shift = 0;
        }
    }
