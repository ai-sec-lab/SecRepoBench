if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices.\n");
        return AVERROR_INVALIDDATA;
    }
    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->matrix_params[mat].output_channel = get_bits(gbp, 5);
        if (s->matrix_params[mat].output_channel > MAX_MATRIX_CHANNEL) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix output channel.\n");
            return AVERROR_INVALIDDATA;
        }
        s->matrix_params[mat].fractional_bits = get_bits(gbp, 3);
        if (s->matrix_params[mat].fractional_bits > 15) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid fractional bits.\n");
            return AVERROR_INVALIDDATA;
        }
        s->matrix_params[mat].lsb_bypass = get_bits1(gbp);
        for (ch = 0; ch <= MAX_MATRIX_CHANNEL; ch++) {
            int i;
            for (i = 0; i < s->num_primitive_matrices; i++) {
                s->matrix_coeff[ch][mat][i] = get_sbits(gbp, 16);
            }
        }
    }
    if (m->noise_present) {
        for (mat = 0; mat < s->num_primitive_matrices; mat++) {
            s->matrix_params[mat].noise_shift = get_bits(gbp, 4);
        }
    } else {
        for (mat = 0; mat < s->num_primitive_matrices; mat++) {
            s->matrix_params[mat].noise_shift = 0;
        }
    }
