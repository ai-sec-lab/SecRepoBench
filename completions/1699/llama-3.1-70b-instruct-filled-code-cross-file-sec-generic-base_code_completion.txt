if (s->num_primitive_matrices > max_primitive_matrices) {
    av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices %d exceeds maximum allowed %d.\n",
           s->num_primitive_matrices, max_primitive_matrices);
    return AVERROR_INVALIDDATA;
}

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    unsigned int out_ch = get_bits(gbp, 5);
    if (out_ch >= MAX_MATRIX_CHANNEL) {
        av_log(m->avctx, AV_LOG_ERROR, "Matrix output channel %d is out of range.\n", out_ch);
        return AVERROR_INVALIDDATA;
    }

    s->matrix_out_channel[mat] = out_ch;

    unsigned int frac_bits = get_bits(gbp, 4);
    if (frac_bits > 24) {
        av_log(m->avctx, AV_LOG_ERROR, "Matrix fractional bits %d is out of range.\n", frac_bits);
        return AVERROR_INVALIDDATA;
    }

    s->matrix_frac_bits[mat] = frac_bits;

    s->matrix_lsb_bypass[mat] = get_bits1(gbp);

    for (ch = 0; ch < MAX_MATRIX_CHANNEL; ch++) {
        s->matrix_coeff[mat][ch] = get_bits(gbp, 8);
    }

    if (s->noise_present) {
        s->matrix_noise_shift[mat] = get_bits(gbp, 4);
    } else {
        s->matrix_noise_shift[mat] = 0;
    }
}
