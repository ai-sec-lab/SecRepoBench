if (s->num_primitive_matrices > max_primitive_matrices) {
    av_log(m->avctx, AV_LOG_ERROR, "Too many primitive matrices: %d\n", s->num_primitive_matrices);
    return AVERROR_INVALIDDATA;
}

for (mat = 0; mat < s->num_primitive_matrices; mat++) {
    s->matrix_out_ch[mat] = get_bits(gbp, 4);
    s->matrix_encoding[mat] = get_bits(gbp, 4);
    s->matrix_outch[mat] = get_bits(gbp, 4);

    if (s->matrix_outch[mat] >= m->avctx->channels) {
        av_log(m->avctx, AV_LOG_ERROR, "Invalid matrix output channel: %d\n", s->matrix_outch[mat]);
        return AVERROR_INVALIDDATA;
    }

    for (ch = 0; ch < s->matrix_out_ch[mat]; ch++) {
        s->matrix_coeff[mat][ch] = get_sbits(gbp, 6);
    }
}

s->num_primitive_matrices_out = get_bits(gbp, 4);

if (s->num_primitive_matrices_out > MAX_MATRICES_OUT) {
    av_log(m->avctx, AV_LOG_ERROR, "Too many output matrices: %d\n", s->num_primitive_matrices_out);
    return AVERROR_INVALIDDATA;
}
