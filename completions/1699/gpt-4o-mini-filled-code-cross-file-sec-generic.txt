if (s->num_primitive_matrices > max_primitive_matrices) {
        av_log(m->avctx, AV_LOG_ERROR, "Number of primitive matrices %u exceeds maximum %d.\n",
               s->num_primitive_matrices, max_primitive_matrices);
        return AVERROR_INVALIDDATA;
    }

    for (mat = 0; mat < s->num_primitive_matrices; mat++) {
        s->primitive_matrices[mat].output_channel = get_bits(gbp, 4);
        s->primitive_matrices[mat].fractional_bits = get_bits(gbp, 4);
        s->primitive_matrices[mat].lsb_bypass = get_bits1(gbp);

        // Validate output channel
        if (s->primitive_matrices[mat].output_channel >= MAX_CHANNELS) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid output channel %u for matrix %u.\n",
                   s->primitive_matrices[mat].output_channel, mat);
            return AVERROR_INVALIDDATA;
        }

        // Validate fractional bits
        if (s->primitive_matrices[mat].fractional_bits > MAX_FRACTIONAL_BITS) {
            av_log(m->avctx, AV_LOG_ERROR, "Invalid fractional bits %u for matrix %u.\n",
                   s->primitive_matrices[mat].fractional_bits, mat);
            return AVERROR_INVALIDDATA;
        }

        // Read matrix coefficients
        for (ch = 0; ch < MAX_MATRIX_CHANNEL; ch++) {
            s->primitive_matrices[mat].coefficients[ch] = get_sbits(gbp, 16);
        }

        // Read noise shift value if noise is present
        if (s->primitive_matrices[mat].noise_present) {
            s->primitive_matrices[mat].noise_shift = get_bits(gbp, 4);
        } else {
            s->primitive_matrices[mat].noise_shift = 0;
        }
    }
