// we currently do not generate additional keys for LightWave's
        // pre-behaviours.  However, ClearAnimRangeSetup() expects us to
        // remember the position of the first and last *original* keys so
        // that it can later remove any keys which might be inserted here
        // in a future implementation.  Since we do not alter the key list
        // at this point, the first original key is at index 0 and the last
        // original key is at index keys.size()-1.
        (*it).old_first = 0;
        (*it).old_last  = static_cast<size_t>((*it).keys.size() - 1);
