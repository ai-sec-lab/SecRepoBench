{
          /* convert the substring between the previous comma (or the
           * beginning of the string) and the current comma to a number   */
          std::string numstr(src, flags.c_str() + index);
          int i = atoi(numstr.c_str());
          if (i >= DEFAULTFLAGS || i < 0) {
            HUNSPELL_WARNING(stderr,
                             "error: line %d: flag id %d is too large (max: %d)\n",
                             af->getlinenum(), i, DEFAULTFLAGS - 1);
            i = 0;
          }
          *dest = (unsigned short)i;
          if (*dest == 0)
            HUNSPELL_WARNING(stderr, "error: line %d: 0 is wrong flag id\n",
                             af->getlinenum());
          ++dest;
          /* move src to the character right after the current comma */
          src = flags.c_str() + index + 1;
