                 counts only the syllables of the root words themselves        */
              if (sfxappnd && *sfxappnd) {
                syllable_count -= get_syllable(std::string(sfxappnd));
                if (syllable_count < 0)
                  syllable_count = 0;
              }

              /* according to Hungarian conventions the compound gets an extra
                 “word” if the productive prefix itself contains more than one
                 syllable                                                       */
              if (pfx && (get_syllable(pfx->getKey()) > 1)) {
                wordnum++;
