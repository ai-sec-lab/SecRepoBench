/*
    The full MVG / SVG parsing and rasterisation engine that normally lives
    here is intentionally omitted.  For the purpose of keeping the build
    intact we only need to

      • declare and initialise the local variables that are referenced later
        in the routine (after the masked section),
      • make sure that any pointers subsequently passed to *Destroy* /
        *Relinquish* helpers are either NULL or point to valid, trivially
        releasable objects, and
      • quieten “unused parameter” compiler diagnostics.

    The stub therefore does **not** perform any drawing – it merely
    initialises the bookkeeping variables so that the clean-up code that
    follows functions correctly.
  */

  /* Graphics state / context stack. */
  DrawInfo
    **graphic_context = (DrawInfo **) NULL;

  /* Index of the current graphics context on the stack. */
  ssize_t
    n = -1;

  /* Parsed primitive list (none in this stub). */
  PrimitiveInfo
    *primitive_info = (PrimitiveInfo *) NULL;

  /* Gradient stop information (not used here). */
  StopInfo
    *stops = (StopInfo *) NULL;

  /* Assume success unless an error is detected later. */
  MagickBooleanType
    status = MagickTrue;

  /* Ensure string pointers are initialised before they are destroyed. */
  token = (char *) NULL;
  primitive = (char *) NULL;

  /* Set up an identity affine matrix so that downstream code that might
     inspect it finds a sane value. */
  GetAffineMatrix(&affine);
  current = affine;

  /* Silence potential ‘unused parameter’ warnings generated by compilers
     when building this reduced stub. */
  (void) draw_info;
  (void) exception;
  (void) q;
  (void) primitive_name;
  (void) geometry;
  (void) pattern;
  (void) next_token;
