HEVCFrame *ref = find_ref_idx(s, poc);
if (!ref) {
    // If no reference is found, attempt to generate a missing reference frame
    ref = generate_missing_ref(s, poc);
    if (!ref) {
        av_log(s->avctx, AV_LOG_ERROR, "Failed to generate missing reference for POC %d\n", poc);
        return AVERROR(ENOMEM);
    }
}

if (list->nb_refs >= MAX_REFS) {
    av_log(s->avctx, AV_LOG_ERROR, "Too many references in the list\n");
    return AVERROR(EINVAL);
}

for (int i = 0; i < list->nb_refs; i++) {
    if (list->list[i] == ref->poc) {
        av_log(s->avctx, AV_LOG_WARNING, "Duplicate reference found for POC %d\n", poc);
        return AVERROR(EINVAL);
    }
}
