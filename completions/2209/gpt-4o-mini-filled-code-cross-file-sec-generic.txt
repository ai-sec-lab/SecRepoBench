HEVCFrame *ref = find_ref_idx(s, poc);
    if (!ref) {
        av_log(s->avctx, AV_LOG_ERROR, "No reference found for POC %d\n", poc);
        return AVERROR_INVALIDDATA;
    }

    // Check for duplicate reference
    for (int i = 0; i < list->nb_refs; i++) {
        if (list->list[i] == ref->poc) {
            av_log(s->avctx, AV_LOG_ERROR, "Duplicate reference found for POC %d\n", ref->poc);
            return AVERROR_INVALIDDATA;
        }
    }

    // Check for overflow
    if (list->nb_refs >= MAX_REF_PICS) {
        av_log(s->avctx, AV_LOG_ERROR, "Reference list overflow\n");
        return AVERROR_INVALIDDATA;
    }

    // Add the reference
    list->list[list->nb_refs] = ref->poc;
    list->ref[list->nb_refs]  = ref;
    list->nb_refs++;

    mark_ref(ref, ref_flag);
    return 0;
