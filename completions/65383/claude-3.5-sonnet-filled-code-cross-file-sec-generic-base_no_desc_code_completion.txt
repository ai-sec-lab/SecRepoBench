case 'Z':
    if (!tm->blk) {
        if (!(tm->blk = cram_new_block(EXTERNAL, key)))
            goto err;
        codec->u.e_byte_array_len.val_codec->out = tm->blk;
    }

    aux += 3;
    while (aux < aux_end && *aux) {
        BLOCK_APPEND_CHAR(tm->blk, *aux);
        aux++;
    }
    if (aux >= aux_end)
        goto err;
    BLOCK_APPEND_CHAR(tm->blk, *aux);
    aux++;
    break;

case 'H':
    if (!tm->blk) {
        if (!(tm->blk = cram_new_block(EXTERNAL, key)))
            goto err;
        codec->u.e_byte_array_len.val_codec->out = tm->blk;
    }

    aux += 3;
    while (aux < aux_end && *aux) {
        BLOCK_APPEND_CHAR(tm->blk, *aux);
        aux++;
    }
    if (aux >= aux_end)
        goto err;
    BLOCK_APPEND_CHAR(tm->blk, *aux);
    aux++;
    break;

case 'B':
    {
        int type = aux[3], size = aux_type_size(type);
        uint32_t count = le_to_u32(&aux[4]);

        if (!tm->blk) {
            if (!(tm->blk = cram_new_block(EXTERNAL, key)))
                goto err;
            codec->u.e_byte_array_len.val_codec->out = tm->blk;
        }

        if (aux_end - aux < 8 + size * count)
            goto err;

        aux += 3;
        BLOCK_APPEND(tm->blk, aux, 5 + size * count);
        aux += 5 + size * count;
    }
    break;
