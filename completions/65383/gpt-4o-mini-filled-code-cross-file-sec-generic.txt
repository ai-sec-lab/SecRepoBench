case 'H': {
            // Handle hexadecimal string type 'H'
            size_t hex_len = strlen((char *)aux + 3);
            if (aux_end - aux < 3 + hex_len + 1) // +1 for null terminator
                goto err;

            if (!tm->blk) {
                if (!(tm->blk = cram_new_block(EXTERNAL, key)))
                    goto err;
                codec->u.e_byte_array_len.val_codec->out = tm->blk;
            }

            aux += 3; // Move past the 'H' type identifier
            BLOCK_APPEND(tm->blk, aux, hex_len);
            BLOCK_APPEND_CHAR(tm->blk, '\0'); // Null-terminate the string
            aux += hex_len + 1; // Move past the data and null terminator
            break;
        }

        case 'B': {
            // Handle byte array type 'B'
            int subtype = aux[3]; // Get the subtype
            int count = bam_aux2i_end((uint8_t *)aux + 4, (uint8_t *)aux_end); // Get the count
            size_t byte_array_size = count * get_size_of_subtype(subtype); // Calculate size based on subtype

            if (aux_end - aux < 4 + byte_array_size) // Check if there's enough data
                goto err;

            if (!tm->blk) {
                if (!(tm->blk = cram_new_block(EXTERNAL, key)))
                    goto err;
                codec->u.e_byte_array_len.val_codec->out = tm->blk;
            }

            aux += 4; // Move past the 'B' type identifier and count
            BLOCK_APPEND(tm->blk, aux, byte_array_size); // Append the byte array data
            aux += byte_array_size; // Move past the byte array data
            break;
        }
