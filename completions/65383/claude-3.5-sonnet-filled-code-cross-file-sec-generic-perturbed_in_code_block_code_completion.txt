case 'H': case 'Z':
    {
        if (!tm->blk) {
            if (!(tm->blk = cram_new_block(EXTERNAL, key)))
                goto err;
            codec->u.e_byte_array_len.val_codec->out = tm->blk;
        }

        aux += 3;
        char *aux_end_z = aux;
        while (*aux_end_z)
            aux_end_z++;
        if (aux_end_z >= aux_end)
            goto err;
        codec->encode(s, codec, aux, aux_end_z - aux + 1);
        aux = aux_end_z + 1;
    }
    break;

case 'B':
    {
        int type = aux[3], blen;
        uint32_t count = le_to_u32((uint8_t *)aux+4);

        if (aux_end - aux < 8)
            goto err;

        switch (type) {
        case 'c': case 'C':
            blen = count;
            break;
        case 's': case 'S':
            blen = 2*count;
            break;
        case 'i': case 'I': case 'f':
            blen = 4*count;
            break;
        default:
            goto err;
        }

        if (aux_end - aux < 8 + blen)
            goto err;

        if (!tm->blk) {
            if (!(tm->blk = cram_new_block(EXTERNAL, key)))
                goto err;
            codec->u.e_byte_array_len.val_codec->out = tm->blk;
        }

        codec->encode(s, codec, aux+3, blen+5);
        aux += 8 + blen;
    }
    break;
