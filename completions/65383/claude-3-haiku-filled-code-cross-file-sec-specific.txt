case 'H':
    if (aux_end - aux < 3+1)
        goto err;

    if (!tm->blk) {
        if (!(tm->blk = cram_new_block(EXTERNAL, key)))
            goto err;
        codec->u.e_byte_array_len.val_codec->out = tm->blk;
    }

    aux+=3;
    BLOCK_APPEND(tm->blk, aux, strlen(aux)+1);
    aux += strlen(aux) + 1;
    break;

case 'B':
    if (aux_end - aux < 3+1)
        goto err;

    if (!tm->blk) {
        if (!(tm->blk = cram_new_block(EXTERNAL, key)))
            goto err;
        codec->u.e_byte_array_len.val_codec->out = tm->blk;
    }

    int sub_type = aux[3];
    int count = bam_aux_len((uint8_t *)aux+3) - 1;
    aux += 4;

    switch (sub_type) {
    case 'c': case 'C':
        BLOCK_APPEND(tm->blk, aux, count);
        aux += count;
        break;
    case 's': case 'S':
        BLOCK_APPEND(tm->blk, aux, count * 2);
        aux += count * 2;
        break;
    case 'i': case 'I': case 'f':
        BLOCK_APPEND(tm->blk, aux, count * 4);
        aux += count * 4;
        break;
    case 'd':
        BLOCK_APPEND(tm->blk, aux, count * 8);
        aux += count * 8;
        break;
    default:
        hts_log_error("Unsupported BAM array type '%c'", sub_type);
        goto err;
    }
    break;
