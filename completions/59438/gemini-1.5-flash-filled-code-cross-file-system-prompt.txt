uint32_t len;
	unsigned int nr = 0, nt = 0;

	while (uc < ue) {
		len = getlen(uc, ue);
		if (len == 0xFFFFFFFF)
			break;
		if (len & 1) {
			if (uc + len + 1 > ue)
				return 0;
			uc += len + 1;
			nr++;
		} else {
			if (uc + len > ue)
				return 0;
			if (getlen(uc + len, ue) != len)
				return 0;
			uc += len * 2;
			nr++;
		}
		if (++nt >= SIMH_TAPEMARKS)
			break;
	}
	return nr > 0;
