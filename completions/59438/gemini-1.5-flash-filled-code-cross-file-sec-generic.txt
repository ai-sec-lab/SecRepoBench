uint32_t len;
	uint32_t reclen;
	int tapemarks = 0;

	while (uc < ue) {
		len = getlen(uc, ue);
		if (len == 0)
			return 0;
		uc += 4;
		reclen = getlen(uc, ue);
		if (reclen == 0)
			return 0;
		if (len != reclen)
			return 0;
		uc += 4 + len;
		if (len == 0xFFFFFFFF)
			break;
		if (++tapemarks >= SIMH_TAPEMARKS)
			return 0;
	}
	return tapemarks > 0;
