case CirclePrimitive:
{
  if (primitive_info[j].coordinates != 2)
  {
    status=MagickFalse;
    break;
  }
  coordinates=GetCircleCoordinates(primitive_info[j].point,
    primitive_info[j+1].point);
  break;
}
case ArcPrimitive:
{
  if (primitive_info[j].coordinates != 3)
  {
    status=MagickFalse;
    break;
  }
  coordinates=GetArcCoordinates(primitive_info[j].point,
    primitive_info[j+1].point,primitive_info[j+2].point);
  break;
}
