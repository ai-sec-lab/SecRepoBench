int strength = 0;
const char *p = line;

if (p == NULL || *p == '\0')
    return -1;

while (*p && isdigit((unsigned char)*p)) {
    int digit = *p - '0';
    if (strength > (INT_MAX - digit) / 10) {
        file_error(ms, 0, "Strength value too large");
        return -1;
    }
    strength = strength * 10 + digit;
    p++;
}

if (*p != '\0' && !isspace((unsigned char)*p)) {
    file_error(ms, 0, "Invalid character in strength");
    return -1;
}

if (strength < 0 || strength > 100) {
    file_error(ms, 0, "Strength must be between 0 and 100");
    return -1;
}
