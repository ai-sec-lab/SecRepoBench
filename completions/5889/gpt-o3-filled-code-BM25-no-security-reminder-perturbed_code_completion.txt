case ArcPrimitive:
      {
        /*
          Estimate the number of points needed to render an arc.  We use the
          diagonal of the bounding box as an estimate for the radius and add
          a generous margin based on Ï€, BezierQuantum and an additional
          constant to ensure the buffer is large enough.
        */
        double
          alpha,
          beta,
          radius;

        alpha=bounds.x2-bounds.x1;
        beta=bounds.y2-bounds.y1;
        radius=hypot((double) alpha,(double) beta);
        points_extent+=2.0*ceil((double) MagickPI*radius)+6*BezierQuantum+360;
        break;
      }
      case EllipsePrimitive:
      {
        /*
          Estimate the number of points required for an ellipse.  The major
          and minor axes are derived from the bounding box, converted to an
          equivalent radius and then used to compute a conservative point
          estimate.
        */
        double
          alpha,
          beta,
          radius;

        alpha=bounds.x2-bounds.x1;
        beta=bounds.y2-bounds.y1;
        radius=hypot((double) alpha,(double) beta);
        points_extent+=2.0*ceil((double) MagickPI*radius)+6*BezierQuantum+360;
        break;
      }
      case CirclePrimitive:
      {
        /*
          Estimate the number of points required for a circle.  The radius is
          obtained from the bounding box and the circumference is used as the
          basis for our point estimate.
        */
        double
          alpha,
          beta,
          radius;

        alpha=bounds.x2-bounds.x1;
        beta=bounds.y2-bounds.y1;
        radius=hypot((double) alpha,(double) beta);
        points_extent+=2.0*ceil((double) MagickPI*radius)+360;
        break;
      }
