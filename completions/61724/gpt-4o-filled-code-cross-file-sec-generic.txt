std::unordered_map<unsigned int, glyph_variation_data_t> glyph_variation_map;

    for (const auto& mapping : c->glyph_mapping)
    {
      unsigned int new_glyph_id = mapping.first;
      unsigned int old_glyph_id = mapping.second;

      glyph_variation_data_t variation_data;
      if (!get_glyph_variation_data(old_glyph_id, variation_data))
      {
        // Handle error in retrieving glyph variation data
        return false;
      }

      glyph_variation_map[new_glyph_id] = variation_data;
    }

    // Special case for glyph ID 0
    if (c->glyph_mapping.find(0) != c->glyph_mapping.end() && !c->notdef_outline)
    {
      glyph_variation_map[0] = glyph_variation_data_t(); // Default empty data
    }

    if (glyph_variation_map.empty())
    {
      // Handle error if no glyph variations were populated
      return false;
    }

    return create_glyph_variations(c->axis_count, c->shared_tuples, c->plan, glyph_variation_map, glyph_vars);
