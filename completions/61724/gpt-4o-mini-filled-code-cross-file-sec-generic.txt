std::unordered_map<uint32_t, glyph_variation_data_t> glyph_variation_map;

    for (const auto& mapping : c->glyph_id_mappings) {
        uint32_t new_glyph_id = mapping.new_glyph_id;
        uint32_t old_glyph_id = mapping.old_glyph_id;

        // Retrieve glyph variation data for the old glyph ID
        glyph_variation_data_t variation_data;
        if (!retrieve_glyph_variation_data(old_glyph_id, variation_data)) {
            // Handle error in retrieving variation data
            return false; // or appropriate error handling
        }

        // Store the variation data in the map
        glyph_variation_map[new_glyph_id] = variation_data;
    }

    // Handle special case for glyph ID zero
    if (glyph_variation_map.find(0) == glyph_variation_map.end() && !(c->flags & NOTDEF_OUTLINE)) {
        glyph_variation_map[0] = default_glyph_variation_data(); // Set default data for ID zero
    }

    // Check for errors in the glyph variation map
    if (glyph_variation_map.empty()) {
        // Handle error: no valid glyph variations found
        return false; // or appropriate error handling
    }

    // Create glyph variations using the populated map
    if (!create_glyph_variations(c->axis_count, c->shared_tuples, c->plan, glyph_variation_map, glyph_vars)) {
        // Handle error in creating glyph variations
        return false; // or appropriate error handling
    }

    return true; // Success
