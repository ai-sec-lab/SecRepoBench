std::unordered_map<uint32_t, glyph_variation_data_t> glyph_variation_map;

    // Iterate over the list of new glyph IDs and their corresponding old glyph IDs
    for (const auto& mapping : c->glyph_id_mappings) {
        uint32_t new_glyph_id = mapping.new_id;
        uint32_t old_glyph_id = mapping.old_id;

        // Validate the new glyph ID
        if (new_glyph_id == 0) {
            // Handle special case for ID zero
            if (!(c->flags & NOTDEF_OUTLINE)) {
                glyph_variation_map[new_glyph_id] = glyph_variation_data_t(); // Default empty data
            }
            continue; // Skip further processing for ID zero
        }

        // Retrieve glyph variation data for the old glyph ID
        glyph_variation_data_t variation_data;
        if (!retrieve_glyph_variation_data(old_glyph_id, variation_data)) {
            // Handle error: unable to retrieve data for old glyph ID
            log_error("Failed to retrieve glyph variation data for old ID: " + std::to_string(old_glyph_id));
            return false; // Indicate failure
        }

        // Store the retrieved variation data in the map
        glyph_variation_map[new_glyph_id] = variation_data;
    }

    // Check for errors in the glyph variation map population
    if (glyph_variation_map.empty()) {
        log_error("No glyph variations were populated.");
        return false; // Indicate failure
    }

    // Create glyph variations using the populated map
    if (!create_glyph_variations(c->axis_count, c->shared_tuples, c->plan, glyph_variation_map)) {
        log_error("Failed to create glyph variations.");
        return false; // Indicate failure
    }

    return true; // Indicate success
