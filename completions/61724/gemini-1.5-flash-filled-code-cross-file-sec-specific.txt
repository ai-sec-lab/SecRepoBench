std::map<hb_codepoint_t, glyph_variation_data_t> new_glyph_variations;
    for (const auto& pair : new_glyph_id_mapping_) {
      hb_codepoint_t new_glyph_id = pair.first;
      hb_codepoint_t old_glyph_id = pair.second;
      glyph_variation_data_t variation_data;
      if (get_glyph_variation_data(old_glyph_id, variation_data)) {
        new_glyph_variations[new_glyph_id] = variation_data;
      } else {
        // Handle error: Could not retrieve variation data for old glyph ID.  Consider logging or throwing an exception.
        return false;
      }
    }

    if (new_glyph_id_mapping_.empty() && !(flags_ & NOTDEF_OUTLINE)) {
      new_glyph_variations[0] = {}; // Default empty data for glyph ID 0
    }


    if (!create_glyph_variations(c, axis_count_, shared_tuples_, plan_, new_glyph_variations, glyph_vars)) {
      return false;
    }
    return true;
