std::unordered_map<uint32_t, glyph_variation_data_t> variation_map;

    for (const auto& mapping : c->glyph_id_mappings) {
        uint32_t new_glyph_id = mapping.new_glyph_id;
        uint32_t old_glyph_id = mapping.old_glyph_id;

        variation_map[new_glyph_id] = get_glyph_variation_data(old_glyph_id);

        if (new_glyph_id == 0 && !(c->flags & NOTDEF_OUTLINE)) {
            variation_map[new_glyph_id] = default_empty_variation_data();
        }
    }

    if (variation_map.empty()) {
        return false; // Error handling for empty variation map
    }

    create_glyph_variations(c->axis_count, c->shared_tuples, c->plan, variation_map, glyph_vars);
