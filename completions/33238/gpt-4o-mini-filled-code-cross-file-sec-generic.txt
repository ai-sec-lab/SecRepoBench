// Remove leading spaces or new lines
        std::string::iterator it = in.begin();
        while (it != in.end() && (std::isspace(*it) || *it == '\n')) {
            ++it;
        }
        if (it != in.begin()) {
            in.erase(in.begin(), it);
        }

        const char separator = getOsSeparator();
        char last = 0;

        for (it = in.begin(); it != in.end(); ++it) {
            int remaining = std::distance(in.end(), it);
            // Skip over "://" and "\\\\" sequences
            if (remaining >= 3 && !strncmp(&*it, "://", 3)) {
                it += 3;
                continue;
            }
            if (it == in.begin() && remaining >= 2 && !strncmp(&*it, "\\\\", 2)) {
                it += 2;
                continue;
            }

            // Normalize path delimiters
            if (*it == '/' || *it == '\\') {
                *it = separator;

                // Remove double delimiters
                if (last == *it) {
                    it = in.erase(it);
                    --it;
                }
            } else if (*it == '%' && remaining > 2) {
                // Handle hexadecimal sequences in URIs
                if (IsHex(*(it + 1)) && IsHex(*(it + 2))) {
                    *it = HexOctetToDecimal(&*it);
                    it = in.erase(it + 1, it + 3);
                    --it;
                }
            }

            last = *it;
        }
