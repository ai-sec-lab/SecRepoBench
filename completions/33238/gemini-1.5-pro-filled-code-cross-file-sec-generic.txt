if (in.empty()) {
            return;
        }

        // Remove leading spaces or new lines from the path
        char last = 0;
        std::string::iterator it = in.begin();
        while (IsSpaceOrNewLine(*it)) {
            ++it;
        }
        if (it != in.begin()) {
            in.erase(in.begin(), it);
        }


        const char separator = getOsSeparator();
        for (it = in.begin(); it != in.end(); ++it) {
            // Check for protocol strings like "://" and "\\\\"
            size_t remaining = std::distance(it, in.end());
            if (remaining >= 3 && *it == ':' && *(it + 1) == '/' && *(it + 2) == '/') {
                it += 2;
                continue;
            }
            if (it == in.begin() && remaining >= 2 && *it == '\\' && *(it + 1) == '\\') {
                it += 1;
                continue;
            }

            // Replace forward and backward slashes with OS specific separator
            if (*it == '/' || *it == '\\') {
                *it = separator;
                if (last == separator) {
                    it = in.erase(it);
                    --it;
                }
            } else if (*it == '%' && remaining > 2) {
                // Convert the hex code to its respective character in URI
                if (IsHex(*(it + 1)) && IsHex(*(it + 2))) {
                    char hex[3] = {*(it + 1), *(it + 2), '\0'};
                    char decodedChar = static_cast<char>(std::strtol(hex, nullptr, 16));
                    *it = decodedChar;
                    it = in.erase(it + 1, it + 3);
                    --it;
                }

            }
            last = *it;
        }
