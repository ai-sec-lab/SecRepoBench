// <MASK>
        if (in.empty()) {
            return;
        }

        // Remove leading spaces or new lines
        size_t start = in.find_first_not_of(" \n");
        if (start != std::string::npos) {
            in = in.substr(start);
        }

        const char separator = getOsSeparator();
        for (size_t i = 0; i < in.length(); i++) {
            // Replace forward slashes and backslashes with the OS directory separator
            if (in[i] == '/' || in[i] == '\\') {
                in[i] = separator;

                // Remove double delimiters
                if (i > 0 && in[i-1] == separator) {
                    in.erase(i--, 1);
                }
            }
            // Handle hexadecimal sequences in URIs
            else if (in[i] == '%' && i+2 < in.length() && IsHex(in[i+1]) && IsHex(in[i+2])) {
                in[i] = HexOctetToDecimal(&in[i]);
                in.erase(i+1, 2);
                i--;
            }
