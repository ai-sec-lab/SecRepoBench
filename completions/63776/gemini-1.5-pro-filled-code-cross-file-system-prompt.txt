if (obj->size != (dat->byte - oldpos))
                  {
                    if (dat->version < R_11)
                      {
                        if (obj->size > (dat->byte - oldpos))
                          {
                            LOG_ERROR ("Invalid object size");
                            UNKNOWN_UNTIL (oldpos + obj->size);
                          }
                        else
                          obj->size = (dat->byte - oldpos);
                      }
                    else
                      {
                        size_t crc_start = dat->byte;
                        // R13 has CRC after each entity
                        if (dat->version > R_13)
                          {
                            BITCODE_3BD crc;
                            crc = bit_read_3BD (dat);
                            LOG_TRACE ("CRC: " FORMAT_3BD "\n", crc);
                          }
                        if (obj->size > (dat->byte - oldpos))
                          {
                            if (obj->size > dat->size)
                              {
                                LOG_ERROR ("Invalid object size");
                                UNKNOWN_UNTIL (dat->size);
                              }
                            else
                              UNKNOWN_UNTIL (oldpos + obj->size);
                          }
                        else
                          {
                            if (obj->address + obj->size != dat->byte)
                              {
                                size_t offset = obj->address + obj->size;
                                size_t len = dat->byte - offset;
                                LOG_TRACE ("trailing unknown data (%zu):\n",
                                           len);
                                obj->trailing_unknown
                                    = malloc (sizeof (char) * len);
                                if (obj->trailing_unknown)
                                  {
                                    bit_get_bytes (dat, offset, len,
                                                   obj->trailing_unknown);
                                    LOG_TRACE_BYTES (obj->trailing_unknown,
                                                     len);
                                  }
                                LOG_TRACE ("\n");
                              }
                          }
                        obj->size = (dat->byte - oldpos) & 0xFFFFFFFF;
                      }
                  }
