if (dat->version < R_11)
                  {
                    if (obj->size > (BITCODE_RL)(dat->size - oldpos))
                      {
                        LOG_ERROR ("size too big: %x > %" PRIuSIZE "\n",
                                   obj->size, dat->size - oldpos);
                        obj->size = (BITCODE_RL)(dat->size - oldpos);
                      }
                    if (obj->size < (BITCODE_RL)(dat->byte - oldpos))
                      {
                        LOG_ERROR ("size too small: %x < %" PRIuSIZE "\n",
                                   obj->size, dat->byte - oldpos);
                        obj->size = (BITCODE_RL)(dat->byte - oldpos);
                      }
                    if (obj->size > (BITCODE_RL)(dat->byte - oldpos))
                      {
                        size_t remaining = obj->size - (dat->byte - oldpos);
                        obj->unknown_data = malloc (remaining);
                        if (obj->unknown_data)
                          {
                            bit_read_data (dat, remaining, obj->unknown_data);
                            LOG_TRACE ("unknown_data [%u]: ", (unsigned)remaining);
                            LOG_TRACE_DATA (obj->unknown_data, remaining);
                            LOG_TRACE ("\n");
                          }
                        else
                          {
                            LOG_ERROR ("Out of memory");
                            return DWG_ERR_OUTOFMEM;
                          }
                      }
                  }
                else
                  {
                    BITCODE_32 crc;
                    size_t remaining;
                    crc = bit_read_U32 (dat);
                    obj->crc = crc;
                    LOG_TRACE ("crc: %x [U32]\n", crc);
                    if (obj->size != (BITCODE_RL)(dat->byte - oldpos))
                      {
                        if (obj->size < (BITCODE_RL)(dat->byte - oldpos))
                          {
                            LOG_ERROR ("size too small: %x < %" PRIuSIZE "\n",
                                       obj->size, dat->byte - oldpos);
                            obj->size = (BITCODE_RL)(dat->byte - oldpos);
                          }
                        remaining = obj->size - (dat->byte - oldpos);
                        if (remaining > 0)
                          {
                            size_t offset = dat->byte - obj->address;
                            if (offset + remaining > dat->size)
                              {
                                LOG_ERROR ("Invalid size of unknown data: %" PRIuSIZE
                                           " + %" PRIuSIZE " > %" PRIuSIZE "\n",
                                           offset, remaining, dat->size);
                                remaining = dat->size - offset;
                              }
                            if (remaining > 0)
                              {
                                obj->unknown_data = malloc (remaining);
                                if (obj->unknown_data)
                                  {
                                    bit_read_data (dat, remaining,
                                                   obj->unknown_data);
                                    LOG_TRACE ("unknown_data [%u]: ",
                                               (unsigned)remaining);
                                    LOG_TRACE_DATA (obj->unknown_data, remaining);
                                    LOG_TRACE ("\n");
                                  }
                                else
                                  {
                                    LOG_ERROR ("Out of memory");
                                    return DWG_ERR_OUTOFMEM;
                                  }
                              }
                          }
                      }
                  }
