size_t unknown_data_size;
                if (dat->byte > oldpos + obj->size)
                  {
                    unknown_data_size = dat->byte - (oldpos + obj->size);
                    obj->unknown_data = (BITCODE_T)malloc (unknown_data_size);
                    if (!obj->unknown_data)
                      {
                        LOG_ERROR ("Out of memory");
                        return DWG_ERR_OUTOFMEM;
                      }
                    memcpy (obj->unknown_data,
                            (char *)dat->byte - unknown_data_size,
                            unknown_data_size);
                    LOG_WARN ("Unknown data after object at 0x%zx, size %zu",
                              (char *)dat->byte - unknown_data_size,
                              unknown_data_size);
                    dat->byte -= unknown_data_size;
                  }
                else if (dat->byte < oldpos + obj->size)
                  {
                    unknown_data_size = oldpos + obj->size - dat->byte;
                    LOG_ERROR ("Object size too small, missing %zu bytes",
                               unknown_data_size);
                    return DWG_ERR_INVALIDDWG;
                  }
              }
