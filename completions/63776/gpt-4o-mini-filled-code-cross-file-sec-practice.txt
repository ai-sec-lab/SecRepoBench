// Validate the size of the object to ensure it does not exceed the expected size
            if (obj->size > (dat->size - (dat->byte - oldpos))) {
                LOG_ERROR("Invalid object size: %u exceeds available data size: %zu", obj->size, dat->size - (dat->byte - oldpos));
                return DWG_ERR_INVALIDDWG;
            }

            // Extract unknown data if the size does not match the expected size
            if (obj->size < (dat->byte - oldpos)) {
                size_t unknown_size = (dat->byte - oldpos) - obj->size;
                obj->unknown_bits = (unsigned char *)malloc(unknown_size);
                if (!obj->unknown_bits) {
                    LOG_ERROR("Out of memory while allocating unknown bits");
                    return DWG_ERR_OUTOFMEM;
                }
                memcpy(obj->unknown_bits, dat->byte, unknown_size);
                obj->num_unknown_bits = unknown_size;
                LOG_TRACE("Extracted unknown data of size: %zu", unknown_size);
            }

            // Update the data position after processing the object
            dat->byte += obj->size;
            LOG_TRACE("Updated data position to: %zu", dat->byte);
