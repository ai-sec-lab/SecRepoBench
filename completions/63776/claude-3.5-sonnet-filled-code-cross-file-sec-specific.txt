BITCODE_RL obj_size = bit_read_RL(dat);
obj->size = obj_size;
if (obj_size < 2 || obj_size > size)
  {
    LOG_ERROR("Invalid object size %u at pos 0x%lx", obj_size,
              (unsigned long)(obj->address));
    obj->size = size;
    error |= DWG_ERR_VALUEOUTOFBOUNDS;
  }
if (dat->byte + obj_size > oldpos + size)
  {
    LOG_ERROR("Object size %u overflow at pos 0x%lx", obj_size,
              (unsigned long)(obj->address));
    obj_size = (oldpos + size) - dat->byte;
    error |= DWG_ERR_VALUEOUTOFBOUNDS;
  }
if (dat->byte + obj_size < oldpos + size)
  {
    BITCODE_RL offset = (oldpos + size) - (dat->byte + obj_size);
    LOG_TRACE("Unknown data of %u bytes at end of object\n", offset)
    if (offset > 0 && offset < 0x100000)
      {
        obj->unknown_rest = (unsigned char*)calloc(offset, 1);
        if (obj->unknown_rest)
          {
            memcpy(obj->unknown_rest, &dat->chain[dat->byte + obj_size], offset);
            obj->num_unknown_rest = offset;
          }
      }
  }
dat->byte += obj_size;
