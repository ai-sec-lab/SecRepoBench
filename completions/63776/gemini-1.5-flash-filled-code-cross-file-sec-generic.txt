BITCODE_RL unknown_data_size;
                if (dat->version < R_11)
                  {
                    unknown_data_size = (oldpos + obj->size) - dat->byte;
                    if (unknown_data_size > 0)
                      {
                        obj->unknown_data = malloc (unknown_data_size);
                        if (!obj->unknown_data)
                          {
                            LOG_ERROR ("Out of memory");
                            return DWG_ERR_OUTOFMEM;
                          }
                        memcpy (obj->unknown_data, &dat->byte,
                                unknown_data_size);
                        dat->byte += unknown_data_size;
                        obj->unknown_data_size = unknown_data_size;
                      }
                  }
                else
                  {
                    BITCODE_RC crc;
                    BITCODE_RL expected_size;
                    expected_size = obj->size;
                    if (obj->size > (dat->size - dat->byte))
                      {
                        LOG_ERROR (
                            "Object size exceeds buffer size: %" PRIuSIZE
                            " > %" PRIuSIZE,
                            obj->size, dat->size - dat->byte);
                        return DWG_ERR_INVALIDDWG;
                      }
                    crc = bit_read_RC (dat);
                    if (crc != obj->crc)
                      {
                        LOG_ERROR ("CRC mismatch: expected 0x%x, got 0x%x",
                                   obj->crc, crc);
                        return DWG_ERR_INVALIDDWG;
                      }
                    unknown_data_size = (oldpos + expected_size) - dat->byte;
                    if (unknown_data_size > 0)
                      {
                        obj->unknown_data = malloc (unknown_data_size);
                        if (!obj->unknown_data)
                          {
                            LOG_ERROR ("Out of memory");
                            return DWG_ERR_OUTOFMEM;
                          }
                        memcpy (obj->unknown_data, &dat->byte,
                                unknown_data_size);
                        dat->byte += unknown_data_size;
                        obj->unknown_data_size = unknown_data_size;
                      }
