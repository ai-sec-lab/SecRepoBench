{
  obj->size = bit_read_RL(dat);
  LOG_TRACE("size: " FORMAT_RL "\n", obj->size);
  if (obj->size > size || obj->size > dat->size - dat->byte)
    {
      LOG_ERROR("Invalid object size %" PRIuSIZE " > %" PRIuSIZE,
                obj->size, size);
      return DWG_ERR_INVALIDDWG;
    }
  if (dat->byte + obj->size > oldpos + size)
    {
      LOG_ERROR("Object size exceeds section size");
      return DWG_ERR_INVALIDDWG;
    }
  if (dat->byte + obj->size < oldpos + size)
    {
      size_t unknown_size = oldpos + size - (dat->byte + obj->size);
      LOG_TRACE("Unknown data size: %" PRIuSIZE "\n", unknown_size);
      obj->unknown_bits = (uint8_t *)malloc(unknown_size);
      if (!obj->unknown_bits)
        {
          LOG_ERROR("Out of memory for unknown data");
          return DWG_ERR_OUTOFMEM;
        }
      memcpy(obj->unknown_bits, dat->chain + dat->byte + obj->size, unknown_size);
      obj->num_unknown_bits = unknown_size * 8;
    }
  dat->byte += obj->size;
}
