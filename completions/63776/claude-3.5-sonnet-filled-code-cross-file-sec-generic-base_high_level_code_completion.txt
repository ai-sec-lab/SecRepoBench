BITCODE_RL obj_size = bit_read_RL(dat);
if (obj_size > (end - dat->byte) || (dat->byte + obj_size) > dat->size)
{
  LOG_ERROR("Invalid object size %u at pos %" PRIuSIZE ", remaining %" PRIuSIZE,
            obj_size, dat->byte, end - dat->byte);
  obj_size = (BITCODE_RL)(end - dat->byte);
  error |= DWG_ERR_VALUEOUTOFBOUNDS;
}
obj->size = obj_size;

SINCE(R_11)
{
  BITCODE_RS crc;
  dat->byte += obj_size - 2;
  crc = bit_read_RS(dat);
  LOG_TRACE("crc: %04X [RSx]\n", crc);
  dat->byte -= obj_size;
}

if (dat->byte + obj_size < oldpos + size)
{
  BITCODE_RL unknown_size = (oldpos + size) - (dat->byte + obj_size);
  LOG_TRACE("unknown: %d byte%s\n", unknown_size, unknown_size > 1 ? "s" : "")
  bit_read_fixed(dat, &obj->unknown_bits, unknown_size);
  obj->num_unknown_bits = unknown_size * 8;
  LOG_TRACE_TF(obj->unknown_bits, unknown_size);
  LOG_TRACE("\n");
}

dat->byte = oldpos + size;
