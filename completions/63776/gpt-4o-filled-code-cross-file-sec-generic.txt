{
  obj->size = bit_read_RL(dat);
  LOG_TRACE("size: " FORMAT_RL "\n", obj->size);
  if (obj->size > dat->size - dat->byte)
  {
    LOG_ERROR("Object size %u exceeds available data %zu", obj->size, dat->size - dat->byte);
    return DWG_ERR_INVALIDDWG;
  }
  if (obj->size != (dat->byte - obj->address))
  {
    size_t unknown_size = obj->size - (dat->byte - obj->address);
    if (unknown_size > 0)
    {
      obj->unknown_bits = (uint8_t *)malloc(unknown_size);
      if (!obj->unknown_bits)
      {
        LOG_ERROR("Out of memory for unknown bits");
        return DWG_ERR_OUTOFMEM;
      }
      memcpy(obj->unknown_bits, dat->chain + dat->byte, unknown_size);
      dat->byte += unknown_size;
      LOG_TRACE("Extracted unknown data of size %zu\n", unknown_size);
    }
  }
  else
  {
    obj->unknown_bits = NULL;
  }
}
