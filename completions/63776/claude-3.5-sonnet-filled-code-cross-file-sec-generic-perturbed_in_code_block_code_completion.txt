PRE (R_11)
{
  obj->size = bit_read_RS (dat);
  if (obj->size > size)
    {
      LOG_ERROR ("Invalid object size %d > %d", (int)obj->size, (int)size);
      obj->size = size;
    }
}
VERSIONS (R_11, R_13)
{
  obj->size = bit_read_RL (dat);
  if (obj->size > size)
    {
      LOG_ERROR ("Invalid object size %d > %d", (int)obj->size, (int)size);
      obj->size = size;
    }
}
if (obj->size + obj->address > oldpos + size)
  {
    LOG_ERROR ("Invalid object size %d > %d", (int)obj->size,
               (int)(oldpos + size - obj->address));
    obj->size = oldpos + size - obj->address;
  }
if (dat->byte + obj->size > oldpos + size)
  {
    LOG_ERROR ("Invalid object size %d > %d", (int)obj->size,
               (int)(oldpos + size - dat->byte));
    obj->size = oldpos + size - dat->byte;
  }
if (obj->size < 2)
  {
    LOG_ERROR ("Invalid object size %d < 2", (int)obj->size);
    obj->size = 2;
  }
if (dat->byte + obj->size != oldpos + obj->size)
  {
    unsigned long offset = obj->size - (dat->byte - oldpos);
    LOG_TRACE ("-unknown (%lu): ", offset);
    LOG_TRACE_TF (dat->chain + dat->byte, offset);
    LOG_TRACE ("\n");
    bit_advance_position (dat, offset * 8);
  }
SINCE (R_11)
{
  BITCODE_RS crc;
  bit_set_position (dat, oldpos + obj->size - 2);
  crc = bit_read_RS (dat);
  LOG_TRACE ("crc: %04X [RS]\n", crc);
  bit_set_position (dat, oldpos);
}
