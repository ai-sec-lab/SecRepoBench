BITCODE_RL obj_size = bit_read_RL (dat);
LOG_TRACE (" size: " FORMAT_RL " [RL]", obj_size);
if (obj_size > size || obj_size < 2)
  {
    LOG_ERROR ("Invalid object size " FORMAT_RL " > %u or < 2", obj_size, size);
    obj_size = size;
    error |= DWG_ERR_VALUEOUTOFBOUNDS;
  }
obj->size = obj_size;
if (dat->byte + obj_size > oldpos + size)
  {
    LOG_ERROR ("Invalid object size " FORMAT_RL " > %u", obj_size,
               (unsigned)(size - (dat->byte - oldpos)));
    obj_size = size - (dat->byte - oldpos);
    error |= DWG_ERR_VALUEOUTOFBOUNDS;
  }
if (dat->version >= R_11)
  {
    BITCODE_RS crc;
    dat->byte += obj_size - 2;
    crc = bit_read_RS (dat);
    LOG_TRACE (", crc: %04X", crc);
    dat->byte = oldpos;
  }
LOG_TRACE ("\n");
if (obj->size + 2 < (long)(dat->byte - oldpos))
  {
    long offset = obj->size + 2 - (dat->byte - oldpos);
    LOG_TRACE ("Unknown (%ld):\n", offset);
    LOG_TRACE_TF (dat->chain + dat->byte, offset);
    bit_advance_position (dat, offset * 8);
  }
if (obj->size > dat->byte - obj->address)
  {
    long offset = obj->size - (dat->byte - obj->address);
    obj->unknown_rest = (unsigned char *)calloc (offset, 1);
    if (obj->unknown_rest)
      {
        memcpy (obj->unknown_rest, &dat->chain[dat->byte], offset);
        obj->num_unknown_rest = offset;
        LOG_TRACE ("unknown_rest [%ld]:\n", offset);
        LOG_TRACE_TF (obj->unknown_rest, offset);
      }
    dat->byte += offset;
    LOG_TRACE ("\n");
  }
