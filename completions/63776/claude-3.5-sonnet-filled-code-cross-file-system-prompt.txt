PRE (R_11)
{
  obj->size = bit_read_RS (dat);
  if (obj->size > size)
    {
      LOG_ERROR ("Invalid object size %d > %d", (int)obj->size, (int)size);
      obj->size = size;
    }
  if (obj->size < 2)
    {
      LOG_ERROR ("Invalid object size %d < 2", (int)obj->size);
      obj->size = 2;
    }
}
VERSIONS (R_11, R_13)
{
  obj->size = bit_read_RL (dat);
  if (obj->size > size)
    {
      LOG_ERROR ("Invalid object size %d > %d", (int)obj->size, (int)size);
      obj->size = size;
    }
  if (obj->size < 4)
    {
      LOG_ERROR ("Invalid object size %d < 4", (int)obj->size);
      obj->size = 4;
    }
}

if (obj->size + obj->address > dat->size)
  {
    LOG_ERROR ("Invalid object size %d, overflow", (int)obj->size);
    obj->size = dat->size - obj->address;
    error |= DWG_ERR_VALUEOUTOFBOUNDS;
  }
if (obj->address + obj->size != dat->byte)
  {
    long off = obj->address + obj->size - dat->byte;
    LOG_TRACE ("Unknown data of %ld bytes:\n", off);
    bit_print_bits (dat, off);
    bit_advance_position (dat, off * 8);
    LOG_TRACE ("\n");
  }
