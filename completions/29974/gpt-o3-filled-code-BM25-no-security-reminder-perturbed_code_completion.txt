                                   offset + 4, 4, rightedge,
                                   "right edge = %u%s", rightedge,
                                   (tcp_analyze_seq && tcp_relative_seq) ? " (relative)" : "");

        /* Add info for the Info column */
        tcp_info_append_uint(pinfo, "SACK_LE", leftedge);
        tcp_info_append_uint(pinfo, "SACK_RE", rightedge);

        /* Count this SACK range */
        sack_range_count++;

        /* Save SACK range for bytes-in-flight calculations */
        if (tcp_analyze_seq && tcp_track_bytes_in_flight && tcpd &&
            tcpd->fwd && tcpd->fwd->tcp_analyze_seq_info &&
            tcpd->fwd->tcp_analyze_seq_info->num_sack_ranges < MAX_TCP_SACK_RANGES) {

            guint idx = tcpd->fwd->tcp_analyze_seq_info->num_sack_ranges;
            tcpd->fwd->tcp_analyze_seq_info->sack_left_edge[idx]  = leftedge;
            tcpd->fwd->tcp_analyze_seq_info->sack_right_edge[idx] = rightedge;
            tcpd->fwd->tcp_analyze_seq_info->num_sack_ranges++;
