/* The packet length looks consistent.  Check a couple of well-known
	 PPStream signature bytes before declaring the flow as PPStream.        */
      if(packet->payload_packet_len > 2 &&
	 (packet->payload[2] == 0x43 ||   /* 'C' */
	  packet->payload[2] == 0x53)) { /* 'S' */
	/* increase count pkt ppstream over udp */
	stream->l4.udp.ppstream_stage++;

	ndpi_int_ppstream_add_connection(ndpi_struct, stream);
	return;
