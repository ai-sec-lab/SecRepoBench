case TIFF_SHORT:
{
  uint16
    shorty;

  if (exif_info[i].variable_length == 0)
  {
    shorty=0;
    if (TIFFGetField(tiff,exif_info[i].tag,&shorty,sans) == 1)
      (void) FormatLocaleString(value,MagickPathExtent,"%d",shorty);
  }
  else
  {
    uint16
      *array;

    if (TIFFGetField(tiff,exif_info[i].tag,&count,&array) == 1)
      if (count > 0)
        (void) FormatLocaleString(value,MagickPathExtent,"%d",array[0]);
  }
  break;
}
