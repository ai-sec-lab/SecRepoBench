case TIFF_SHORT:
{
  uint16
    shorty;

  shorty=0;
  if (TIFFGetField(tiff,exif_info[i].tag,&shorty,sans) == 1)
  {
    if (exif_info[i].length == 1)
      (void) FormatLocaleString(value,MagickPathExtent,"%d",shorty);
    else
    {
      uint16
        *shorts;

      shorts=(uint16 *) NULL;
      if (TIFFGetField(tiff,exif_info[i].tag,&shorts,sans) == 1 && shorts != (uint16 *) NULL)
        (void) FormatLocaleString(value,MagickPathExtent,"%d",shorts[0]);
    }
  }
  break;
}
