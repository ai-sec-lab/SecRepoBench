case TIFF_SHORT:
{
  uint16
    shorty;

  shorty=0;
  if (TIFFGetField(tiff,exif_info[i].tag,&shorty,sans) == 1)
  {
    if (exif_info[i].length == 1)
      (void) FormatLocaleString(value,MagickPathExtent,"%d",shorty);
    else
    {
      *value='\0';
      for (ssize_t j=0; j < exif_info[i].length; j++)
      {
        if (j > 0)
          (void) ConcatenateString(&value,",");
        (void) FormatLocaleString(value+strlen(value),MagickPathExtent-strlen(value),"%d",shorty);
      }
    }
  }
  break;
}
