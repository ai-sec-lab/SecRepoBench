case TIFF_SHORT:
{
  uint16
    shorty;

  if (exif_info[i].variable_length == 0)
  {
    shorty=0;
    if (TIFFGetField(tiff,exif_info[i].tag,&shorty,sans) == 1)
      (void) FormatLocaleString(value,MagickPathExtent,"%d",shorty);
  }
  else
  {
    uint16
      *shorts;

    shorts=(uint16 *) NULL;
    if ((TIFFGetField(tiff,exif_info[i].tag,&count,&shorts) == 1) &&
        (shorts != (uint16 *) NULL) && (count > 0))
      (void) FormatLocaleString(value,MagickPathExtent,"%d",shorts[0]);
  }
  break;
}
