case TIFF_SHORT:
      {
        uint16
          shorty;

        shorty=0;
        if (TIFFGetField(tiff,exif_info[i].tag,&shorty,sans) == 1)
          (void) FormatLocaleString(value,MagickPathExtent,"%d",shorty);
        else
          if (TIFFGetField(tiff,exif_info[i].tag,&shorty,sans) == 1)
            {
              register ssize_t
                j;

              for (j=0; j < (ssize_t) shorty; j++)
              {
                uint16
                  shorts[MagickPathExtent];

                if (TIFFGetField(tiff,exif_info[i].tag,shorts,sans) != 1)
                  break;
                if (j == 0)
                  (void) FormatLocaleString(value,MagickPathExtent,"%d",
                    shorts[0]);
              }
            }
        break;
      }
