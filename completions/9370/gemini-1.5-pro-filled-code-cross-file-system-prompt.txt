case TIFF_SHORT:
      {
        uint16
          shorty;

        if (exif_info[i].variable_length == 0)
          {
            shorty=0;
            if (TIFFGetField(tiff,exif_info[i].tag,&shorty,sans) == 1)
              (void) FormatLocaleString(value,MagickPathExtent,"%d",shorty);
          }
        else
          {
            uint16
              *short_ptr;

            short_ptr=0;
            if ((TIFFGetField(tiff,exif_info[i].tag,&short_ptr,sans) == 1) &&
                (short_ptr != (uint16 *) NULL))
              (void) FormatLocaleString(value,MagickPathExtent,"%d",*short_ptr);
          }
        break;
      }
