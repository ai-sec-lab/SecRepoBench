case TIFF_SHORT:
      {
        uint16
          shorty;

        if (exif_info[i].length == 0)
          {
            shorty=0;
            if (TIFFGetField(tiff,exif_info[i].tag,&shorty,sans) == 1)
              (void) FormatLocaleString(value,MagickPathExtent,"%d",shorty);
          }
        else
          {
            uint16
              *short_array;

            short_array=(uint16 *) NULL;
            if ((TIFFGetField(tiff,exif_info[i].tag,&short_array,sans) == 1) &&
                (short_array != (uint16 *) NULL))
              (void) FormatLocaleString(value,MagickPathExtent,"%d",short_array[0]);
          }
        break;
      }
